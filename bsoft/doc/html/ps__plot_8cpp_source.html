501  double y_step = (y2-y1)*increment/(max - min);
502 
503  if ( digits < 0 ) digits = 0;

511  }
512  if ( integer_size < 1 ) integer_size = 1;
513  integer_size++; // Add one for the sign
514

516  double x_offset = x1 - number_size*fontsize/5;
517  double y_offset = y1 - fontsize/3;
518  if ( x2 - x1 > y2 - y1 )
519  y_offset -= fontsize;
520  else
521  x_offset -= number_size*fontsize/3;
522 
523  if ( tick_angle > 0.75*M_PI && tick_angle < 1.25*M_PI )
524  x_offset = x1 + fontsize/5;

529  }
530

532  *fps << "/Helvetica findfont " << fontsize << " scalefont setfont " << endl;

534  *fps << fixed << setprecision(2) <<
535  x1+i*x_step << " " << y1+i*y_step << " moveto " <<

541  << fixed << setprecision(digits) << right << setw(number_size) << showval << ") (";
542  for ( j=0; j<integer_size; j++ ) *fps << "x";
543  *fps << ".";
544  for ( j=0; j<digits; j++ ) *fps << "x";
545  *fps << ") cvs show" << endl;

549  fps->flags(fl);
