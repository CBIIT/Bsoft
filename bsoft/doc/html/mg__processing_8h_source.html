483/************************************************************************
484@Object: class Bproject
485@Description:
486 Project parameter structure.
487@Features:
488 A project is defined as a set of micrographs taken of a single
489 specimen and the associated reconstructions.
490 A specimen is understood to be the result of a particular
491 preparation method.
492 This structure is the top level in a hierarchy describing
493 micrographs and relationships between them.
494 Orientation parameters can be given as view vector and rotation angle
495 around the view vector (default).
496 Alternatively, orientation can be described by Euler angles.
497 Only one orientation representation is allowed to avoid ambiguity,
498 and the choice is indicated by the "Euler" flag.
499 Older software relying on the negative of the third Euler angle, -psi
500 or omega, can make use of the omega flag.
501 Multiple FOM's are allowed and the tags provide program-specific
502 descriptions of the FOM's in the particle structure
503*************************************************************************/
504class Bproject {
505private:
506 void initialize() {
507 next = NULL;
508 split = 0;
509 filename = 0;
510 comment = 0;
511 reference = NULL;
512 select = 0;
513 euler_flag = 0;
514 omega_flag = 0;
515 field = NULL;
516 rec = NULL;
517 class_avg = NULL;
518 for ( int f=0; f<NFOM; ++f ) fom_tag[f] = NoFOM;
519 fom_tag[0] = FOM;
520 }
521public:
522 Bproject* next; // Pointer to the next structure in the list
523 Bstring filename; // Parameter file name
524 Bstring comment; // Variable length comment field
525 int split; // Split code, 0=no split, 1-4=number of digits in file name, 9=use mg id's
526 Bstring* reference; // Linked list of reference map file names
527 int select; // Selects micrographs (0) or reconstructions (1)
528 int euler_flag; // Set if Euler angles are used
529 int omega_flag; // Set if omega is used in stead of psi
530 FOMType fom_tag[NFOM]; // STAR tags for up to NFOM FOM values
531 Bfield* field; // First field-of-view in list
532 Breconstruction* rec; // First reconstruction in list
533 Bparticle* class_avg; // 2D or 3D particle classes
534 Bproject() { initialize(); }
535} ;
