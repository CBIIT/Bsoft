181 with the angles between the links */
182 Vector3<double> component_normal(Bcomponent* comp)
183 {
184  Vector3<double> normal;
185 
186  if ( !comp->link[1] ) return normal;
187 
188  long i, j;
189  Vector3<double> v1, v2;
190 
191  for ( i=1; i<comp->link.size(); i++ ) {
192  for ( j=0; j<i; j++ ) {
193  v1 = comp->link[i]->location() - comp->location();
194  v2 = comp->link[j]->location() - comp->location();
195  v1 = v1.cross(v2);
196  if ( comp->location().angle(v1) > M_PI_2 ) v1 = -v1;
197  v1.normalize();
198  normal += v1;
199  }
200  }
201  if ( normal.length2() > 1e-10 ) normal.normalize();
202 
203  return normal;
204 }

209 View2<float> component_view(Bcomponent* comp)
210 {
211  Vector3<double> w(comp->view().vector3());
212 
213  if ( comp->link[1] ) w = component_normal(comp);
214 
215  if ( w.length2() < 0.1 ) w = comp->location();
216 
217  long i, im;
218  Vector3<double> v, u;
219  double a(0);
220 
221  for ( i=1, im=0; i<comp->link.size() && comp->link[i]; i++ ) {
222  v = comp->location().normal(comp->link[im]->location(), comp->link[i]->location());
223  a = w.angle(v);
224  if ( a > M_PI_2 ) im = i;
225  }
226 
227  w.normalize();
228  u = comp->link[im]->location() - comp->location();
229  u.normalize();
230  v = w.cross(u);
231  v.normalize();
232  u = v.cross(w);
233  a = angle_set_negPI_to_PI(atan2(w[1], w[0]) + atan2(v[2], -u[2]));
234 
235  View2<float> view(w[0], w[1], w[2], a);
236
