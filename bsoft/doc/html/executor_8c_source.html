35 void fftw_strided_copy(int n, fftw_complex *in, int ostride,
36  fftw_complex *out)
37 {
38  int i;
39  fftw_real r0, r1, i0, i1;
40  fftw_real r2, r3, i2, i3;
41 
42  i = 0;
43 
44  for (; i < (n & 3); ++i) {
45  out[i * ostride] = in[i];
46  }
47 
48  for (; i < n; i += 4) {
49  r0 = c_re(in[i]);
50  i0 = c_im(in[i]);
51  r1 = c_re(in[i + 1]);
52  i1 = c_im(in[i + 1]);
53  r2 = c_re(in[i + 2]);
54  i2 = c_im(in[i + 2]);
55  r3 = c_re(in[i + 3]);
56  i3 = c_im(in[i + 3]);
57  c_re(out[i * ostride]) = r0;
58  c_im(out[i * ostride]) = i0;
59  c_re(out[(i + 1) * ostride]) = r1;
60  c_im(out[(i + 1) * ostride]) = i1;
61  c_re(out[(i + 2) * ostride]) = r2;
62  c_im(out[(i + 2) * ostride]) = i2;
63  c_re(out[(i + 3) * ostride]) = r3;
64  c_im(out[(i + 3) * ostride]) = i3;
65  }
66 }

402 void fftw(fftw_plan plan, int howmany, fftw_complex *in, int istride,
403  int idist, fftw_complex *out, int ostride, int odist)
404 {
405  int n = plan->n;
406 
407  if (plan->flags & FFTW_IN_PLACE) {
408  if (howmany == 1) {
409  executor_simple_inplace(n, in, out, plan->root, istride,
410  plan->recurse_kind);
411  } else {
412  executor_many_inplace(n, in, out, plan->root, istride, howmany,
413  idist, plan->recurse_kind);
414  }
415  } else {
416  if (howmany == 1) {
417  fftw_executor_simple(n, in, out, plan->root, istride, ostride,
418  plan->recurse_kind);
419  } else {
420 #ifdef FFTW_ENABLE_VECTOR_RECURSE
421  int vector_size = plan->vector_size;
422  if (vector_size <= 1)
423 #endif
424  executor_many(n, in, out, plan->root, istride, ostride,
425  howmany, idist, odist, plan->recurse_kind);
426 #ifdef FFTW_ENABLE_VECTOR_RECURSE
427  else {
428  int s;
429  int num_vects = howmany / vector_size;
430  fftw_plan_node *root = plan->root;
431 
432  for (s = 0; s < num_vects; ++s)
433  executor_many_vector(n,
434  in + s * (vector_size * idist),
435  out + s * (vector_size * odist),
436  root,
437  istride, ostride,
438  vector_size, idist, odist);
439 
440  s = howmany % vector_size;
441  if (s > 0)
442  executor_many(n,
443  in + num_vects * (vector_size * idist),
444  out + num_vects * (vector_size * odist),
445  root,
446  istride, ostride,
447  s, idist, odist,
448  FFTW_NORMAL_RECURSE);
449  }
450 #endif
451  }
452  }
