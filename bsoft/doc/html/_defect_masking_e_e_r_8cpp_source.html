168 unsigned DefectElectronAdder::execute(ElectronPos* pListPtr, unsigned nElect, const CameraDefectNeighborInformation& camDefectNeighborInfo)
169 {
170  defectCounts.clear();
171  // visit all electrons and add to the map if they are direct neighbors
172  for (unsigned i=0; i<nElect;++i)
173  {
174  uint32_t eOfs = ((pListPtr[i].y >> nSubPixBits) << 12) | (pListPtr[i].x >> nSubPixBits);
175  uint32_t eOfsGain = ((pListPtr[i].y / gainImageSizeFactor) * gainImageSize) | (pListPtr[i].x /gainImageSizeFactor);
176  defectNeighborInfo_t di = camDefectNeighborInfo.neighborSpec[eOfs];
177  if (di==1)
178  {
179  // electron at defect!! make it harmless
180  pListPtr[i].x = 0xffff;
181  pListPtr[i].y = 0xffff;
182  }
183  if (di & bitMask_DefectNeighborTypeMask)
184  {
185  uint16_t subPix = distSubPix(generator);
186  pListPtr[i].x = (pListPtr[i].x & 0xfffc) | (subPix&3);
187  pListPtr[i].y = (pListPtr[i].y & 0xfffc) | (subPix>>2);
188  float pcGain = camDefectNeighborInfo.gainImage[eOfsGain];
189  //std::cout<<"pcGain"<<pcGain<<std::endl;
190  defectNeighborInfo_t defDist = (di & bitMask_Distance)+1;
191  int32_t defOfs = (di & bitMask_DefectUpDown)? 4096:1;
192  defOfs = (di&bitMask_DefectBack)? -defOfs : +defOfs;
193  uint8_t mult = ((di & bitMask_DefectNeighborTypeMask) == bitMask_Defect2DInterpol)? 2 : 1;
194  for (uint8_t k=1; k<=defDist;++k)
195  {
196  defectCounts[eOfs + k*defOfs] += pcGain * ((float)((defDist-k+1)))/(mult*(defDist+1)); // unordered_map creates zero-initialized element on-the-fly if not there yet. so convenient, yet confusing
197  }
198  if ((di & bitMask_DefectNeighborTypeMask) == bitMask_DefectCorner && eOfs >= defOfs && eOfs < 4096*4096+defOfs)
199  {
200  defectNeighborInfo_t diOrt = camDefectNeighborInfo.neighborSpec[eOfs - defOfs];
201  // get the info of the horizontal defect. (by construction in CreateDefectNeighborInfoImage, the corner points now have the vertical one)
202  // note; gain should not be obtained from the nieghbor!
203  defectNeighborInfo_t defDistOrt = (diOrt & bitMask_Distance)+1;
204  int32_t defOfsOrt = (diOrt & bitMask_DefectUpDown)? 4096:1;
205  defOfsOrt = (diOrt & bitMask_DefectBack)? -defOfsOrt : +defOfsOrt;
206  for (uint8_t k=0; k<=defDist;++k)
207  {
208  for (uint8_t m=1; m<=defDistOrt;++m)
209  {
210  defectCounts[eOfs + k*defOfs + m*defOfsOrt] +=
211  pcGain * ((float)((defDist-k+1)))/(defDist+1) * ((float)((defDistOrt-m+1)))/(defDistOrt+1);
212  }
213  }
214  }
215  }
216  }
217 
218  // probalistic addition of electrons.
219  unsigned plc = nElect;
220  for (const auto &pair : defectCounts)
221  {
222  uint32_t defectOfs = pair.first;
223  float electProb = pair.second;
224  if (distHit(generator) < electProb)
225  {
226  uint16_t subPix = distSubPix(generator);
227  uint16_t posX = (((defectOfs & 4095) << nSubPixBits) + (subPix>>2));
228  uint16_t posY = (((defectOfs >> 12) << nSubPixBits) + (subPix&3));
229  //if ((defectOfs >> 12)!=2142)
230  // std::cerr<<"HUH "<<posY<<"; "<<(((defectOfs >> 12) << nSubPixBits) + (subPix>>2))<<" ; "<<subPix<<std::endl;
231  pListPtr[plc++] = ElectronPos(posX, posY); // What with subpixel info? random?
232  }
233  }
234  return plc;
235 }
