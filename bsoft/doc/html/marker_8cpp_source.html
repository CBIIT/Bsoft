796 /*
797 @author Samuel Payne
798 @brief Produces a match of two sets and estimate the transform parameters.
799 @param Bmarker* setA the first coordinate set.
800 @param Bmarker* setB the second coordinate set.
801 @param double dist_cutoff max difference allowed in matching d_rel variables
802 @return double* the transform parameters, the set error, and size.
803 
804  This finds the rough matching of two sets of points. These sets are
805  assumed to be sorted relative to their distance to a relative point.
806  It then computes a histogram of all the angular deviations. Taking
807  the largest bin as the true angle (not averaging all values as that
808  is subject to outliers). It removes all matches whose angular
809  deviations are further from the true angle than the angle tolerance
810  allows. Finally it calculates the x and y shift parameters, error and
811  set size.
812 **/
813 double* markers_RoughMatch(Bmarker* setA, Bmarker* setB, double dist_cutoff)
814 {
815  int a, i, j, bins = 360, max;
816  double best_phi=0, phi, ang_tol = M_PI/10, xshift, yshift, error = 1e37;
817  double scale = bins*1.0/(TWOPI);
818  double dx, dy, sumdx_sq, sumdy_sq, sumdx, sumdy;
819 
820  long nA = count_list((char *) setA);
821 // long nB = count_list((char *) setB);
822 
823  Bmarker *mA, *mB;
824  double d;
825 
826  Bmarker** match = new Bmarker*[nA];
827  int* histo = new int[bins];
828  double* params = new double[5];
829  double* phi_devs = new double[nA];
830 
831  /* The loop starts at a=1, b=1 because the first points have a zero
832  distances. They always match each other (a trivial solution). they
833  also have a zero angle, which is not characteristic of the set in
834  general. so it should be excluded. */
835  for ( a=1, mA=setA->next, mB=setB->next; mA && mB; mA=mA->next, mB=mB->next, a++ ) {
836 // for( a=1, b=1; a<nA && b<nB; a++, b++ ) {
837  /* These while loops find an 'a' and 'b' that match each other according to the
838  relative distance variable. while the next point of b matches better than the
839  current than advance the b counter. same for a. This strategy works because the
840  sets are sorted according to the d_rel variable (relative distance). If the difference
841  between the distance vars is less than a cutoff the match is accepted. */
842  d = mA->err.length2();
843  while ( mB->next &&
844  ( fabs(d - mB->err.length2()) > fabs(d - mB->next->err.length2()) ) ) {
845  mB = mB->next;
846  }
847  d = mB->err.length2();
848  while ( mA->next &&
849  ( fabs(mA->err.length2() - d) > fabs(mA->next->err.length2() - d)) ) {
850  mA = mA->next;
851  a++;
852  }
853  if ( fabs(mA->err.length2() - mB->err.length2()) < dist_cutoff ) {
854  /* Here a histogram of the angular deviations is set up. There are 360
855  bins which are meant to correspond to the 360 degrees in a unit circle.
856  The angles are shifted between 0 and 2PIs so they can be scaled into a
857  bin easily. The additional 0.5 shift before casting to int ensures that
858  values slightly under a bin are put in bin they are closest to. In the
859  bottom else j==bin, therefore j = 360, which corresponds to 360 degrees,
860  the same as 0 degrees. therefore it is put into histo[0]. */
861  match[a] = mB;
862  phi = angle_set_negPI_to_PI(mA->fom - mB->fom);
863  phi_devs[a] = phi;
864  j = (int) (scale*(phi)+ 0.5);
865  if ( j < bins ) histo[j]++;
866  else histo[0]++;
867  }
868  }
869 
870  //find the Max bin value. that is an initial estimate of the angle
871  for ( max=0, i=0; i<bins; i++ ) {
872  if ( histo[i] > max ) {
873  best_phi = i; //now holds the degree value of the angle
874  max = histo[i];
875  }
876  }
877  best_phi = best_phi / scale; //now holds the RADIAN value
878  xshift = yshift = j = 0;
879  for ( a=1, mA=setA; mA && a<nA; a++, mA=mA->next ) {
880  if ( match[a] ) {
881  phi = fabs(phi_devs[a] - best_phi);
882  if ( phi > M_PI ) phi -= TWOPI;
883  if ( fabs(phi) > ang_tol ) {
884  match[a] = NULL;
885  } else {
886  dx = match[a]->loc[0] - mA->loc[0]*cos(best_phi) - mA->loc[1]*sin(best_phi);
887  dy = match[a]->loc[1] + mA->loc[0]*sin(best_phi) - mA->loc[1]*cos(best_phi);
888  xshift += dx;
889  yshift += dy;
890  j++; //here j is acting as a count variable
891  }
892  }
893  }
894  xshift = xshift/j;
895  yshift = yshift/j;
896  sumdx = sumdy = sumdx_sq = sumdy_sq = j = 0;
897 
898  for ( a=1, mA=setA; mA && a<nA; a++, mA=mA->next ) {
899  if ( match[a] ) {
900  dx = match[a]->loc[0] - mA->loc[0]*cos(best_phi) - mA->loc[1]*sin(best_phi) - xshift;
901  dy = match[a]->loc[1] + mA->loc[0]*sin(best_phi) - mA->loc[1]*cos(best_phi) - yshift;
902  sumdx += dx;
903  sumdy += dy;
904  sumdx_sq += dx*dx;
905  sumdy_sq += dy*dy;
906  j++; //here j is acting as a count var
907  }
908  }
909 
910  if ( j ) {
911  sumdx = sumdx/j;
912  sumdy = sumdy/j;
913  error = sqrt((sumdx_sq + sumdy_sq)/j - sumdx*sumdx - sumdy*sumdy);
914  if ( !isfinite(error) ) error = 1e37;
915  }
916 
917  params[0] = error;
918  params[1] = best_phi;
919  params[2] = xshift;
920  params[3] = yshift;
921  params[4] = j; //num_matches
922 
923  if ( verbose & VERB_DEBUG ) {
924  /*if ( error < 100 && j > nA/5 )*/
925  if ( max > 15 ){
926  cout << endl << "DEBUG markers_RoughMatch:" << endl;
927  cout << "error " << error << " angle " << best_phi << " xshift " <<
928  xshift << " yshift " << yshift << " count " << j << endl;
929  cout << "DEBUG markers_RoughMatch: angular deviation histogram bins" << endl;
930  for ( a=0; a<bins; a++ ) {
931  if ( histo[a] != 0 )
932  cout << "bin " << a << " value " << histo[a] << endl;
933  }
934  }
935  }
936 
937  delete[] histo;
938  delete[] match;
939  delete[] phi_devs;
940 
941  return params;
942 }
