683  } else if ( sym.point() < 500 ) { // Octahedral

687  dang = M_PI_2;
688  nstep = 4;
689  }

700

703  cout << "Axis\t\t\tAngle\tCC" << setprecision(4) << endl;
704  }

707  if ( sym.point() == 202 ) { // D2
708  switch ( step ) {
709  case 1: angle = M_PI_2; axis = Vector3<double>(0,0,1); break;
710  case 2: angle = M_PI_2; axis = Vector3<double>(0,1,0); break;
711  case 3: angle = M_PI_2; axis = Vector3<double>(1,0,0); break;
712  case 4: angle = M_PI/1.5L; axis = Vector3<double>(1,1,1); break;
713  case 5: angle = -M_PI/1.5L; axis = Vector3<double>(1,1,1); break;
714  default: angle = 0;
715  }
716  axis.normalize();
717  } else if ( sym.point() < 300 ) { // Dihedral
718  switch ( step ) {
719  case 1: angle = dang; break;
720  case 2: angle = M_PI_2; break;
721  case 3: angle = M_PI_2 - dang; break;
722  default: angle = 0;
723  }
724  axis.normalize();
725  } else if ( sym.point() == 532 ) { // Icosahedral
726  switch ( step ) {
727  case 1: angle = dang; break;
728  case 2: angle = atan(1/GOLDEN); axis = Vector3<double>(1,0,0); break;
729  case 3: q2 = Quaternion(axis, angle); q = q2*q;
730  axis = q.axis(); angle = q.angle(); break;
731  default: angle = 0;
732  }
733  axis.normalize();

736  }

741  ccbest = cc;
742  best_angle = angle;
743  best_axis = axis;
744 // if ( verbose & VERB_PROCESS )
745 // cout << "Better CC: " << cc << " (" << angle*180.0/M_PI << ")" << endl;
746  }
747  delete pt;
748  if ( verbose )
749  cout << axis << tab << angle*180.0/M_PI << tab << cc << endl;
750  }
751 
752  if ( verbose ) {
753  cout << "Best axis and angle: " << best_axis << tab << best_angle*180.0/M_PI << endl;
754  cout << "Best CC: " << ccbest << endl;
755  cout << "Fixing symmetry " << sym.label() << " orientation" << endl << endl;
756  }
757 
758  mat = Matrix3(best_axis, best_angle);
759
