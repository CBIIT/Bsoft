13/************************************************************************
14@Object: class Bsymop
15@Description:
16 General symmetry operator.
17@Features:
18 Both point groups and space groups are covered.
19 A symmetry operator describes the complete series of rotations around
20 an axis, with the number of operations given by the order field
21 and the rotation angle given by the angle field. The shift field
22 is used in gliding operations, such as needed for helical symmetry.
23*************************************************************************/
24class Bsymop {
25private:
26 Vector3<double> ax; // Symmetry axis
27 int n; // Number of times the operation is applied, 1=identity
28 double ang; // Rotation angle, derived from order when 0
29 double sh; // Shift up symmetry axis for helix
30 void initialize() {
31 n = 0;
32 ang = 0;
33 sh = 0;
34 }
35public:
36 Bsymop() { initialize(); }
37 Bsymop(double x, double y, double z, int nn, double a) :
38 ax(Vector3<double>(x,y,z)), n(nn), ang(a), sh(0) {}
39 void order(int i) { n = i; }
40 int order() { return n; }
41 void angle(double a) { ang = a; }
42 double angle() { return ang; }
43 void shift(double s) { sh = s; }
44 double shift() { return sh; }
45 void axis(Vector3<double> a) { ax = a; }
46 void axis(double x, double y, double z) { ax = Vector3<double>(x,y,z); }
47 Vector3<double>& axis() { return ax; }
48 double normalize() { return ax.normalize(); }
49 Matrix3 matrix() {
50 return Matrix3(ax, M_PI*2.0/n);
51 }
52} ;
