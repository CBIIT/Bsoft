239  for ( i=1; i<n; i++ ) {
240  for ( j=0; j<i; j++ ) {
241  new_mat[i][j] = matrix[order[i]][order[j]];
242  if ( min > new_mat[i][j] ) min = new_mat[i][j];
243  if ( max < new_mat[i][j] ) max = new_mat[i][j];
244  }
245  }
246

253  for ( j=i+2; j<n; j++ ) {

255  new_mat[i][j] += new_mat[k][k-1];
256  if ( new_min > new_mat[i][j] ) new_min = new_mat[i][j];
257  if ( new_max < new_mat[i][j] ) new_max = new_mat[i][j];
258  }
259  }
260
