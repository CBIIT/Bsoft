565 
566  if ( sb ) {
567  t[0] = v[2]; // Swap byte order
568  t[1] = v[3];
569  t[2] = v[0];

573  t[1] = v[0];

576  }
577  for ( i=0; i<4; i++ ) v[i] = t[i];
578

580  if ( expnt == 0 )
581  if ( v[0] == 0 ) // Special case zero
582  memset(t, '\0', 4);
583  else { // Special case infinity or NaN
584  t[0] = 0xff;
585  t[1] = v[1] | 0x80;
586  }
587  else if ( expnt > 2 ) // Normalized value
588  t[0] = v[0] - 0x01; // Subtract 2 from the exponent
589  else { // Denormalized value
590  shift = 3 - expnt;
591  t[0] = v[0] & 0x80; // Keep sign, expnt = 0
592  t[1] = ( (v[1] & 0x7f) >> shift ) | ( 0x10 << expnt );
593  t[2] = ( v[1] << ( 8-shift ) ) | ( v[2] >> shift );
594  t[3] = ( v[2] << ( 8-shift ) ) | ( v[3] >> shift );
595  }
