153 string read(ifstream& fstar) {
154 string s, t;
155 while ( !fstar.eof() && !regex_search(s, regex("^data_")) ) {
156 s = get_clean_line(fstar);
157// cout << "=" << s << "=" << endl;
158 if ( s.length() && s[0] != '#' ) {
159 if ( s[0] == '_' ) { // single item
160 size_t i(s.find_first_of(' '));
161 if ( i != string::npos ) {
162 t = s.substr(1, i-1); // Strip off the underscore
163 it[t] = quote_or_not(s.substr(i));
164 } else {
165 t = s.substr(1); // Strip off the underscore
166 }
167 } else if ( s[0] == ';' ) { // multiline
168 string v;
169 if ( s.length() > 1 ) v = s.substr(1);
170 while ( !fstar.eof() ) {
171 s = get_clean_line(fstar);
172 if ( s[0] == ';' ) break;
173 v += s;
174 };
175 it[t] = v;
176 } else if ( regex_search(s, regex("^loop_")) ) {
177 lp.push_back(BstarLoop(fstar));
178 }
179 }
180 }
181 return s;
182 }
