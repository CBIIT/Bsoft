587  long hx1 = (p->sizeX()+1)/2, hy1 = (p->sizeY()+1)/2, hz1 = (p->sizeZ()+1)/2;

595  z = ( l < 0 ) ? l + p->sizeZ() : l;
596  sz = l*frac_mat[2][2];
597  sz *= sz;
598  for ( k=-hy; k<hy1; k++ ) {
599  y = ( k < 0 ) ? k + p->sizeY() : k;
600  sy = k*frac_mat[1][1];
601  sy *= sy;
602  for ( h=-hx; h<=0; h++ ) {
603  x = ( h < 0 ) ? h + p->sizeX() : h;
604  sx = h*frac_mat[0][0];
605  sx *= sx;
606  s = sqrt(sx + sy + sz);
607  if ( s < smax ) {
608  i = p->index(x, y, z);
609  s *= inv_recip_int;
610  fom[i] = s;
611  n++;
612  }
613  }
614  }
615  }
616 
617  fom[0] = 1e-10; // Make F0 non-zero
618 
619 #ifdef HAVE_GCD
620  dispatch_apply(imgsize, dispatch_get_global_queue(0, 0), ^(size_t i){
621  if ( fom[i] ) one_sf(molgroup, p, i, fom[i], scale, scat);
622  });
623 #else
624 #pragma omp parallel for
625  for ( i=0; i<imgsize; i++ ) if ( fom[i] )
626  one_sf(molgroup, p, i, fom[i], scale, scat);
627 #endif
628 
629 // cout << "first half done" << endl;
630 
631  // Fill in the second half from the first
632  for ( l=-hz; l<hz1; l++ ) {
633  z = ( l < 0 ) ? l + p->sizeZ() : l;
634  zi = ( z == 0 ) ? 0 : p->sizeZ() - z;
635  sz = l*frac_mat[2][2];
636  sz *= sz;
637  for ( k=-hy; k<hy1; k++ ) {
638  y = ( k < 0 ) ? k + p->sizeY() : k;
639  yi = ( y == 0 ) ? 0 : p->sizeY() - y;
640  sy = k*frac_mat[1][1];
641  sy *= sy;
642  for ( h=1; h<hx1; h++ ) {
643  sx = h*frac_mat[0][0];
644  sx *= sx;
645  s = sqrt(sx + sy + sz);
646  if ( s < smax ) {
647  i = p->index(p->sizeX() - h, yi, zi);
648  j = p->index(h, y, z);
649  p->set(j, (p->complex(i)).conj());
650  n++;
651  }
652  }
653  }
654  }
655 
656  if ( verbose & VERB_PROCESS )
657  cout << "Reflections calculated: " << n << endl << endl;
658 
659  delete[] scat;
660 
661  return 0;
662 }
