318 /**************************************************************************
319 @author David Belnap
320 @brief Gets criteria for selecting micrographs (from a field-of-view) for a
321  handedness-determination calculation.
322 @param &tag "mgsel", "mg1", or "mg2".
323 @param &tagvalue string with one of options listed above.
324 @param *mg_select micrograph selection criteria, views 1 & 2.
325 @param *mg_index indices for micrograph selection, views 1 & 2.
326 @param *mg_ang angles for micrograph selection, views 1 & 2.
327 @return int 0.
328 
329  User has four options:
330  1. angle,a (tilt angle)
331  2. index,n (index within a series, beginning with 1)
332  3. near,n (nth nearest-to-focus, beginning with 1)
333  4. far,n (nth farthest-from-focus, beginning with 1)
334 
335 **/
336 int hand_get_options_mgselect(Bstring& tag, Bstring& tagvalue, int* mg_select, int* mg_index, double* mg_ang)
337 {
338  Bstring* strarr = tagvalue.split(",");
339 
340  if ( tag == "mgsel" ) {
341  if ( tagvalue.contains("ang") ) {
342  mg_select[0] = mg_select[1] = 4;
343  mg_ang[0] = strarr->next->real() * M_PI/180.0;
344  mg_ang[1] = strarr->next->next->real() * M_PI/180.0;
345  } else {
346  mg_index[0] = strarr->next->integer();
347  mg_index[1] = strarr->next->next->integer();
348  if ( tagvalue.contains("near") ) mg_select[0] = mg_select[1] = 0;
349  else if ( tagvalue.contains("far") ) mg_select[0] = mg_select[1] = 1;
350  else if ( tagvalue.contains("ind") ) mg_select[0] = mg_select[1] = 2;
351  }
352  } else if ( tag == "mg1" ) {
353  if ( tagvalue.contains("ang") ) {
354  mg_select[0] = 4;
355  mg_ang[0] = strarr->next->real() * M_PI/180.0;
356  } else {
357  mg_index[0] = strarr->next->integer();
358  if ( tagvalue.contains("near") ) mg_select[0] = 0;
359  else if ( tagvalue.contains("far") ) mg_select[0] = 1;
360  else if ( tagvalue.contains("ind") ) mg_select[0] = 2;
361  }
362  } else if ( tag == "mg2" ) {
363  if ( tagvalue.contains("ang") ) {
364  mg_select[1] = 4;
365  mg_ang[1] = strarr->next->real() * M_PI/180.0;
366  } else {
367  mg_index[1] = strarr->next->integer();
368  if ( tagvalue.contains("near") ) mg_select[1] = 0;
369  else if ( tagvalue.contains("far") ) mg_select[1] = 1;
370  else if ( tagvalue.contains("ind") ) mg_select[1] = 2;
371  }
372  }
373 
374  string_kill(strarr);
375 
376  return 0;
377 }
