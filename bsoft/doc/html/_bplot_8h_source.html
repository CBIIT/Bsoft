251 double cut(long icol, double threshold, int dir=1) {
252 long i, i1, j, j1;
253 double val((data[0]+data[1])/2);
254 for ( i=0, i1=1, j=icol*nr, j1=j+1; i1<nr; i++, i1++, j++, j1++ ) {
255 if ( dir < 0 ) { // decreasing threshold
256 if ( data[j] >= threshold && data[j1] < threshold )
257 val = data[i] + (data[i1] - data[i])*(threshold - data[j])/(data[j1] - data[j]);
258 } else { // increasing threshold
259 if ( data[j] <= threshold && data[j1] > threshold )
260 val = data[i] + (data[i1] - data[i])*(threshold - data[j])/(data[j1] - data[j]);
261 }
262 }
263 if ( val < data[1] ) {
264 if ( dir < 0 ) {
265 if ( data[j] > threshold ) val = data[i];
266 } else {
267 if ( data[j] < threshold ) val = data[i];
