121 proc updateSelection { } {
122  global tool theimg
123 # global px py
124 # global mx my
125  global selectiontype
126  global sx1 sx2 sy1 sy2 sz1 sz2
127 
128  set c [getImageCanvas $theimg]
129  $c delete selection
130 
131  if { [winfo exists .wsel] && $tool == "select" && $sx1 != $sx2 && $sy1 != $sy2 } {
132  set wc [getControlWindow $theimg]
133  set slice_num [$wc.slice.scale get]
134  if { $sz1 <= $slice_num && $sz2 >= $slice_num } {
135  set scale [$wc.scale.scale get]
136  set h [lindex [$c cget -scrollregion] 3]
137  if { $selectiontype == "Square" || $selectiontype == "Rectangle" || $selectiontype == "Cube" } {
138  set cx1 [expr int($sx1 * $scale)]
139  set cy1 [expr int($h - $sy1 * $scale) ]
140  set cx2 [expr int($sx2 * $scale)]
141  set cy2 [expr int($h - $sy2 * $scale) ]
142  $c create rectangle $cx1 $cy1 $cx2 $cy2 -tags selection -outline red
143  }
144  if { $selectiontype == "Circle" || $selectiontype == "Ellipse" || $selectiontype == "Sphere" } {
145  set t 0.0
146  set z [expr ($sz1 + $sz2)*0.5 - $slice_num]
147  if { $z != 0 } {
148  set r [expr ($sz1 - $sz2)*0.5]
149  set t [expr $z/$r]
150  }
151  set t [expr sqrt(1.0 - $t*$t)]
152  set tn [expr 0.5*(1.0 - $t)]
153  set tp [expr 0.5*(1.0 + $t)]
154  set cx1 [expr $sx1 * $scale]
155  set cy1 [expr $h - $sy1 * $scale ]
156  set cx2 [expr $sx2 * $scale]
157  set cy2 [expr $h - $sy2 * $scale ]
158  set ccx1 [expr int($tp*$cx1 + $tn*$cx2)]
159  set ccx2 [expr int($tn*$cx1 + $tp*$cx2)]
160  set ccy1 [expr int($tp*$cy1 + $tn*$cy2)]
161  set ccy2 [expr int($tn*$cy1 + $tp*$cy2)]
162  $c create oval $ccx1 $ccy1 $ccx2 $ccy2 -tags selection -outline red
163  }
164  }
165  }
166 }

171 proc updateSelectionStats { } {
172  global theimg
173  global selectiontype calc_stats
174  global sx1 sx2 sy1 sy2 sz1 sz2
175  if { ! $calc_stats } { return }
176 # set c [getImageCanvas $theimg]
177  set wc [getControlWindow $theimg]
178  set img_num [$wc.image.scale get]
179  set slice_num [$wc.slice.scale get]
180  set vlist "$sx1 $sy1 $sz1 $sx2 $sy2 $sz2"
181  if { $selectiontype == "Square" || $selectiontype == "Rectangle" || $selectiontype == "Cube" } {
182  set stats [Bimage stats $theimg $img_num 1 $vlist]
183  }
184  if { $selectiontype == "Circle" || $selectiontype == "Ellipse" || $selectiontype == "Sphere" } {
185  set stats [Bimage stats $theimg $img_num 2 $vlist]
186  }
187  if { $selectiontype == "Polygon" } {
188  set poly ""
189  append poly "$sx1 $sy1 $slice_num "
190  append poly "$sx1 $sy2 $slice_num "
191  append poly "$sx2 $sy2 $slice_num "
192  append poly "$sx2 $sy1 $slice_num "
193 # puts $poly
194  set stats [Bimage stats $theimg $img_num 3 $poly]
195  }
196  .wsel.num.value config -text [lindex $stats 0]
197  .wsel.sum.value config -text [lindex $stats 1]
198  .wsel.avg.value config -text [lindex $stats 2]
199  .wsel.std.value config -text [lindex $stats 3]
200 # puts "Expected size [expr (abs($sx1 - $sx2) + 1)*(abs($sy1 - $sy2) + 1)]"
