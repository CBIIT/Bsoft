105 /*
106  * TIFF Image File Directories are comprised of a table of field
107  * descriptors of the form shown below. The table is sorted in
108  * ascending order by tag. The values associated with each entry are
109  * disjoint and may appear anywhere in the file (so long as they are
110  * placed on a word boundary).
111  *
112  * If the value is 4 bytes or less, then it is placed in the offset
113  * field to save space. If the value is less than 4 bytes, it is
114  * left-justified in the offset field.
115  */
116 typedef struct {
117  uint16 tdir_tag; /* see below */
118  uint16 tdir_type; /* data type; see below */
119  uint32 tdir_count; /* number of items; length in spec */
120  uint32 tdir_offset; /* byte offset to field data */
121 } TIFFDirEntry;
