46/************************************************************************
47@Object: struct Bstar_block
48@Description:
49 Structure for a data block with multiple items in a STAR database.
50@Features:
51 Each data block contains a set set of unique items defined by tags
52 and with the associate data as single or multiple values.
53 The order of the items in the data block are important and preserved.
54*************************************************************************/
55struct Bstar_block {
56 Bstar_block* next; // Pointer to next block
57 int number; // Block number (classic)
58 Bstring tag; // Block identifier
59 Bstring comment; // Comment following tag-value pair
60 Bstring filename; // File containing this block
61 Bstar_item* item; // Pointer to items
62} ;

64/************************************************************************
65@Object: struct Bstar_old
66@Description:
67 Overall STAR structure to hold all data blocks and options for I/O..
68@Features:
69 The split flag allows the user to output data blocks in separte files
70 in stead of one big file.
71 The line length field allows the user to output long lines without
72 wrapping it around.
73 The comments are ignored but output to the a new file - this can be used
74 to document the history of the file.
75 The STAR database is a hierarchy consisting of blocks, each with a set
76 of items.
77*************************************************************************/
78struct Bstar_old {
79 int split; // Flag to indicate writing each data block in its own file
80 int line_length; // Length of lines in output file
81 Bstring comment; // List of comments before 1st data block
82 Bstar_block* block; // Pointer to first block
83} ;
