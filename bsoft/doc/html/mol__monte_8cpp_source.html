763 double molgroup_atom_overlap(Bmolgroup* molgroup, Bmd* md)
764 {
765  if ( md->Ksep <= 0 ) return 0;
766 
767  long n(0), i, x, y, z;
768  Vector3<double> box = molgroup->box;
769  Vector3<double> sampling(md->sepdist, md->sepdist, md->sepdist);
770  Vector3<int> size;
771  size[0] = (int) (box[0]/sampling[0] + 0.001);
772  size[1] = (int) (box[1]/sampling[1] + 0.001);
773  size[2] = (int) (box[2]/sampling[2] + 0.001);
774  size = size.max(1);
775  sampling[0] = box[0]/size[0] + 0.001;
776  sampling[1] = box[1]/size[1] + 0.001;
777  sampling[2] = box[2]/size[2] + 0.001;
778  long boxsize = (long) size.volume();
779 
780  Bmolgroup* mg;
781  Bmolecule* mol;
782  Bresidue* res;
783  Batom* atom;
784 
785  Bimage* p = new Bimage(Float, TSimple, size, 1);
786  float* data = (float *) p->data_pointer();
787  p->next = new Bimage(Float, TSimple, p->size(), p->images());
788  float* dataone = (float *) p->next->data_pointer();
789 
790  for ( mg = molgroup; mg; mg = mg->next ) {
791  for ( mol = mg->mol; mol; mol = mol->next ) {
792  for ( res = mol->res; res; res = res->next ) {
793  for ( atom = res->atom; atom; atom = atom->next ) {
794  x = (long) ((atom->coord[0] - mg->min[0])/sampling[0]);
795  y = (long) ((atom->coord[1] - mg->min[1])/sampling[1]);
796  z = (long) ((atom->coord[2] - mg->min[2])/sampling[2]);
797  if ( x>=0 && x<size[0] && y>=0 && y<size[1] && z>=0 && z<size[2] ) {
798  i = p->index(x, y, z);
799  dataone[i] = 1;
800  }
801  }
802  }
803  if ( !mg->select.contains("FILE") ) {
804  for ( i=0; i<boxsize; i++ ) {
805  data[i] += dataone[i];
806  dataone[i] = 0;
807  }
808  }
809  }
810  if ( mg->select.contains("FILE") ) {
811  for ( i=0; i<boxsize; i++ ) {
812  data[i] += dataone[i];
813  dataone[i] = 0;
814  }
815  }
816  }
817 
818  if ( verbose & VERB_DEBUG ) {
819  Bstring filename("overlap.map");
820  write_img(filename, p, 0);
821  }
822 
823  md->Esep = 0;
824 
825  for ( i=n=0; i<boxsize; i++ ) {
826  if ( data[i] ) {
827  n++;
828  if ( data[i] > 1 ) md->Esep += data[i] - 1;
829  }
830  }
831 
832  md->Esep *= md->Ksep * md->sepdist/n;
833 
834  delete p;
835 
836  return md->Esep;
837 }
