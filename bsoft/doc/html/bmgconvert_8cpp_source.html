296 int read_project_dat(Bstring& filename, Bproject* project)
297 {
298  int err(0), nx(0), ny(0), n(0);
299  char inputLine[1024];
300  int unitsInHeaderI;
301  Bstring unitsInHeaderC; // "angstrom", "nm" or "pixel"
302  double px, volt, amp, def, dev, ast, cs;
303 
304  Bstring field_id = filename.base();
305  Bstring mg_id = field_id;
306  Bfield* field = field_add(&project->field, field_id);
307  Bmicrograph* mg = micrograph_add(&field->mg, mg_id);
308  Bparticle* part = NULL;
309  CTFparam* ctf = mg->ctf = new CTFparam;
310 
311  if ( verbose ) cout << "Reading a DAT file: " << filename << endl;
312 
313  /*========================
314  Read DAT-format file
315  ========================*/
316  ifstream fdat(filename.c_str());
317  if ( fdat.fail() ) {
318  cerr << "File " << filename << " does not exist!. Exit program." << endl;
319  bexit(-1);
320  }
321 
322  fdat.getline(inputLine, 1023);
323  mg->fpart = inputLine;
324  mg->fpart = mg->fpart.no_lead_space();
325  mg->fpart = mg->fpart.pre(' ');
326  mg->fpart = mg->fpart.remove('\n');
327  if ( mg->fpart.empty() ) {
328  cerr << "ERROR reading line 1 of PFT-format DAT file. Exit program" << endl;
329  exit(-1);
330  }
331 
332  // General micrograph parameters
333  sscanf(strchr(inputLine+mg->fpart.length(), ' '), " %d %d %d", &nx, &ny, &n);
334 // cout << "\"" << inputLine << "\"" << nx << " " << ny << " " << n << endl;
335 
336  fdat.getline(inputLine, 1023);
337  if ( !fdat.good() ) {
338  cout << "ERROR reading line 2 of PFT-format DAT file. Exit program" << endl;
339  exit(-1);
340  }
341  if ( sscanf(inputLine, "%lf,%d,%lf,%lf,%lf,%lf,%lf,%lf\n", &px, &unitsInHeaderI,
342  &volt, &amp, &def, &dev, &ast, &cs) == 8) {
343  switch(unitsInHeaderI) {
344  case 0:
345  unitsInHeaderC = "pixels";
346  break;
347  case 1:
348  unitsInHeaderC = "angstroms";
349  break;
350  case 2:
351  unitsInHeaderC = "nm";
352  default:
353  break;
354  }
355  } else if ( sscanf(inputLine, "%lf,%lf,%lf,%lf,%lf,%lf,%lf\n", &px, &volt,
356  &amp, &def, &dev, &ast, &cs) == 7) {
357  } else {
358  cout << "ERROR interpreting line 2 of PFT-format DAT file. Exit program" << endl;
359  exit(-1);
360  }
361 
362  if ( volt < 1000.0 ) volt *= 1000; // Volt units for STAR-format file
363  cs *= 10000000; // Angstrom units for STAR-format file
364  if ( def < 100 ) def *= 1e4; // Angstrom units for STAR-format file
365  if ( dev < 100 ) dev *= 1e4; // Angstrom units for STAR-format file
366  def = (def + dev)/2;
367  dev -= def;
368  if ( dev < 0 ) dev = -dev;
369 
370  if ( nx && ny ) mg->box_size = Vector3<int>(nx, ny, 1);
371  mg->pixel_size = Vector3<double>(px, px, 1);
372  ctf->volt(volt);
373  ctf->amp_shift(asin(amp));
374  ctf->defocus_average(def);
375  ctf->defocus_deviation(dev);
376  ctf->astigmatism_angle(ast*M_PI/180.0);
377  ctf->Cs(cs);
378 
379  char line[DATLINELENGTH];
380  int numpart(0), pid;
381  float theta, phi, omega, ox, oy, mag, fom1, fom2, fom3;
382 
383  while ( fdat.getline(line, DATLINELENGTH) && fdat.good() ) {
384  if ( sscanf(line, "%d,%f,%f,%f,%f,%f,%f,%f,%f,%f", &pid, &theta, &phi, &omega, &ox, &oy, &mag, &fom1, &fom2, &fom3) > 6 ) {
385  part = particle_add(&part, pid);
386  if ( !mg->part ) mg->part = part;
387  part->ori[0] = ox;
388  part->ori[1] = oy;
389  part->mag = mag;
390  part->fom[0] = fom1;
391  part->fom[1] = fom2;
392  part->fom[2] = fom3;
393  part->view = Euler(-omega*M_PI/180.0,
394  theta*M_PI/180.0, phi*M_PI/180.0).view();
395  numpart++;
396  }
397  }
398 
399  fdat.close(); // close input DAT file
400 
401  if ( n && numpart != n )
402  cerr << "Warning: The number of particles (" << numpart <<
403  ") does not agree with the number in the header (" << n << ")" << endl;
404 
405  if ( verbose )
406  cout << "Number of particles read: " << numpart << endl << endl;
407 
408  return err;
409 }
