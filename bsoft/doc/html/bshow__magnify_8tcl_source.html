377  }
378  }
379

383  set yz_y_radius [expr $magnification * $yrad * sqrt(1-(($xr*$xr)/($xrad*$xrad)))]
384  set xmin [expr ($lx-$zdiff-$zrad*$magnification)]
385  set xmax [expr ($lx-$zdiff+$zrad*$magnification) + $magnification - 1]
386  set ymin [expr ($ly-$ydiff-$yrad*$magnification)]
387  set ymax [expr ($ly-$ydiff+$yrad*$magnification) + $magnification - 1]

393  $w.frame.c1 create oval $cxmin $cymin $cxmax $cymax -outline $box_color -width 1 -tags [list box $idtag]

398  } else {

401  }

543 proc drawMagSphere { x y z ptx pty ptz rad col width lbl tag } {
544  global mag3D
545 
546  set w .wmag
547  if ![winfo exists $w] { return }
548 
549  set c $w.frame.c
550  set c1 $w.frame.c1
551  set c2 $w.frame.c2
552 
553  set magnification [$w.p.mag get]
554  set magsize [$w.p.size get]
555 
556  set rm [expr $rad * $magnification]
557  if { $rm < 1 } { set rm 1 }
558 
559  set lx [expr 0.5 * $magnification * $magsize]
560  set ly $lx
561 
562  set max [expr $lx + $rm]
563  set m2 [expr $lx * $lx]
564 
565  set xr [expr ($ptx - $x)*$magnification]
566  set yr [expr ($y - $pty)*$magnification]
567  set zr [expr ($ptz - $z)*$magnification]
568 
569  # Test if within the magnification window
570  set xr2 [expr $xr*$xr]
571  if { $xr2 > $m2 } { return }
572 
573  set yr2 [expr $yr*$yr]
574  if { $yr2 > $m2 } { return }
575 
576  set zr2 [expr $zr*$zr]
577  if { $zr2 > $m2 } { return }
578 
579  set xxd [expr $lx - $xr]
580  set yyd [expr $ly - $yr]
581 
582  set mag_1 [expr $magnification - 1]
583 
584  set rm2 [expr $rm*$rm]
585 
586  set dz2 [expr $rm2 - $zr2]
587 
588  if { $dz2 >= 0 } {
589  set xy_radius [expr sqrt($dz2)]
590  set xmin [expr $xxd - $xy_radius]
591  set xmax [expr $xxd + $xy_radius + $mag_1]
592  set ymin [expr $yyd - $xy_radius]
593  set ymax [expr $yyd + $xy_radius + $mag_1]
594  $c create oval $xmin $ymin $xmax $ymax -outline $col -width $width -tags $tag
595  if { [string length $lbl] } {
596  $c create text $xmax $ymax -text $lbl -fill $col -anchor w -tags $tag
597  }
598  }
599 
600  if { $mag3D } {
601  set xzd [expr $lx - $zr]
602  set yzd [expr $ly + $zr]
603  set dx2 [expr $rm2 - $xr2]
604  if { $dx2 >= 0 } {
605  set yz_radius [expr sqrt($dx2)]
606  set xmin [expr $xzd - $yz_radius]
607  set xmax [expr $xzd + $yz_radius + $mag_1]
608  set ymin [expr $yyd - $yz_radius]
609  set ymax [expr $yyd + $yz_radius + $mag_1]
610  $c1 create oval $xmin $ymin $xmax $ymax -outline $col -width $width -tags $tag
611  if { [string length $lbl] } {
612  $c1 create text $xmax $ymax -text $lbl -fill $col -anchor w -tags $tag
613  }
614  }
615 
616  set dy2 [expr $rm2 - $yr2]
617  if { $dy2 >= 0 } {
618  set xz_radius [expr sqrt($dy2)]
619  set xmin [expr $xxd - $xz_radius]
620  set xmax [expr $xxd + $xz_radius + $mag_1]
621  set ymin [expr $yzd - $xz_radius]
622  set ymax [expr $yzd + $xz_radius + $mag_1]
623  $c2 create oval $xmin $ymin $xmax $ymax -outline $col -width $width -tags $tag
624  if { [string length $lbl] } {
625  $c2 create text $xmax $ymax -text $lbl -fill $col -anchor w -tags $tag
626  }

691  set zmax [expr $ly - $zediff]

695  }

746 proc animateSlices {} {
747  global theimg
748  global aniproc2
749  global aniaxis
750 
751  set w .wmag
752 
753  if ![winfo exists $w] { return }
754  if { $aniaxis=="x" } {
755  set anisource "$w.n.x"
756  set lim [expr [Bimage get $theimg width] - 1]
757  }
758  if { $aniaxis=="y" } {
759  set anisource "$w.n.y"
760  set lim [expr [Bimage get $theimg height] - 1]
761  }
762  if { $aniaxis=="z" } {
763  set anisource "$w.q.slice"
764  set lim [expr [Bimage get $theimg nslices] - 1]
765  }
766  set aniceiling [expr [$anisource get] + [$w.r.range get]]
767  set anifloor [expr [$anisource get] - [$w.r.range get]]
768  if { $aniceiling > $lim } { set aniceiling $lim }
769  if { $anifloor < 0 } { set anifloor 0 }
770  set anidirection 1
771  set speed [$w.r.speed get]
772  if { $speed == "" || $speed < 0 } {
773  set speed 15
774  }
775  if { $speed > 200 } {
776  set speed 200
777  }
778  $w.r.speed delete 0 end
779  $w.r.speed insert 0 $speed
780 
781  if { $aniproc2 == 0 } {
782  $w.r.animate configure -text "Stop"
783  $w.r.axis configure -state disabled
784  $w.r.range configure -state disabled
785  $w.r.speed configure -state disabled
786  set aniproc2 1
787  } else {
788  $w.r.animate configure -text "Animate"
789  $w.r.axis configure -state normal
790  $w.r.range configure -state normal
791  $w.r.speed configure -state normal
792  set aniproc2 0
793  }
794 
795  while { [winfo exists $w] && $aniproc2 == 1 } {
796  set slice [$anisource get]
797  if { $slice == $aniceiling } {
798  set anidirection 0
799  }
800  if { $slice == $anifloor } {
801  set anidirection 1
802  }
803  after [expr 1000/[$w.r.speed get]]
804  updateSlicer $aniaxis $anidirection
805  update
806  }
