401 /*
402 @author Daniel Nemecek and Bernard Heymann
403 @brief Calculates the coverage of the component in a map.
404 @param *p image.
405 @param img_num sub-image number.
406 @param comp component.
407 @param threshold threshold to count covered voxels.
408 @return double fraction of examined sphere covered.
409 
410  The map must have positive density (higher values are considered density).
411  The coverage for a component is defined as the fraction of voxels within
412  a sphere around the component location that is above the threshold.
413  The radius of the sphere is taken from the component radius.

426  start[2] = (int) (p->image->origin()[2] + (comp->location()[2] - comp->radius())/p->sampling(0)[2]);

430 
431  if ( verbose & VERB_DEBUG ) {
432  cout << "DEBUG component_coverage: start=" << start << endl;
433  cout << "DEBUG component_coverage: end=" << end << endl;
434  }
435 
436  if ( ( start[0] < 0 ) && ( end[0] < 0 ) ) return 0;
437  if ( ( start[1] < 0 ) && ( end[1] < 0 ) ) return 0;
438  if ( ( start[2] < 0 ) && ( end[2] < 0 ) ) return 0;
439  if ( ( start[0] >= p->sizeX() ) && ( end[0] >= p->sizeX() ) ) return 0;
440  if ( ( start[1] >= p->sizeY() ) && ( end[1] >= p->sizeY() ) ) return 0;
441  if ( ( start[2] >= p->sizeZ() ) && ( end[2] >= p->sizeZ() ) ) return 0;

446  if ( start[2] >= p->sizeZ() ) start[2] = p->sizeZ() - 1;
