734 /*
735 @brief Converts micrograph data from a STAR data base to a project structure.
736 @param &star STAR data base.
737 @param *project project structure.
738 @return int error code (<0 means failure).
739 
740  The function sets up the project hierarchy from a STAR database file.
741  Micrograph data blocks must contain the "micrograph.id" tag.
742  If the "micrograph.field_of_view_id" is present, the micrographs are
743  distributed into multi-micrograph fields-of-view. Otherwise, each
744  micrograph is assumed to be a unique field-of-view.
745  All angles given in the STAR data base are assumed to be in degrees,
746  and are converted here only to radians.
747 
748 **/
749 int star_to_project(Bstar2& star, Bproject* project)
750 {
751  int err(0);
752  int nmg(0), nrec(0);
753 
754  if ( verbose & VERB_DEBUG )
755  cout << "DEBUG star_to_project:" << endl;
756 
757  mg_star_update_tags(star);
758 
759  nmg = star.number_of_blocks(MICROGRAPH_ID);
760  if ( nmg < 1 ) nmg = star.number_of_blocks(MICROGRAPH_FILE);
761  if ( nmg < 1 ) nmg = star.number_of_blocks(MICROGRAPH_FRAMES_FILE);
762  if ( nmg < 1 ) nmg = star.number_of_blocks(MICROGRAPH_TRANSFORM_FILE);
763  if ( nmg < 1 ) nmg = star.number_of_blocks(MICROGRAPH_POWERSPEC_FILE);
764 
765  nrec = star.number_of_blocks(MAP_ID);
766  if ( nrec < 1 ) nrec = star.number_of_blocks(MAP_RECONSTRUCTION);
767  if ( nrec < 1 ) nrec = star.number_of_blocks(MAP_TRANSFORM_FILE);
768  if ( nrec < 1 ) nrec = star.number_of_blocks(MAP_POWERSPEC_FILE);
769 
770  if ( nmg < 1 && nrec < 1 ) {
771  cerr << "Warning: No micrograph or reconstruction data blocks found" << endl;
772  return 0;
773  }
774 
775  BstarBlock block;
776  Bfield* field = NULL;
777  Bmicrograph *mg, *mg2;
778  Breconstruction *rec, *rec2;
779 
780  for ( rec2 = project->rec; rec2 && rec2->next; rec2 = rec2->next ) ;
781 
782  if ( star.exists(MAP_REFERENCE) ) {
783  block = star.find(MAP_REFERENCE);
784  for ( auto il: block.loops() )
785  for ( auto ir: il.data() )
786  string_add(&project->reference, ir[0].c_str());
787  }
788 
789  if ( ( verbose & VERB_DEBUG ) && project->reference )
790  cout << "DEBUG star_to_project: reference map = " << *(project->reference) << endl;
791 
792  int i, j, f, nfield(0);
793  Bstring field_id, mgid;
794  Bstring base;
795 
796  // Make sure FOM tags are properly set up
797  if ( star.number_of_blocks(PARTICLE_FOM) ) project->fom_tag[0] = FOM;
798  if ( star.number_of_blocks(PARTICLE_FOM_CV) ) project->fom_tag[1] = FOM_CV;
799  if ( star.number_of_blocks(PARTICLE_HANDA_FOM) ) project->fom_tag[1] = FOM_HAND_A;
800  if ( star.number_of_blocks(PARTICLE_HANDB_FOM) ) project->fom_tag[2] = FOM_HAND_B;
801 
802  project_get_fom_tags(star, project->fom_tag);
803  for ( f=i=0; f<NFOM; f++ ) i += project->fom_tag[f];
804  if ( i < 2 ) project->fom_tag[0] = FOM;
805  if ( verbose & VERB_DEBUG )
806  for ( f=0; f<NFOM; f++ ) if ( project->fom_tag[f] )
807  cout << "DEBUG star_to_project: fom_tag=" << project->fom_tag[f] << endl;
808 
809  // The orientation convention used is set based on whether the view angle
810  // array or the phi euler angle array is present in the file.
811  // If both are present, the one with the larger number of values wins,
812  // otherwise only the view vector and rotation angle
813  // representation will be used.
814  i = star.number_of_blocks(PARTICLE_PHI);
815  j = star.number_of_blocks(PARTICLE_VIEW_ANGLE);
816  project->euler_flag = 0;
817  project->omega_flag = 0;
818  if ( i > 0 && j > 0 )
819  cerr << "Error: The STAR file contains both views and Euler angles!" << endl;
820  else if ( i > j ) {
821  project->euler_flag = 1;
822  if ( star.number_of_blocks(PARTICLE_OMEGA) > 0 )
823  project->omega_flag = 1;
824  }
825 
826  nmg = i = 0;
827  for ( auto ib: star.blocks() ) {
828  if ( ib.exists(MICROGRAPH_ID) || ib.exists(MICROGRAPH_FILE) ) {
829  mgid = 0;
830  field_id = 0;
831  /* This deals with old style access to micrograph data blocks */
832  if ( ib.exists(MICROGRAPH_ID) ) {
833  mgid = ib[MICROGRAPH_ID];
834  } else {
835  if ( ib.exists(MICROGRAPH_FILE) ) mgid = ib[MICROGRAPH_FILE];
836  else if ( ib.exists(MICROGRAPH_FRAMES_FILE) ) mgid = ib[MICROGRAPH_FRAMES_FILE];
837  else if ( ib.exists(PARTICLE_FILE) ) mgid = ib[PARTICLE_FILE];
838  else if ( ib.exists(FILAMENT_FILE) ) mgid = ib[FILAMENT_FILE];
839  else if ( ib.exists(MICROGRAPH_TRANSFORM_FILE) ) mgid = ib[MICROGRAPH_TRANSFORM_FILE];
840  else if ( ib.exists(MICROGRAPH_POWERSPEC_FILE) ) mgid = ib[MICROGRAPH_POWERSPEC_FILE];
841  if ( mgid.length() > 0 ) mgid = mgid.base();
842  }
843  if ( mgid.length() > 0 ) {
844  if ( ib.exists(MICROGRAPH_FIELD_ID) ) field_id = ib[MICROGRAPH_FIELD_ID];
845  if ( field_id.length() < 1 ) field_id = mgid;
846  if ( !project->field ) {
847  field_add(&field, field_id);
848  project->field = field; // First field
849  nfield++;
850  } else {
851  field = NULL;
852  if ( field_id.length() > 0 )
853  field = field_find_id(project->field, field_id);
854  if ( !field ) {
855  field = project->field;
856  field = field_add(&field, field_id);
857  nfield++;
858  }
859  }
860  mg = micrograph_from_starblock(ib, mgid, project->fom_tag);
861  mg->block = i++;
862  if ( verbose & VERB_FULL )
863  cout << "Reading field \"" << field->id << "\", micrograph \"" << mg->id <<"\"" << endl;
864  if ( field->mg ) {
865  for ( mg2 = field->mg; mg2->next; mg2 = mg2->next ) ;
866  mg2->next = mg;
867  } else field->mg = mg;
868  mgid = 0;
869  field_id = 0;
870  nmg++;
871  }
872  } else if ( ib.exists(MAP_ID) ||
873  ib.exists(MAP_RECONSTRUCTION) ||
874  ib.exists(MAP_TRANSFORM_FILE) ||
875  ib.exists(MAP_POWERSPEC_FILE) ) {
876  rec = reconstruction_from_starblock(ib, project->fom_tag);
877  if ( !project->rec ) project->rec = rec;
878  else rec2->next = rec;
879  rec2 = rec;
880  } else if ( ib.exists_loop(PARTICLE_ID) ) {
881  project->class_avg = particle_from_starblock(ib, project->fom_tag);
882  }
883  }
884 
885  star.replace_tag(CTF_DEF_MIN, CTF_DEF_AVG);
886  star.replace_tag(CTF_DEF_MAX, CTF_DEF_DEV);
887 
888  return err;
889 }
