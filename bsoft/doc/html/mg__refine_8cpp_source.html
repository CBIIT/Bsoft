659 double img_recip_space_fom(Bimage* p, Bimage* pref, double hi_res, double lo_res,
660  int fom_type, vector<double>& weight, int apply_ctf, CTFparam* em_ctf)
661 {
662  if ( pref->sampling(0)[0] < 0.01 ) pref->sampling(1,1,1);
663  if ( lo_res > 0 && hi_res > lo_res ) swap(hi_res, lo_res);
664  pref->check_resolution(hi_res);
665 
666  double max_rad = pref->real_size()[0]/hi_res;
667  double max_rad_cv = 1.1*max_rad;
668  if ( max_rad_cv <= max_rad ) max_rad_cv = max_rad + 1;
669 
670  double max_rad_sq = max_rad*max_rad;
671  double max_rad_cv_sq = max_rad_cv*max_rad_cv;
672  double min_rad_sq(0);
673  if ( lo_res > 0 ) {
674  min_rad_sq = pref->real_size()[0]/lo_res;
675  min_rad_sq *= min_rad_sq;
676  }
677 
678  if ( verbose & VERB_DEBUG )
679  cout << "DEBUG img_recip_space_fom: max_rad=" << max_rad << " (" << max_rad_sq << ")" << endl;
680 
681  double dphi, ctf_fac;
682  if ( !em_ctf ) apply_ctf = 0;
683  else if ( em_ctf->defocus_average() < 1 ) apply_ctf = 0;
684 
685  long i, j, xx, yy;
686  long hx = (p->sizeX() - 1)/2, hy = (p->sizeY() - 1)/2;
687  double sx2, sy2, s2, d2, amp, denom, f, cv, wsum;
688  Vector3<double> m, d, iv;
689  Vector3<double> freq_scale(1.0/p->real_size()[0], 1.0/p->real_size()[1], 1.0);
690 // cout << "size = " << p->size() << endl;
691 // cout << "sampling = " << p->image->sampling() << endl;
692 // cout << "real size = " << p->real_size() << endl;
693 // bexit(-1);
694 
695  double* w = new double[p->sizeX()];
696  for ( j=0; j<p->sizeX(); j++ ) w[j] = 1;
697  if ( weight.size() )
698  for ( size_t j=0; j<weight.size(); j++ ) if ( weight[j] ) w[j] = weight[j];
699 
700  double* fom = new double[p->sizeX()];
701  double* dsum = new double[p->sizeX()];
702  double* msum = new double[p->sizeX()];
703  for ( i=0; i<p->sizeX(); i++ ) fom[i] = dsum[i] = msum[i] = 0;
704 
705  Complex<float>* data = (Complex<float> *) p->data_pointer();
706  Complex<float>* refdata = (Complex<float> *) pref->data_pointer();
707  Complex<float> dtemp;
708 
709  for ( yy=i=0; yy<p->sizeY(); yy++ ) {
710  iv[1] = yy;
711  if ( iv[1] > hy ) iv[1] -= p->sizeY();
712  sy2 = freq_scale[1]*iv[1];
713  sy2 *= sy2;
714  for ( xx=0; xx<p->sizeX(); xx++, i++ ) {
715  iv[0] = xx;
716  if ( iv[0] > hx ) iv[0] -= p->sizeX();
717  sx2 = freq_scale[0]*iv[0];
718  sx2 *= sx2;
719  d2 = iv.length2();
720  if ( d2 >= min_rad_sq && d2 <= max_rad_cv_sq ) {
721  s2 = sx2 + sy2;
722  j = (long) sqrt(d2);
723  dtemp = refdata[i];
724  if ( apply_ctf ) {
725  ctf_fac = em_ctf->calculate(s2, atan2(iv[1],iv[0]));
726  dtemp *= -ctf_fac;
727  }
728  if ( fom_type < 1 ) {
729  fom[j] += data[i].real()*dtemp.real() + data[i].imag()*dtemp.imag();
730  msum[j] += dtemp.power();
731  dsum[j] += data[i].power();
732  } else {
733  amp = data[i].amp() + dtemp.amp();
734  dphi = angle_set_negPI_to_PI(data[i].phi() - dtemp.phi());
735  fom[j] += amp*dphi*dphi;
736  msum[j] += amp;
737  }
738  }
739  }
740  }
741 
742  if ( fom_type < 1 ) {
743  for ( f=wsum=0, j=0; j<max_rad; j++ ) {
744  wsum += w[j];
745  denom = dsum[j]*msum[j];
746  if ( denom > 0 ) f += w[j]*fom[j]/sqrt(denom);
747  }
748  f /= wsum;
749  for ( cv=wsum=0, j=(long)max_rad; j<max_rad_cv; j++ ) {
750  wsum += w[j];
751  denom = dsum[j]*msum[j];
752  if ( denom > 0 ) cv += w[j]*fom[j]/sqrt(denom);
753  }
754  cv /= wsum;
755  } else {
756  for ( f=wsum=0, j=0; j<max_rad; j++ ) {
757  wsum += w[j];
758  if ( msum[j] ) f += w[j]*cos(sqrt(fom[j]/msum[j]));
759  }
760  f /= wsum;
761  for ( cv=wsum=0, j=(long)max_rad; j<max_rad_cv; j++ ) {
762  wsum += w[j];
763  if ( msum[j] ) cv += w[j]*cos(sqrt(fom[j]/msum[j]));
764  }
765  cv /= wsum;
766  }
767 
768  delete[] fom;
769  delete[] dsum;
770  delete[] msum;
771  delete[] w;
772 
773  p->image->FOM(cv);
774 
775  return f;
776 }
