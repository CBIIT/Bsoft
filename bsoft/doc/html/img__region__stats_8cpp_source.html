207 /*
208 @brief Determines if a vector is inside or outside a polyhedral structure.
209 
210  Whether a point is inside or outside is based on the polygon edges.
211  First the normal to an edge is calculated and tested whether it points
212  inside or outside the polygon by finding the distance to the center:
213  dist_cent = normal . edge_vertex
214  which must be positive. Then the distance to the given point is:
215  dist = normal . (edge_vertex - point)
216  The minimum distance gives the distance of the point to the closest edge.
217  If this distance is positive, the point is inside the polygon.
218  Requirement: The polygon vertices must be given in order of connectivity.
219 @param vec vector to test for.
220  int nvert number of vertices.
221  int ndim number of dimensions.
222  Vector3<float>* poly set of vertices.
223 @return double distance from closest edge (inside positive).
224 **/

226 {
227  int i, j;
228  double d, dmin;
229  Vector3<float> normal;
230 
231  // Find the closest edge
232  dmin = 1e30;
233  for ( i=0, j=1; i<nvert; i++, j++ ) {
234  if ( j >= nvert ) j = 0;
235  normal = poly[i] - poly[j];
236  normal = Vector3<float>(-normal.y(), normal.x(), 0);
237  normal.normalize();
238  if ( normal.scalar(poly[i]) < 0 ) normal = -normal;
