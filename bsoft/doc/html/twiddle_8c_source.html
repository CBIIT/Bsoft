41 /* compute (x * y) mod p, but watch out for integer overflows; we must

44 {

55 static fftw_complex *fftw_compute_rader_twiddle(int n, int r, int g)
56 {
57  FFTW_TRIG_REAL twoPiOverN;
58  int m = n / r;
59  int i, j, gpower;
60  fftw_complex *W;
61 
62  twoPiOverN = FFTW_K2PI / (FFTW_TRIG_REAL) n;
63  W = (fftw_complex *) fftw_malloc((r - 1) * m * sizeof(fftw_complex));
64  for (i = 0; i < m; ++i)
65  for (gpower = 1, j = 0; j < r - 1; ++j,
66  gpower = MULMOD(gpower, g, r)) {
67  int k = i * (r - 1) + j;
68  FFTW_TRIG_REAL
69  ij = (FFTW_TRIG_REAL) (i * gpower);
70  c_re(W[k]) = FFTW_TRIG_COS(twoPiOverN * ij);
71  c_im(W[k]) = FFTW_FORWARD * FFTW_TRIG_SIN(twoPiOverN * ij);
72  }
73 
74  return W;
75 }

77 /*
78  * compute the W coefficients (that is, powers of the root of 1)
79  * and store them into an array.
80  */
81 static fftw_complex *fftw_compute_twiddle(int n, const fftw_codelet_desc *d)
82 {
83  FFTW_TRIG_REAL twoPiOverN;
84  int i, j;
85  fftw_complex *W;
86 
87  twoPiOverN = FFTW_K2PI / (FFTW_TRIG_REAL) n;
88 
89  if (!d) {
90  /* generic codelet, needs all twiddles in order */
91  W = (fftw_complex *) fftw_malloc(n * sizeof(fftw_complex));
92  for (i = 0; i < n; ++i) {
93  c_re(W[i]) = FFTW_TRIG_COS(twoPiOverN * (FFTW_TRIG_REAL) i);
94  c_im(W[i]) = FFTW_FORWARD * FFTW_TRIG_SIN(twoPiOverN * (FFTW_TRIG_REAL) i);
95  }
96  } else if (d->type == FFTW_RADER)
97  W = fftw_compute_rader_twiddle(n, d->size, d->signature);
98  else {
99  int r = d->size;
100  int m = n / r, m_alloc;
101  int r1 = d->ntwiddle;
102  int istart;
103 
104  if (d->type == FFTW_TWIDDLE) {
105  istart = 0;
106  m_alloc = m;
107  } else if (d->type == FFTW_HC2HC) {
108  /*
109  * This is tricky, do not change lightly.
110  */
111  m = (m + 1) / 2;
112  m_alloc = m - 1;
113  istart = 1;
114  } else {
115  fftw_die("compute_twiddle: invalid argument\n");
116  /* paranoia for gcc */
117  m_alloc = 0;
118  istart = 0;
119  }
120 
121  W = (fftw_complex *) fftw_malloc(r1 * m_alloc * sizeof(fftw_complex));
122  for (i = istart; i < m; ++i)
123  for (j = 0; j < r1; ++j) {
124  int k = (i - istart) * r1 + j;
125  FFTW_TRIG_REAL
126  ij = (FFTW_TRIG_REAL) (i * d->twiddle_order[j]);
127  c_re(W[k]) = FFTW_TRIG_COS(twoPiOverN * ij);
128  c_im(W[k]) = FFTW_FORWARD * FFTW_TRIG_SIN(twoPiOverN * ij);
129  }
130  }
131 
132  return W;
133 }

142 /* true if the two codelets can share the same twiddle factors */
143 static int compatible(const fftw_codelet_desc *d1, const fftw_codelet_desc *d2)
144 {
145  int i;
146 
147  /* true if they are the same codelet */
148  if (d1 == d2)
149  return TRUE;
150 
151  /* false if one is null and the other is not */
152  if (!d1 || !d2)
153  return FALSE;
154 
155  /* false if size is different */
156  if (d1->size != d2->size)
157  return FALSE;
158 
159  /* false if different types (FFTW_TWIDDLE/FFTW_HC2HC/FFTW_RADER) */
160  if (d1->type != d2->type)
161  return FALSE;
162 
163  /* false if they need different # of twiddles */
164  if (d1->ntwiddle != d2->ntwiddle)
165  return FALSE;
166 
167  /* false if the twiddle orders are different */
168  for (i = 0; i < d1->ntwiddle; ++i)
169  if (d1->twiddle_order[i] != d2->twiddle_order[i])
170  return FALSE;
171 
172  return TRUE;
173 }

175 fftw_twiddle *fftw_create_twiddle(int n, const fftw_codelet_desc *d)
176 {
177  fftw_twiddle *tw;
178 
179  /* lookup this n in the twiddle list */
180  for (tw = twlist; tw; tw = tw->next)
181  if (n == tw->n && compatible(d, tw->cdesc)) {
182  ++tw->refcnt;
183  return tw;
184  }
185  /* not found --- allocate a new struct twiddle */
186  tw = (fftw_twiddle *) fftw_malloc(sizeof(fftw_twiddle));
187  fftw_twiddle_size += n;
188 
189  tw->n = n;
190  tw->cdesc = d;
191  tw->twarray = fftw_compute_twiddle(n, d);
192  tw->refcnt = 1;
193 
194  /* enqueue the new struct */
195  tw->next = twlist;
196  twlist = tw;
197 
198  return tw;
199 }
