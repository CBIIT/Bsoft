67 fftw_time fftw_time_diff(fftw_time t1, fftw_time t2)
68 /*
69  * This function takes the difference t1 - t2 of two 64 bit
70  * integers, represented by the 32 bit lo and hi words.
71  * if t1 < t2, returns 0.
72  */
73 {
74  fftw_time diff;
75 
76  if (t1.hi < t2.hi) { /* something is wrong...t1 < t2! */
77  diff.hi = diff.lo = 0;
78  return diff;
79  } else
80  diff.hi = t1.hi - t2.hi;
81 
82  if (t1.lo < t2.lo) {
83  if (diff.hi > 0)
84  diff.hi -= 1; /* carry */
85  else { /* something is wrong...t1 < t2! */
86  diff.hi = diff.lo = 0;
87  return diff;
88  }
89  }
90  diff.lo = t1.lo - t2.lo;
91 
92  return diff;
93 }

104 unsigned long GetPerfTime(void)
105 {
106  LARGE_INTEGER lCounter;
107 
108  if (gFirstTime) {
109  gFirstTime = 0;
110 
111  if (QueryPerformanceFrequency(&gFreq)) {
112  gHaveHiResTimer = 1;
113  }
114  }
115  if (gHaveHiResTimer) {
116  QueryPerformanceCounter(&lCounter);
117  return lCounter.u.LowPart;
118  } else {
119  return (unsigned long) clock();
120  }
121 }
