114 ## @brief Calculates the histogram
115 #
116 # @param ngauss Number of gaussians to fit.
117 
118 proc calcHistogram { ngauss } {
119  global histplot histfit gprm theimg
120  global color
121  if { [info exists histplot] } { unset histplot }
122  if { [info exists histfit] } { unset histfit }
123  set xmin [Bimage get $theimg min]
124  set xmax [Bimage get $theimg max]
125  set bins [.whis.bins.entry get]
126  if { $bins < 2 } {
127  set bins 250
128  .whis.bins.entry delete 0 end
129  .whis.bins.entry insert 0 $bins
130  }
131  set hist [Bimage histogram $theimg $bins]
132  if { $ngauss > 0 } {
133  set gprm [Bimage histofit $theimg $bins $ngauss]
134  }
135  set n [expr [llength $hist] / $bins]
136  set color { LightBlue }
137  if { $n == 3 } {
138  set color { red green blue }
139  }
140 # puts "Number of channels = $n"
141  set xscale [expr ($xmax - $xmin)/$bins]
142  for { set i 0 } { $i < $bins } { incr i } {
143  set histplot(x,$i) [expr $i*$xscale + $xmin]
144  set histplot(y,$i) [lindex $hist $i]
145  }
146  if { $ngauss > 0 } {
147  for { set i 0 } { $i < $bins } { incr i } {
148  set x [expr $i*$xscale + $xmin]
149  set v [expr ([lindex $gprm 1] - $x)/[lindex $gprm 2]]
150  set histfit(x,$i) $x
151  set histfit(y,$i) [expr [lindex $gprm 0] * exp(-0.5 * $v * $v)]
152  }
153  }
154  for { set j 1 } { $j < $n } { incr j } {
155  for { set i 0 } { $i < $bins } { incr i } {
156  append histplot(y,$i) " " [lindex $hist [expr $j*$bins + $i]]
157  }
158  }
159  if { $ngauss > 1 } {
160  for { set j 1 } { $j < $ngauss } { incr j } {
161  set a [lindex $gprm [expr 3*$j]]
162  set d [lindex $gprm [expr 3*$j+1]]
163  set s [lindex $gprm [expr 3*$j+2]]
164  for { set i 0 } { $i < $bins } { incr i } {
165  set x [expr $i*$xscale + $xmin]
166  set v [expr ($d - $x)/$s]
167  append histfit(y,$i) " " [expr $a * exp(-0.5 * $v * $v)]
168  }
169  }
170  }
171 }

592 proc getLimits { w plotdata } {
593  upvar $plotdata plot
594 # puts "$plot(x,0) $plot(y,0)"
595  set number [expr int( [array size plot] / 2 ) ]
596  if { $number < 2 } { return }

602  if { $xmin > $plot(x,$i) } { set xmin $plot(x,$i) }
603  if { $xmax < $plot(x,$i) } { set xmax $plot(x,$i) }
604  for { set j 0 } { $j < [llength $plot(y,$i)] } { incr j } {
605  if { $ymin > [lindex $plot(y,$i) $j] } { set ymin [lindex $plot(y,$i) $j] }
606  if { $ymax < [lindex $plot(y,$i) $j] } { set ymax [lindex $plot(y,$i) $j] }
607  }
608  }
609  if { $xmin == $xmax } { set xmax [expr $xmin + 1] }
610  if { $ymin == $ymax } { set ymax [expr $ymin + 1] }
611  updateLimits $w $xmin $xmax $ymin $ymax
612 }
