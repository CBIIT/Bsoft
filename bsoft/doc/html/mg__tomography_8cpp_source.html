445 double lambda_R(Bsimplex& simp)
446 {
447  long i;
448  double R(0), df, fs(0), fs2(0);
449  vector<double>& f = simp.dependent_values();
450  vector<double>& x = simp.independent_values();
451 
452  for ( i=0; i<simp.points(); i++ ) {
453  df = cos(x[i] + simp.parameter(2));
454  if ( df )
455  df = f[i] - simp.parameter(0)*exp(-simp.parameter(1)/df);
456  else
457  df = f[i] - simp.parameter(0);
458  fs += f[i];
459  fs2 += f[i] * f[i];
460  R += df*df;
461  }
462 
463  df = i*fs2-fs*fs;
464  if ( df > 0 ) R = sqrt(R/df);
465  else R = sqrt(R/i);
466 
467  return R;
