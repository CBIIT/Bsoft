365 Bimage* img_students_t_test(int n, Bimage* p, vector<double>& weight)
366 {
367  Bimage* pt = new Bimage(Float, TSimple, p->size(), 1);
368  pt->origin(p->image->origin());
369  pt->sampling(p->sampling(0));
370 
371  float* var = (float *) p->next->data_pointer();
372 
373  pt->next = new Bimage(Float, TSimple, pt->size(), 1);
374  float* siglev = (float *) pt->next->data_pointer();
375 
376  long i, datasize = p->data_size();
377  double ws(0), d;
378 
379  for ( i=0; i<n; i++ ) ws += weight[i];
380 
381  double dof = ws*(1 - 1.0L/n);
382  double hdof = 0.5*dof;
383  double avg_var = 0;
384 
385  for ( i=0; i<datasize; i++ ) avg_var += var[i];
386  avg_var /= datasize;
387 
388  if ( verbose ) {
389  cout << "Student's t test:" << endl;
390  cout << "Number: " << n << endl;
391  cout << "Average variance: " << avg_var << endl;
392  cout << "Degrees of freedom: " << dof << endl << endl;
393  }
394 
395  avg_var *= avg_var;
396 
397  for ( i=0; i<datasize; i++ ) {
398  d = var[i]/n;
399  if ( d < avg_var ) d = avg_var;
400  d = sqrt(d);
401  pt->set(i, (*p)[i]/d);
402  siglev[i] = betai(hdof, 0.5, dof/(dof + (*pt)[i]*(*pt)[i]));
403  }
404 
405  return pt;

433 Bimage* img_students_t_test_equal(int set1, int set2, Bimage* p1, Bimage* p2, vector<double>& weight)
434 {
435  // Set arrays and image parameters
436  Bimage* pt = new Bimage(Float, TSimple, p1->size(), 1);
437  pt->origin(p1->image->origin());
438  pt->sampling(p1->sampling(0));
439  pt->next = new Bimage(Float, TSimple, pt->size(), 1);
440 
441  long i, j, datasize = p1->sizeX()*p1->sizeY()*p1->sizeZ()*p1->images();
442 
443  float* var1 = (float *) p1->next->data_pointer(); // pointers to variance data
444  float* var2 = (float *) p2->next->data_pointer();
445  float* sig = (float *) pt->next->data_pointer();
446 
447  // Compute Student's t-test
448  if ( verbose ) {
449  cout << endl << "Student's t-test, assuming same variance in sets 1 and 2" << endl;
450  cout << "Set 1 contains " << set1 << " images" << endl;
451  cout << "Set 2 contains " << set2 << " images" << endl;
452  }
453 
454  double ws1, ws2;
455  for ( i=0, j=0, ws1=0; i<set1; i++, j++ ) ws1 += weight[j];
456  for ( i=0, ws2=0; i<set2; i++, j++ ) ws2 += weight[j];
457 
458  double inv_weight_sum = (1.0L / ws1) + (1.0L / ws2);
459 
460  double dof1 = ws1*(1.0L - 1.0L/set1); // degrees of freedom
461  double dof2 = ws2*(1.0L - 1.0L/set2);
462  double dof = dof1 + dof2;
463  double hdof = 0.5*dof;
464  double SD;
465  double Vn1, Vn2;
466 
467  long zero_count1 = 0, zero_count2 = 0; // Counts numbers of zero values in t-test
468  for ( i=0; i<datasize; i++ ) {
469  Vn1 = var1[i] * dof1; // Compute numerator of variance from the variance
470  Vn2 = var2[i] * dof2;
471  SD = sqrt( ( (Vn1 + Vn2) / dof ) * inv_weight_sum ); // Standard error of the difference of the means
472  if ( SD > 0 ) { // test for zero in denominator
473  pt->set(i, ( (*p1)[i] - (*p2)[i] ) / SD); // t-test value
474  if ( (*pt)[i] == 0 ) zero_count2++;
475  sig[i] = betai(hdof, 0.5L, dof/(dof + (*pt)[i]*(*pt)[i]) ); // significance level
476  } else {
477  pt->set(i, 0);
478  sig[i] = 1;
479  zero_count1++;
480  }
481  }
482 
483  if ( verbose & VERB_FULL ) {
484  cout << zero_count1 << " pixels had a zero value for the standard error of the differences of the means, and t was set to zero." << endl;
485  cout << zero_count2 << " pixels had standard deviation > 0 and t = 0." << endl;
486  cout << datasize << " = total pixels in image" << endl;
487  }
488 
489  return pt;
490 }

522 
523  long i, j, datasize = p1->sizeX()*p1->sizeY()*p1->sizeZ()*p1->images();
524 
525  float* var1 = (float *) p1->next->data_pointer(); // pointers to variance data

530  if ( verbose ) {

532  cout << "Set 1 contains " << set1 << " images" << endl;
533  cout << "Set 2 contains " << set2 << " images" << endl;
534  }

539 
540  double dof1 = ws1*(1.0L - 1.0L/set1); // degrees of freedom
541  double dof2 = ws2*(1.0L - 1.0L/set2);
542  double dof;
543  double Vn1, Vn2;
544 
545  long zero_count1 = 0, zero_count2 = 0; // Counts numbers of zero values in t-test
546  for ( i=0; i<datasize; i++ ) {
547  if ( var1[i] > 0 || var2[i] > 0 ) { // test for zero in denominator
548  Vn1 = var1[i]/ws1;
549  Vn2 = var2[i]/ws2;
550  pt->set(i, ( (*p1)[i] - (*p2)[i] ) / sqrt(Vn1 + Vn2));
551  if ( (*pt)[i] == 0 ) zero_count2++;
552  dof = (Vn1 + Vn2)*(Vn1 + Vn2) / (Vn1*Vn1/dof1 + Vn2*Vn2/dof2);
553  sig[i] = betai(0.5L*dof, 0.5L, dof/(dof + (*pt)[i]*(*pt)[i]) ); // significance level
554  } else {
555  pt->set(i, 0);
556  sig[i] = 1;
557  zero_count1++;
558  }
559  }
560

562  cout << zero_count1 << " pixels had zero variance and t was set to zero." << endl;
563  cout << zero_count2 << " pixels had variance not equal to 0 and t = 0." << endl;
564  cout << datasize << " = total pixels in image" << endl;
565  }
566

589 Bimage* img_f_test(int set1, int set2, Bimage* p1, Bimage* p2, vector<double>& weight)
590 {
591  // Set image parameters and arrays
592  Bimage* pf = new Bimage(Float, TSimple, p1->size(), 1);
593  pf->origin(p1->image->origin());
594  pf->sampling(p1->sampling(0));
595  pf->next = new Bimage(Float, TSimple, pf->size(), 1);
596 
597  long i, j, datasize = p1->sizeX()*p1->sizeY()*p1->sizeZ()*p1->images();
598 
599  float* var1 = (float *) p1->next->data_pointer(); // pointers to variance data
600  float* var2 = (float *) p2->next->data_pointer();
601  float* sig = (float *) pf->next->data_pointer();
602 
603  // Compute F-test
604  if ( verbose ) {
605  cout << endl << "F-test" << endl;
606  cout << "Set 1 contains " << set1 << " images" << endl;
607  cout << "Set 2 contains " << set2 << " images" << endl;
608  }
609 
610  double ws1, ws2;
611  for ( i=0, j=0, ws1=0; i<set1; i++, j++ ) ws1 += weight[j];
612  for ( i=0, ws2=0; i<set2; i++, j++ ) ws2 += weight[j];
613 
614  double dof1 = ws1*(1.0L - 1.0L/set1);
615  double dof2 = ws2*(1.0L - 1.0L/set2);
616  double df1, df2; // degrees of freedom, sets 1 and 2
617 
618  for ( i=0; i<datasize; i++ ) {
619  if ( var1[i] < SMALLFLOAT || var2[i] < SMALLFLOAT ) {
620  pf->set(i, 0);
621  sig[i] = 1;
622  j++;
623  } else {
624  if ( var1[i] > var2[i] ) { // make f the ratio of the larger variance to the smaller one
625  pf->set(i, var1[i] / var2[i]);
626  df1 = dof1;
627  df2 = dof2;
628  } else {
629  pf->set(i, var2[i] / var1[i]);
630  df1 = dof2;
631  df2 = dof1;
632  }
633  sig[i] = 2*betai(0.5L*df2, 0.5L*df1, df2/(df2 + df1*(*pf)[i]) ); // significance level
634  if ( sig[i] > 1 ) sig[i] = 2 - sig[i];
635  }
636  }
637  if ( verbose & VERB_FULL ) {
638  cout << j << " pixels in set 1 or set 2 had zero variance, and the F-values were set to zero." << endl;
639  cout << datasize << " = total pixels in image" << endl;
640  }
641 
642  return pf;
643 }
