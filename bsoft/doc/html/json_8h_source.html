432 JSvalue parseJSvalue() {
433 JSvalue val;
434 for ( ; it!=content.end() && *it != '}' && *it != ']' && *it != ','; ++it ) {
435 if ( *it == '{' ) {
436 return parseJSobject();
437 } else if ( *it == '[' ) {
438 return parseJSarray();
439 } else if ( *it == '"' ) {
440 return parseJSstring();
441 } else if ( isdigit(*it) || *it == '.' || *it == '+' || *it == '-' ) {
442 return parseJSnumber();
443 } else if ( *it == 't' || *it == 'f' || *it == 'n' ) {
444 return parseJSbare();
445 } else if ( !isspace(*it) ) {
446 fail("Found an unquoted string.");
447 }
448 }
449 return val;

451 string parseString() {
452 string s;
453 for ( ; it!=content.end() && *it != '"'; ++it )
454 if ( !isspace(*it) ) fail("Unquoted string.");
455 if ( *it != '"' ) fail("No opening quote on a string.");
456 for ( ++it; it!=content.end(); ++it ) {
457 if ( *it == '\\' ) { s += *it; ++it; s += *it; }
458 else if ( *it == '"' ) break;
459 else s += *it;
460// cout << *it;
461 }
462// cout << s << " \"" << *it << "\"" << endl;
463 if ( *it != '"' ) fail("No closing quote on a string.");
464 ++it;

471 JSvalue parseJSnumber() {
472 string s;

475 && *it != ',' && !isspace(*it); ++it ) {
476 if ( *it == '.' || *it == 'e' ) ft = 1;
