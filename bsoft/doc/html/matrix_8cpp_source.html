828 void dsytrd3(Matrix& A, vector<double>& d, vector<double>& e)
829 {
830  const int n = 3;
831  int i, j;
832  double u[3], q[3];
833  double omega, f;
834  double K, h, g;
835  double Q[3][3];
836 
837  // Initialize Q to the identitity matrix
838  for (i=0; i < n; i++)
839  {
840  Q[i][i] = 1.0;
841  for (j=0; j < i; j++)
842  Q[i][j] = Q[j][i] = 0.0;
843  }
844 
845  // Bring first row and column to the desired form
846  h = SQR(A[0][1]) + SQR(A[0][2]);
847  if (A[0][1] > 0)
848  g = -sqrt(h);
849  else
850  g = sqrt(h);
851  e[0] = g;
852  f = g * A[0][1];
853  u[1] = A[0][1] - g;
854  u[2] = A[0][2];
855 
856  omega = h - f;
857  if (omega > 0.0)
858  {
859  omega = 1.0 / omega;
860  K = 0.0;
861  for (i=1; i < n; i++)
862  {
863  f = A[1][i] * u[1] + A[i][2] * u[2];
864  q[i] = omega * f; // p
865  K += u[i] * f; // u* A u
866  }
867  K *= 0.5 * SQR(omega);
868 
869  for (i=1; i < n; i++)
870  q[i] = q[i] - K * u[i];
871 
872  d[0] = A[0][0];
873  d[1] = A[1][1] - 2.0*q[1]*u[1];
874  d[2] = A[2][2] - 2.0*q[2]*u[2];
875 
876  // Calculate updated A[1][2] and store it in e[1]
877  e[1] = A[1][2] - q[1]*u[2] - u[1]*q[2];
878 
879  // Store inverse Householder transformation in Q
880  for (j=1; j < n; j++)
881  {
882  f = omega * u[j];
883  for (i=1; i < n; i++)
884  Q[i][j] = Q[i][j] - f*u[i];
885  }
886 
887  }
888  else
889  {
890  for (i=0; i < n; i++)
891  d[i] = A[i][i];
892  e[1] = A[1][2];
893  }
894 
895  for ( i=0; i<3; ++i )
896  for ( j=0; j<3; ++j )
897  A[j][i] = Q[j][i];
898 
899 }

914 vector<double> dsyevq3(Matrix& A)
915 {
916  const int n(3);
917  vector<double> e(3); // The third element is used only as temporary workspace
918  double g, r, p, f, b, s, c, t; // Intermediate storage
919  int nIter;
920  int m, k, l, i;
921  vector<double> w(3); /* Array with unordered eigenvalues */
922 
923  // Transform A to real tridiagonal form by the Householder method
924  dsytrd3(A, w, e);
925  // Calculate eigensystem of the remaining real symmetric tridiagonal matrix
926  // with the QL method
927  //
928  // Loop over all off-diagonal elements
929  for (l=0; l < n-1; l++)
930  {
931  nIter = 0;
932  while (1)
933  {
934  // Check for convergence and exit iteration loop if off-diagonal
935  // element e(l) is zero
936  for (m=l; m <= n-2; m++)
937  {
938  g = fabs(w[m])+fabs(w[m+1]);
939  if (fabs(e[m]) + g == g)
940  break;
941  }
942  if (m == l)
943  break;
944 
945  if (nIter++ >= 30)
946  return w;
947 
948  // Calculate g = d_m - k
949  g = (w[l+1] - w[l]) / (e[l] + e[l]);
950  r = sqrt(SQR(g) + 1.0);
951  if (g > 0)
952  g = w[m] - w[l] + e[l]/(g + r);
953  else
954  g = w[m] - w[l] + e[l]/(g - r);
955 
956  s = c = 1.0;
957  p = 0.0;
958  for (i=m-1; i >= l; i--)
959  {
960  f = s * e[i];
961  b = c * e[i];
962  if (fabs(f) > fabs(g))
963  {
964  c = g / f;
965  r = sqrt(SQR(c) + 1.0);
966  e[i+1] = f * r;
967  c *= (s = 1.0/r);
968  }
969  else
970  {
971  s = f / g;
972  r = sqrt(SQR(s) + 1.0);
973  e[i+1] = g * r;
974  s *= (c = 1.0/r);
975  }
976 
977  g = w[i+1] - p;
978  r = (w[i] - g)*s + 2.0*c*b;
979  p = s * r;
980  w[i+1] = g + p;
981  g = c*r - b;
982 
983  // Form eigenvectors
984  for (k=0; k < n; k++)
985  {
986  t = A[k][i+1];
987  A[k][i+1] = s*A[k][i] + c*t;
988  A[k][i] = c*A[k][i] - s*t;
989  }
990  }
991  w[l] -= p;
992  e[l] = g;
993  e[m] = 0.0;
994  }
995  }
996 
997  return w;
998 }
