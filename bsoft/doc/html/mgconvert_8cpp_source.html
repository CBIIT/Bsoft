645 {

647  long i, pid;
648  Bstring pfile;
649  double defU, defV, prev_defU(0), volt, Cs, amp;
650 
651  Bstar_block* block;
652  Bstar_item* item;
653  Bstring* data;
654 
655  Bstring field_id("1"), mg_id("1");

657  Bmicrograph* mg = NULL;
658  Bparticle* partlist = NULL;
659  Bparticle* part = NULL;
660  Bparticle* prev_part = NULL;
661 
662

664  if ( ( item = item_find(block, "rlnImageName") ) ) {
665  for ( data=item->data; data; data=data->next ) {
666  pid = data->pre('@').integer();
667  pfile = data->post('@');
668  part = particle_add(&part, pid);
669  if ( !partlist ) partlist = part;
670  }
671  }
672  if ( ( item = item_find(block, "rlnDefocusU") ) ) {
673  for ( data=item->data, part=partlist; data && part; data=data->next, part=part->next )
674  part->def = data->real();
675  }
676  if ( ( item = item_find(block, "rlnDefocusV") ) ) {
677  for ( data=item->data, part=partlist; data && part; data=data->next, part=part->next )
678  part->dev = data->real();
679  }
680  if ( ( item = item_find(block, "rlnDefocusAngle") ) ) {
681  for ( data=item->data, part=partlist; data && part; data=data->next, part=part->next )
682  part->ast = data->real() * M_PI/180.0;
683  }
684  if ( ( item = item_find(block, "rlnVoltage") ) ) {
685  for ( data=item->data, part=partlist; data && part; data=data->next, part=part->next )
686  volt = data->real() * 1e3;
687  }
688  if ( ( item = item_find(block, "rlnSphericalAberration") ) ) {
689  for ( data=item->data, part=partlist; data && part; data=data->next, part=part->next )
690  Cs = data->real() * 1e7;
691  }
692  if ( ( item = item_find(block, "rlnAmplitudeContrast") ) ) {
693  for ( data=item->data, part=partlist; data && part; data=data->next, part=part->next )
694  amp = data->real();
695  }
696  }
697 
698  if ( verbose )
699  cout << "Particles: " << pid << endl;
700 
701  // Set up the micrographs
702  for ( i=1, part = partlist; part; part = part->next ) {
703  defU = part->def;
704  defV = part->dev;
705  part->def = (defU + defV)/2;
706  part->dev = (defU - defV)/2;
707  if ( defU != prev_defU ) {
708  mg_id = Bstring(i++, "%04d");
709  mg = micrograph_add(&mg, mg_id);
710  if ( !field->mg ) field->mg = mg;
711  mg->fpart = pfile;
712  mg->ctf = new CTFparam;
713  mg->ctf->volt(volt);
714  mg->ctf->Cs(Cs);
715  mg->ctf->amp_fac(amp);
716  mg->ctf->defocus_average(part->def);
717  mg->ctf->defocus_deviation(part->dev);
718  mg->ctf->astigmatism_angle(part->ast);
719  mg->ctf->zero(1);
720  mg->part = part;
721  if ( prev_part ) prev_part->next = NULL;
722  }
723  prev_defU = defU;
724  prev_part = part;
725  }
726 
727  if ( verbose )
728  cout << "Micrographs: " << mg_id.integer() << endl;
729 
730  // Reconfigure the particle files
731  unsigned long n1(0), n2(0);
732  Bimage* p = read_img(pfile, 1, -1);
733  Bimage* pex;
734 
735  for ( mg = field->mg; mg; mg = mg->next ) {
736  for ( part = mg->part; part->next; part = part->next ) ;
737  n1 = mg->part->id - 1;
738  n2 = part->id - 1;
739  for ( pid=1, part = mg->part; part; part = part->next, pid++ ) {
740  part->id = pid;
741  part->ori = p->size()/2;
742  }
743  if ( verbose )
744  cout << "Extracting sub-images " << n1 << " - " << n2 << endl;
745  pex = p->extract(n1, n2);
746  pex->origin(p->size()/2);
747  mg->fpart = pfile.pre_rev('.') + "_" + mg->id + ".pif";
748  mg->pixel_size = p->sampling().x();
749  mg->box_size = p->size();
750  write_img(mg->fpart, pex);
751  delete pex;
752  }
753 
754  delete p;
755 
756  return err;
757 }
