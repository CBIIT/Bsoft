322 ## @brief Draws a box or bad area
323 #
324 # Draws a box and circle with the current box size and number
325 # for a selected box, or draws a red circle with the current bad size
326 # for a selected bad area.
327 #
328 # @param id Box or bad area identifier.
329 
330 proc boxDraw { id } {
331  global project_item theimg
332  global box_selected box_color_mode
333  global show_box_labels box_color box_select_color bad_color bad_select_color
334  global fom_min fom_max
335  global sel_min sel_max
336  if ![winfo exists .wbox] { return }
337  if { $id == 0 } { return }
338  set c [getImageCanvas $theimg]
339  set wc [getControlWindow $theimg]
340  set n [$wc.image.scale get]
341  set z [$wc.slice.scale get]
342  set mg_item [micrographItem $n]
343  set loc [Bmg box $mg_item location $id]
344 # puts "Location for $mg_item $id: $loc"
345  if { [lindex $loc 0] < 0 } {
346  puts "ID $id not found!"
347  return
348  }
349  set scale [$wc.scale.scale get]
350  set scale_1 [expr $scale - 1]
351  set yi [yFlip [lindex $loc 1]]
352  set xr [expr [lindex $loc 0] * $scale ]
353  set yr [expr $yi * $scale ]
354  set zr [expr abs([lindex $loc 2] - $z) ]
355  if { $id > 0 } {
356  set box_radius_x [expr [.wbox.box_size.x get] * 0.5]
357  set box_radius_y [expr [.wbox.box_size.y get] * 0.5]
358  set box_radius_z [expr [.wbox.box_size.z get] * 0.5]
359  if { $zr > $box_radius_z } { return }
360  set zr2 1
361  if { $box_radius_z > 0 } {
362  set zr [expr $zr*1.0/$box_radius_z]
363  set zr2 [expr sqrt(1 - $zr*$zr)]
364  }
365 # puts "$xr $yr"
366  set box_radius_x [expr $box_radius_x * $scale ]
367  set box_radius_y [expr $box_radius_y * $scale ]
368  set oval_radius_x [expr $box_radius_x*$zr2]
369  set oval_radius_y [expr $box_radius_y*$zr2]
370  set xmin [expr $xr - $box_radius_x ]
371  set xmax [expr $xr + $box_radius_x + $scale_1 ]
372  set ymin [expr $yr - $box_radius_y ]
373  set ymax [expr $yr + $box_radius_y + $scale_1 ]
374  set cxmin [expr $xr - $oval_radius_x ]
375  set cxmax [expr $xr + $oval_radius_x + $scale_1 ]
376  set cymin [expr $yr - $oval_radius_y ]
377  set cymax [expr $yr + $oval_radius_y + $scale_1 ]
378  set idtag [format "box%d" $id]
379  if { [string compare $box_color_mode "FOM"] == 0 } {
380  set fom [Bmg box $project_item fom $id]
381  set color [spectrum $fom $fom_min $fom_max]
382  } elseif { [string compare $box_color_mode "Selection"] == 0 } {
383  set sel [Bmg box $project_item select $id]
384  set color [spectrum $sel $sel_min $sel_max]
385  } elseif { [lsearch $box_selected $id] >= 0 } {
386  set color $box_select_color;
387  } else {
388  set color $box_color;
389  }
390  $c create rect $xmin $ymin $xmax $ymax -outline $color -width 1 -tags [list box $idtag]
391  $c create oval $cxmin $cymin $cxmax $cymax -outline $color -width 1 -tags [list box $idtag]
392  if { $show_box_labels } {
393  $c create text $xr $yr -text $id -fill $color -tags [list box $idtag]
394  }
395  } elseif { $id < 0 } {
396  set bad_radius [.wbox.bad_radius.e get]
397  if { $zr > $bad_radius } { return }
398  set bad_radius [expr $scale * sqrt($bad_radius*$bad_radius - $zr*$zr)]
399  set xmin [expr $xr - $bad_radius ]
400  set xmax [expr $xr + $bad_radius + $scale_1 ]
401  set ymin [expr $yr - $bad_radius ]
402  set ymax [expr $yr + $bad_radius + $scale_1 ]
403  if { [lsearch $box_selected $id] >= 0 } {
404  set color $bad_select_color;
405  } else {
406  set color $bad_color;
407  }
408  $c create oval $xmin $ymin $xmax $ymax -outline $color -width 1 -tags bad
409  }
410 }
