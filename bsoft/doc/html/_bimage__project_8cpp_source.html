416 /*
417 @author Bernard Heymann and Peter Leong
418 @brief Back-projects a 2D image into a 3D volume.
419 @param *p a 2D image.
420 @param resolution high resolution limit (angstrom).
421 @param axis tilt axis for single tilt series (radians).
422 @param planf 2D forward Fourier transform plan.
423 @param planb 2D backward Fourier transform plan.
424 @return int 0.
425 
426  Requirements:
427  The image must be 2D with the orientation parameters in
428  the sub-image structure.
429  The orientation parameters are defined as those giving the projection,
430  and is reversed for backprojection.
431  The 2D image is converted to floating point and rescaled to an average
432  of zero and a standard deviation of one. The 3D map is traversed and
433  the density added from the 2D image projected according to the orientation
434  parameters. The origin of the new map is in the center of the volume.
435 
436 **/
437 int Bimage::back_project(Bimage* p, double resolution, double axis,
438  fft_plan planf, fft_plan planb)
439 {
440  if ( !p->data_pointer() ) return 0;
441 
442  p->change_type(Float);
443  if ( axis ) p->fspace_weigh_ramp(resolution, axis, planf, planb);
444  else p->fspace_weigh_ramp(resolution, planf, planb);
445 
446  p->statistics();
447  p->rescale_to_avg_std(0, 1);
448  p->calculate_background();
449 
450  long i, xx, yy, zz;
451  double fill(p->background(long(0)));
452  Vector3<double> oldorigin(p->image->origin()), neworigin(image->origin());
453  Vector3<double> oldcoor, newcoor;
454  Vector3<double> scale(p->image->magnification(), p->image->magnification(), 1);
455 
456  // The rotation matrix is transposed because the orientation parameters
457  // give the orientation as rotated from the reference view
458  Matrix3 mat = p->image->view().matrix();
459  mat = mat.transpose();
460  mat = scale*mat;
461 // cout << mat << endl;
462 
463  if ( verbose & VERB_LABEL )
464  cout << "Backprojecting" << endl << endl;
465 
466  for ( i=zz=0; zz<z; zz++ ) {
467  newcoor[2] = zz - neworigin[2];
468  for ( yy=0; yy<y; yy++ ) {
469  newcoor[1] = yy - neworigin[1];
470  for ( xx=0; xx<x; xx++, i++ ) {
471  newcoor[0] = xx - neworigin[0];
472  oldcoor = mat * newcoor + oldorigin;
473  oldcoor[2] = 0;
474  add(i, p->interpolate(oldcoor, 0, fill));
475  }
476  }
477  }

479  return 0;
