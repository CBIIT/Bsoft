639  delete[] cc;

644 /*
645 @brief Searches a 2D/3D density map for a template using a specific view.
646 @param *p the Fourier transformed image.
647 @param *ptemp the Fourier transformed template to be searched for.
648 @param view view.
649 @param *cc interpolation kernel.
650 @param hires high resolution limit.
651 @param lores low resolution limit.
652 @param *pmask mask for cross-correlation (ignored if NULL).
653 @return double correlation coefficient.
654 
655  The template is rotated to the view and cross-correlated to find
656  a set of high-scoring fits.
657  The views must be calculated externally to allow for custom sets.
658 **/
659 double img_fcc_for_view(Bimage* p, Bimage* ptemp, View view,
660  FSI_Kernel* kernel, double hires, double lores, Bimage* pmask)
661 {
662  Matrix3 mat = view.matrix();
663 // mat = mat.transpose();
664 
665  long i, n(0);
666  double minrad = (lores>0)? ptemp->real_size()[0]/lores: 0;
667  double maxrad = (hires>0)? ptemp->real_size()[0]/hires: ptemp->sizeX()/2;
668  if ( maxrad > ptemp->sizeX()/2 ) maxrad = ptemp->sizeX()/2;
669 
670 // cout << "minrad=" << minrad << " maxrad=" << maxrad << endl;
671 
672  Complex<double> v1, v2, cv;
673  Vector3<double> c, cr;
674  Vector3<long> min = Vector3<int>((int) (-maxrad - 0.5), (int) (-maxrad - 0.5), (int) (-maxrad - 0.5));
675  Vector3<long> max = Vector3<int>((int) (maxrad + 0.5), (int) (maxrad + 0.5), (int) (maxrad + 0.5));
676  Vector3<long> ci;
677  double d, pwr1(0), pwr2(0), cc(0);
678 
679 // cout << "min=" << min << " max=" << max << endl;
680 
681  for ( c[2]=min[2]; c[2]<=max[2]; c[2]+=1 ) {
682  for ( c[1]=min[1]; c[1]<=max[1]; c[1]+=1 ) {
683  for ( c[0]=min[0]; c[0]<=max[0]; c[0]+=1 ) {
684  d = c.length();
685  if ( d >= minrad && d <= maxrad ) {
686  ci = c;
687  if ( ci[0] < 0 ) ci[0] += p->sizeX();
688  if ( ci[1] < 0 ) ci[1] += p->sizeY();
689  if ( ci[2] < 0 ) ci[2] += p->sizeZ();
690  i = (ci[2]*p->sizeY() + ci[1])*p->sizeX() + ci[0];
691  if ( !pmask || ( pmask && (*pmask)[i] ) ) {
692  v1 = p->complex(i);
693  cr = mat * c;
694  v2 = ptemp->fspace_interpolate(0, cr, kernel);
695  pwr1 += v1.power();
696  pwr2 += v2.power();
697  cv += v1*v2.conj();
698  n++;
699  }
700  }
701  }
702  }
703  }
704 
705  d = pwr1*pwr2;
706  if ( d > 0 ) cc = cv.amp()/sqrt(d);
707 
708  if ( verbose & VERB_FULL )
709  cout << view << tab << n << tab << cc << endl;
710 
711  return cc;
712 }

727  a set of high-scoring fits.

729  The shift is encoded in the image origin.
730 **/
731 double img_cc_for_view(Bimage* p, Bimage* ptemp, View view,
732  FSI_Kernel* kernel, double hires, double lores, double searchrad,
733  Bimage* pmask)
734 {

736 // mat = mat.transpose();
737

740  double maxrad = ptemp->real_size()[0]/hires;
741  if ( maxrad > ptemp->sizeX()/2 ) maxrad = ptemp->sizeX()/2;
742 
743  Complex<double> v1, v2;
744  Vector3<double> c, cr;
745  Vector3<long> min = Vector3<int>((int) (-maxrad - 0.5), (int) (-maxrad - 0.5), (int) (-maxrad - 0.5));
746  Vector3<long> max = Vector3<int>((int) (maxrad + 0.5), (int) (maxrad + 0.5), (int) (maxrad + 0.5));
747  Vector3<long> ci;
748  double d, pwr1(0), pwr2(0);
749 
750  Bimage* pcc = new Bimage(Float, TComplex, ptemp->sizeX(),
751  ptemp->sizeY(), ptemp->sizeZ(), 1);
752 
753  for ( c[2]=min[2]; c[2]<=max[2]; c[2]+=1 ) {
754  for ( c[1]=min[1]; c[1]<=max[1]; c[1]+=1 ) {
755  for ( c[0]=min[0]; c[0]<=max[0]; c[0]+=1 ) {
756  d = c.length();
757  if ( d >= minrad && d <= maxrad ) {
758  ci = c;
759  if ( ci[0] < 0 ) ci[0] += p->sizeX();
760  if ( ci[1] < 0 ) ci[1] += p->sizeY();
761  if ( ci[2] < 0 ) ci[2] += p->sizeZ();
762  i = (ci[2]*p->sizeY() + ci[1])*p->sizeX() + ci[0];

764  v1 = p->complex(i);

766  v2 = ptemp->fspace_interpolate(0, cr, kernel);
767  pwr1 += v1.power();
768  pwr2 += v2.power();
769  pcc->set(i, v1*v2.conj());
770  }
771  }
772  }
773  }
774  }
775 
776  double scale = 1/sqrt(pwr1*pwr2);
777 
778  for ( i=0; i<pcc->data_size(); i++ ) pcc->set(i, (*pcc)[i] * scale);
779 
780  pcc->fft(FFTW_BACKWARD, 0);
781 
782  pcc->complex_to_real();
783 
784  pcc->find_peak(searchrad, 0);
785  double cc = pcc->image->FOM();
786 
787  pcc->refine_peak();
788 
789 // shift = mat * shift;
790 
791  if ( verbose & VERB_FULL )
792  cout << view << tab << pcc->image->origin() << tab << cc << endl;
793 
794  p->origin(ptemp->image->origin() + pcc->image->origin());
795 
796  delete pcc;
797 
798  return cc;
799 }
