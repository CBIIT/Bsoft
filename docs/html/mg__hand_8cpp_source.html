686 /*
687 @brief Computes statistics from data collected in Hand_Stats structure.
688 @param *stats Handedness statistical parameters
689 @param diff_out Switch for measured-predicted angle output
690 @return int 0.
691 
692  Computes average and standard deviation for view1 projection vs.
693  image1, handA-view2 projection vs. image2, and handB-view2 projection
694  vs. image2 correlation coefficients. Also computes these parameters
695  for differences between measured and predicted angles, if user
696  desires.
697 **/
698 int hand_compute_stats(Hand_Stats* stats, int diff_out)
699 {
700  int N; // Number of data points
701  double d;
702 
703  N = stats->CountA + stats->CountB + stats->Tie; // total number of selected particles
704  if (N == 0) return -1; // no data = no statistics
705 
706  // Averages
707  stats->Avg1A = stats->Sum_1A/N;
708  stats->Avg2A = stats->Sum_2A/N;
709  stats->Avg2B = stats->Sum_2B/N;
710 
711  if (diff_out > 0) { // Absolute differences: measured angles - predicted angles
712  stats->AvgAbsDiff_2Aphi = stats->Sum_AbsDiffPhi2A/N;
713  stats->AvgAbsDiff_2Athe = stats->Sum_AbsDiffThe2A/N;
714  stats->AvgAbsDiff_2Apsi = stats->Sum_AbsDiffPsi2A/N;
715  stats->AvgAbsDiff_2Bphi = stats->Sum_AbsDiffPhi2B/N;
716  stats->AvgAbsDiff_2Bthe = stats->Sum_AbsDiffThe2B/N;
717  stats->AvgAbsDiff_2Bpsi = stats->Sum_AbsDiffPsi2B/N;
718  }
719 
720  // Standard deviations
721  if (N < 2) { // prevents divide-by-zero errors if only one data point
722  stats->Std1A = 0; stats->Std2A = 0; stats->Std2B = 0;
723  stats->StdAbsDiff_2Aphi = 0; stats->StdAbsDiff_2Athe = 0; stats->StdAbsDiff_2Apsi = 0;
724  stats->StdAbsDiff_2Bphi = 0; stats->StdAbsDiff_2Bthe = 0; stats->StdAbsDiff_2Bpsi = 0;
725  } else {
726  stats->Std1A = sqrt( (stats->Sum_1Asq - (stats->Sum_1A*stats->Sum_1A)/N) / (N - 1) );
727  stats->Std2A = sqrt( (stats->Sum_2Asq - (stats->Sum_2A*stats->Sum_2A)/N) / (N - 1) );
728  stats->Std2B = sqrt( (stats->Sum_2Bsq - (stats->Sum_2B*stats->Sum_2B)/N) / (N - 1) );
729  if (diff_out > 0) {
730  d = stats->Sum_AbsDiffPhi2Asq - (stats->Sum_AbsDiffPhi2A*stats->Sum_AbsDiffPhi2A)/N;
731  if ( d > 0 ) stats->StdAbsDiff_2Aphi = sqrt( d / (N-1) );
732  else stats->StdAbsDiff_2Aphi = 0;
733  d = stats->Sum_AbsDiffPhi2Asq - (stats->Sum_AbsDiffThe2Asq*stats->Sum_AbsDiffThe2A)/N;
734  if ( d > 0 ) stats->StdAbsDiff_2Athe = sqrt( d / (N-1) );
735  else stats->StdAbsDiff_2Athe = 0;
736  d = stats->Sum_AbsDiffPsi2Asq - (stats->Sum_AbsDiffPsi2A*stats->Sum_AbsDiffPsi2A)/N;
737  if ( d > 0 ) stats->StdAbsDiff_2Apsi = sqrt( d / (N-1) );
738  else stats->StdAbsDiff_2Apsi = 0;
739  d = stats->Sum_AbsDiffPhi2Bsq - (stats->Sum_AbsDiffPhi2B*stats->Sum_AbsDiffPhi2B)/N;
740  if ( d > 0 ) stats->StdAbsDiff_2Bphi = sqrt( d / (N-1) );
741  else stats->StdAbsDiff_2Bphi = 0;
742  d = stats->Sum_AbsDiffThe2Bsq - (stats->Sum_AbsDiffThe2B*stats->Sum_AbsDiffThe2B)/N;
743  if ( d > 0 ) stats->StdAbsDiff_2Bthe = sqrt( d / (N-1) );
744  else stats->StdAbsDiff_2Bthe = 0;
745  d = stats->Sum_AbsDiffPsi2Bsq - (stats->Sum_AbsDiffPsi2B*stats->Sum_AbsDiffPsi2B)/N;
746  if ( d > 0 ) stats->StdAbsDiff_2Bpsi = sqrt( d / (N-1) );
747  else stats->StdAbsDiff_2Bpsi = 0;
748  }
749  }
750 
751  return 0;
752 }
