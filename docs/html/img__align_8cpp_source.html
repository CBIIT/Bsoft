19 /*
20 @brief Real space cross-correlation of two power spectrum annuli in two directions.
21 @param n number of values in the annulus.
22 @param *x first annulus.
23 @param *y second annulus.
24 @return double* array of correlation coefficients.
25 
26  The first halves of the annuli is cross-correlated forward and the second
27  halves backward, representing the correlations of the original annulus and
28  its inverse.
29 **/
30 double* cross_correlate_two_way(int n, double* x, double* y)
31 {
32  int i, j, k, n2 = n/2;
33  double sx(0), sx2(0), sy(0), sy2(0), sxy, d;
34 
35  double* cc = new double[n];
36  for ( i=0; i<n; i++ ) cc[i] = 0;
37 
38  for ( i=0; i<n2; i++ ) {
39  sx += x[i];
40  sx2 += x[i]*x[i];
41  sy += y[i];
42  sy2 += y[i]*y[i];
43  }
44 
45  d = (n2*sx2 - sx*sx)*(n2*sy2 - sy*sy);
46  if ( d > 0 ) d = sqrt(d);
47  else {
48  cerr << "Error in cross_correlate_two_way: denominator is negative!" << endl;
49  return cc;
50  }
51 
52  for ( i=0; i<n2; i++ ) {
53  sxy = 0;
54  for ( j=0, k=i; j<n2; j++, k++ ) {
55  if ( k >= n ) k = 0;
56  sxy += x[j]*y[k];
57  }
58  cc[i] = (n2*sxy - sx*sy)/d;
59  }
60 
61  for ( i=n2; i<n; i++ ) {
62  sxy = 0;
63  for ( j=n2, k=i; j<n; j++, k-- ) {
64  if ( k < 0 ) k = n - 1;
65  sxy += x[j]*y[k];
66  }
67  cc[i] = (n2*sxy - sx*sy)/d;
68  }
69 
70  return cc;
71 }
