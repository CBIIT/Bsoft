21  The first axis is along the link.
22  The second axis is the average of the component normals, usually directed
23  away from the molecule group center.
24  The third axes is generated as the cross product of the first two. i.e.,
25  perpendicular to both.
26  A new second axis is generated as the cross product between the other two,
27  ensuring that all the axes are orthogonal.

29 Matrix3 get_link_matrix(Blink* link)
30 {
31  Matrix3 mat;
32  Vector3<double> u, v, w;
33 
34  u = link->comp[1]->location() - link->comp[0]->location();
35  w = link->comp[0]->view().vector3();
36  w += link->comp[1]->view().vector3();
37  v = w.cross(u);
38  w = u.cross(v);
39  u.normalize();
40  v.normalize();
41  w.normalize();

45 // mat[i+6] = w[i];

48  mat[1] = v;
49  mat[2] = w;
50  mat = mat.transpose();
51 // cout << mat << endl;
