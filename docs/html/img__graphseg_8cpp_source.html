165 {

176  if ( u != v ) {
177  d2 = region[u].a - region[v].a; d2 *= d2;
178  s1 = region[u].s; s2 = region[v].s;
179 // ln1 = lnd + log(1.0 + s1) * ((g < s1)? g: s1);
180 // ln2 = lnd + log(1.0 + s2) * ((g < s2)? g: s2);
181  ln1 = lnd + log(1.0 + s1);
182  ln2 = lnd + log(1.0 + s2);
183  if ( d2 < fac * (ln1/s1 + ln2/s2) )
