178 int img_pack_into_image(Bimage* p, Bimage* ppatch, Vector3<int> start, Vector3<int> overlap)
179 {
180  long i, j, c, x, y, z, xx, yy, zz;
181  Vector3<double> w;
182 
183  overlap *= 0.5;
184 
185  if ( verbose & VERB_PROCESS )
186  cout << "Placing tile of size " << p->size() << " at " << start << endl << endl;
187 
188  for ( z=0; z<p->sizeZ(); z++ ) {
189  zz = start[2] + z;
190  w[2] = 0;
191  if ( z >= overlap[2] && z < p->sizeZ() - overlap[2] ) w[2] = 1;
192  if ( zz <= overlap[2] ) w[2] = 1;
193  if ( zz >= ppatch->sizeZ() - overlap[2] ) w[2] = 1;
194  for ( y=0; y<p->sizeY(); y++ ) {
195  yy = start[1] + y;
196  w[1] = 0;
197  if ( y >= overlap[1] && y < p->sizeY() - overlap[1] ) w[1] = 1;
198  if ( yy <= overlap[1] ) w[1] = 1;
199  if ( yy >= ppatch->sizeY() - overlap[1] ) w[1] = 1;
200  for ( x=0; x<p->sizeX(); x++ ) {
201  xx = start[0] + x;
202  w[0] = 0;
203  if ( x >= overlap[0] && x < p->sizeX() - overlap[0] ) w[0] = 1;
204  if ( xx <= overlap[0] ) w[0] = 1;
205  if ( xx >= ppatch->sizeX() - overlap[0] ) w[0] = 1;
206  if ( w.volume() ) {

208  j = ((zz*ppatch->sizeY() + yy)*ppatch->sizeX() + xx)*p->channels();
