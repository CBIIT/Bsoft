37 /* compute n^m mod p, where m >= 0 and p > 0. */
38 static int power_mod(int n, int m, int p)
39 {
40  if (m == 0)
41  return 1;
42  else if (m % 2 == 0) {
43  int x = power_mod(n, m / 2, p);
44  return MULMOD(x, x, p);
45  }
46  else
47  return MULMOD(n, power_mod(n, m - 1, p), p);
48 }

50 /*
51  * Find the period of n in the multiplicative group mod p (p prime).
52  * That is, return the smallest m such that n^m == 1 mod p.
53  */
54 static int period(int n, int p)
55 {
56  int prod = n, period = 1;
57 
58  while (prod != 1) {
59  prod = MULMOD(prod, n, p);
60  ++period;
61  if (prod == 0)
62  fftw_die("non-prime order in Rader\n");
63  }
64  return period;
65 }

67 /* find a generator for the multiplicative group mod p, where p is prime */
68 static int find_generator(int p)
69 {
70  int g;
71 
72  for (g = 1; g < p; ++g)
73  if (period(g, p) == p - 1)
74  break;
75  if (g == p)
76  fftw_die("couldn't find generator for Rader\n");
77  return g;
78 }

82 static fftw_rader_data *create_rader_aux(int p, int flags)
83 {
84  fftw_complex *omega, *work;
85  int g, ginv, gpower;
86  int i;
87  FFTW_TRIG_REAL twoPiOverN;
88  fftw_real scale = 1.0 / (p - 1); /* for convolution */
89  fftw_plan plan;
90  fftw_rader_data *d;
91 
92  if (p < 2)
93  fftw_die("non-prime order in Rader\n");
94 
95  flags &= ~FFTW_IN_PLACE;
96 
97  d = (fftw_rader_data *) fftw_malloc(sizeof(fftw_rader_data));
98 
99  g = find_generator(p);
100  ginv = power_mod(g, p - 2, p);
101 
102  omega = (fftw_complex *) fftw_malloc((p - 1) * sizeof(fftw_complex));
103 
104  plan = fftw_create_plan(p - 1, FFTW_FORWARD,
105  flags & ~FFTW_NO_VECTOR_RECURSE);
106 
107  work = (fftw_complex *) fftw_malloc((p - 1) * sizeof(fftw_complex));
108 
109  twoPiOverN = FFTW_K2PI / (FFTW_TRIG_REAL) p;
110  gpower = 1;
111  for (i = 0; i < p - 1; ++i) {
112  c_re(work[i]) = scale * FFTW_TRIG_COS(twoPiOverN * gpower);
113  c_im(work[i]) = FFTW_FORWARD * scale * FFTW_TRIG_SIN(twoPiOverN
114  * gpower);
115  gpower = MULMOD(gpower, ginv, p);
116  }
117 
118  /* fft permuted roots of unity */
119  fftw_executor_simple(p - 1, work, omega, plan->root, 1, 1,
120  plan->recurse_kind);
121 
122  fftw_free(work);
123 
124  d->plan = plan;
125  d->omega = omega;
126  d->g = g;
127  d->ginv = ginv;
128  d->p = p;
129  d->flags = flags;
130  d->refcount = 1;
131  d->next = NULL;
132 
133  d->cdesc = (fftw_codelet_desc *) fftw_malloc(sizeof(fftw_codelet_desc));
134  d->cdesc->name = NULL;
135  d->cdesc->codelet = NULL;
136  d->cdesc->size = p;
137  d->cdesc->dir = FFTW_FORWARD;
138  d->cdesc->type = FFTW_RADER;
139  d->cdesc->signature = g;
140  d->cdesc->ntwiddle = 0;
141  d->cdesc->twiddle_order = NULL;
142  return d;
143 }

147 static fftw_rader_data *fftw_create_rader(int p, int flags)
148 {
149  fftw_rader_data *d = fftw_rader_top;
150 
151  flags &= ~FFTW_IN_PLACE;
152  while (d && (d->p != p || d->flags != flags))
153  d = d->next;
154  if (d) {
155  d->refcount++;
156  return d;
157  }
158  d = create_rader_aux(p, flags);
159  d->next = fftw_rader_top;
160  fftw_rader_top = d;
161  return d;

166 /* Compute the prime FFTs, premultiplied by twiddle factors. Below, we
167  * extensively use the identity that fft(x*)* = ifft(x) in order to
168  * share data between forward and backward transforms and to obviate
169  * the necessity of having separate forward and backward plans. */
170 
171 void fftw_twiddle_rader(fftw_complex *A, const fftw_complex *W,
172  int m, int r, int stride,
173  fftw_rader_data * d)
174 {
175  fftw_complex *tmp = (fftw_complex *)
176  fftw_malloc((r - 1) * sizeof(fftw_complex));
177  int i, k, gpower = 1, g = d->g, ginv = d->ginv;
178  fftw_real a0r, a0i;
179  fftw_complex *omega = d->omega;
180 
181  for (i = 0; i < m; ++i, A += stride, W += r - 1) {
182  /*
183  * Here, we fft W[k-1] * A[k*(m*stride)], using Rader.
184  * (Actually, W is pre-permuted to match the permutation that we
185  * will do on A.)
186  */
187 
188  /* First, permute the input and multiply by W, storing in tmp: */
189  /* gpower == g^k mod r in the following loop */
190  for (k = 0; k < r - 1; ++k, gpower = MULMOD(gpower, g, r)) {
191  fftw_real rA, iA, rW, iW;
192  rW = c_re(W[k]);
193  iW = c_im(W[k]);
194  rA = c_re(A[gpower * (m * stride)]);
195  iA = c_im(A[gpower * (m * stride)]);
196  c_re(tmp[k]) = rW * rA - iW * iA;
197  c_im(tmp[k]) = rW * iA + iW * rA;
198  }
199 
200  WHEN_DEBUG( {
201  if (gpower != 1)
202  fftw_die("incorrect generator in Rader\n");
203  }
204  );
205 
206  /* FFT tmp to A: */
207  fftw_executor_simple(r - 1, tmp, A + (m * stride),
208  d->plan->root, 1, m * stride,
209  d->plan->recurse_kind);
210 
211  /* set output DC component: */
212  a0r = c_re(A[0]);
213  a0i = c_im(A[0]);
214  c_re(A[0]) += c_re(A[(m * stride)]);
215  c_im(A[0]) += c_im(A[(m * stride)]);
216 
217  /* now, multiply by omega: */
218  for (k = 0; k < r - 1; ++k) {
219  fftw_real rA, iA, rW, iW;
220  rW = c_re(omega[k]);
221  iW = c_im(omega[k]);
222  rA = c_re(A[(k + 1) * (m * stride)]);
223  iA = c_im(A[(k + 1) * (m * stride)]);
224  c_re(A[(k + 1) * (m * stride)]) = rW * rA - iW * iA;
225  c_im(A[(k + 1) * (m * stride)]) = -(rW * iA + iW * rA);
226  }
227 
228  /* this will add A[0] to all of the outputs after the ifft */
229  c_re(A[(m * stride)]) += a0r;
230  c_im(A[(m * stride)]) -= a0i;
231 
232  /* inverse FFT: */
233  fftw_executor_simple(r - 1, A + (m * stride), tmp,
234  d->plan->root, m * stride, 1,
235  d->plan->recurse_kind);
236 
237  /* finally, do inverse permutation to unshuffle the output: */
238  for (k = 0; k < r - 1; ++k, gpower = MULMOD(gpower, ginv, r)) {
239  c_re(A[gpower * (m * stride)]) = c_re(tmp[k]);
240  c_im(A[gpower * (m * stride)]) = -c_im(tmp[k]);
241  }
242 
243  WHEN_DEBUG( {
244  if (gpower != 1)
245  fftw_die("incorrect generator in Rader\n");
246  }
247  );
248 
249  }
250 
251  fftw_free(tmp);
252 }

337 /*
338  * Make an FFTW_RADER plan node. Note that this function must go
339  * here, rather than in putils.c, because it indirectly calls the
340  * fftw_planner. If we included it in putils.c, which is also used
341  * by rfftw, then any program using rfftw would be linked with all
342  * of the FFTW codelets, even if they were not needed. I wish that the
343  * darn linkers operated on a function rather than a file granularity.
344  */
345 fftw_plan_node *fftw_make_node_rader(int n, int size, fftw_direction dir,
346  fftw_plan_node *recurse,
347  int flags)
348 {
349  fftw_plan_node *p = fftw_make_node();
350 
351  p->type = FFTW_RADER;
352  p->nodeu.rader.size = size;
353  p->nodeu.rader.codelet = dir == FFTW_FORWARD ?
354  fftw_twiddle_rader : fftwi_twiddle_rader;
355  p->nodeu.rader.rader_data = fftw_create_rader(size, flags);
356  p->nodeu.rader.recurse = recurse;
357  fftw_use_node(recurse);
358 
359  if (flags & FFTW_MEASURE)
360  p->nodeu.rader.tw =
361  fftw_create_twiddle(n, p->nodeu.rader.rader_data->cdesc);
362  else
363  p->nodeu.rader.tw = 0;
364  return p;
365 }
