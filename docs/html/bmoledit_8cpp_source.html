253 {

264 
265  for ( mol = molgroup->mol; mol; mol = mol->next ) {

267  for ( atom = res->atom; atom; atom = atom->next, n++ ) {
268  if ( atom->coord.within(start, end) ) {
269  atom->sel = 1;

272  atom->sel = 0;

274  }

279
