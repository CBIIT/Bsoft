12 ## @brief Open a text editing window
13 #
14 # @param w window name to contain ".text" window.

16 proc openTextWindow { w } {
17  global textfile
18 
19  set wn $w
20  if [string equal $w "."] { set wn "" }
21  set twin $wn.text
22 # puts "Window name: $wn"
23 
24  if { ![winfo exists $w] || $w == "." } {
25  if { $w != "." } { toplevel $w }
26 
27  set m $wn.menu
28 # puts "Text window menu: $m"
29  menu $m -tearoff 0
30  $m add cascade -menu $m.file -label "File" -underline 0
31  menu $m.file -tearoff 0
32  $w configure -menu $m
33  $m.file add command -label "Open" \
34  -command "openTextFile $twin" -underline 0
35  $m.file add command -label "Save" \
36  -command [ list "saveTextFile" $twin $textfile ] -underline 0
37  $m.file add command -label "Save as ..." \
38  -command "saveAsTextFile $twin" -underline 0
39  $m.file add command -label "Close" \
40  -command "$wn.text delete 1.0 end" -underline 0
41 
42  frame $wn.frame
43  pack $wn.frame -expand yes -fill both -padx 1 -pady 1
44  text $wn.text -width 80 -height 60 -wrap none -undo 1 -background white\
45  -xscrollcommand "$wn.xscroll set" \
46  -yscrollcommand "$wn.yscroll set" \
47  -setgrid 1 -highlightthickness 0 -pady 2 -padx 3
48  scrollbar $wn.xscroll -command "$wn.text xview" \
49  -highlightthickness 0 -orient horizontal
50  scrollbar $wn.yscroll -command "$wn.text yview" \
51  -highlightthickness 0 -orient vertical
52 
53  grid $wn.text -in $wn.frame -padx 1 -pady 1 \
54  -row 0 -column 0 -rowspan 1 -columnspan 1 -sticky news
55  grid $wn.xscroll -in $wn.frame -padx 1 -pady 1 \
56  -row 1 -column 0 -rowspan 1 -columnspan 1 -sticky news
57  grid $wn.yscroll -in $wn.frame -padx 1 -pady 1 \
58  -row 0 -column 1 -rowspan 1 -columnspan 1 -sticky news
59  grid rowconfig $wn.frame 0 -weight 1 -minsize 0
60  grid columnconfig $wn.frame 0 -weight 1 -minsize 0
61 
62  } else {
63  wm deiconify $wn
64  raise $wn
65  }
66  wm title $w "Text Editor"
67  wm iconname $w "Ted"
68  bind $w <Control-z> { $wn.text edit undo }
69  bind $w <Control-Z> { $wn.text edit redo }
70  bind $w <Command-Z> { $wn.text edit redo }
71  bind $w <Control-w> { destroy $w }
72  bind $w <Command-w> { destroy $w }
73  return $wn.text
74 }

120 proc readTextFile { wtxt filename } {

122  if { [string length $filename] < 1 } { return }

161 ## @brief Updates text originating from an open channel
162 #
163 # @param wtxt a text window
164 # @param chan the channel
165 
166 proc updateText { wtxt chan } {
167  $wtxt configure -state normal
168  $wtxt insert end [read $chan]
169  $wtxt configure -state disabled
170  $wtxt see end
171  if {[eof $chan]} {
172  puts "Closing text output channel"
173  close $chan
174  $wtxt configure -state normal
175  }
176 }

178 ## @brief Executes a command and sends the output to a channel for updating
179 #
180 # @param cmd the command (with arguments)
181 # @param logfile a text file to write to.
182 
183 proc executeCommand { cmd logfile } {
184  global textfile
185  set p [lindex $cmd 0]
186  if [ file executable $p ] {
187  set textfile $logfile
188  puts "Executing: $cmd"
189  set wtxt [showInfo "$cmd\n\n"]
190  set chan [open |[concat $cmd 2>@1]]
191  fconfigure $chan -blocking 0 -buffering line
192  fileevent $chan readable [list updateText $wtxt $chan]
193  } else {
194  puts "$p not executable!"
195  }
196 }
