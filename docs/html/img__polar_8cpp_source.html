734  y = (long) floor(ypoint);

746  }

751  pdata[j] *= psi_ratio;
752  if ( max < pdata[j] ) max = pdata[j];
753  avg[iann] += pdata[j];
754  }

761  }

764  }
765  }

769  for ( ipsi=0; ipsi<npsi; ipsi++ ) {
770  j = (n*nannuli + iann)*npsi + ipsi;
771  pdata[j] -= mov_avg[iann];
772  }
773  }
774  delete[] mov_avg;
775  }
776  }
777 
778  delete[] avg;
779

781  pps->sampling(dpsi, dann*p->sampling().x(), 1);
782  pps->size(npsi, nannuli, 1);
783  pps->page_size(npsi, nannuli, 1);
784  pps->data_type(Float);
785  pps->data_assign((unsigned char *) pdata);

788  cout << "DEBUG img_polar_power_spectrum: Calculating statistics" << endl;
789

808 Bimage* img_polar_power_spectrum(Bimage* p, double resolution, long num_angle)
809 {
810  if ( resolution < 2*p->sampling().x() ) resolution = 2*p->sampling().x();
811 
812  unsigned long max_rad = (unsigned long) (p->sizeX()*p->sampling().x()/resolution);
813  if ( max_rad > p->sizeX()/2 ) max_rad = p->sizeX()/2;
814  unsigned long max_ang = num_angle*max_rad;
815  Bimage* pps = new Bimage(Double, TSimple, max_ang, max_rad, 1, p->images());
816 
817  unsigned long i, j, n, r, a;
818  long ix, ix1, iy, iy1;
819  double ang, x, y, fx, fy;
820  double v00, v01, v10, v11;
821 
822  if ( verbose & VERB_FULL )
823  cout << "Calculating a radial power spectrum: " << max_rad << " annuli x " << max_ang << " angles" << endl;
824 
825  for ( n=0; n<pps->images(); n++ ) {
826  for ( r=1; r<max_rad; r++ ) {
827  max_ang = num_angle*r;
828  for ( a=0; a<max_ang; a++ ) {
829  j = (n*pps->sizeY() + r)*pps->sizeX() + a;
830 // ang = a*fac/r;
831  ang = a*TWOPI/max_ang;
832  x = r*cos(ang);
833  y = r*sin(ang);
834  ix = (long) floor(x);
835  iy = (long) floor(y);
836  ix1 = ix + 1;
837  iy1 = iy + 1;
838  fx = x - ix;
839  fy = y - iy;
840  if ( ix < 0 ) ix += p->sizeX();
841  if ( ix1 < 0 ) ix1 += p->sizeX();
842  if ( iy < 0 ) iy += p->sizeY();
843  if ( iy1 < 0 ) iy1 += p->sizeY();
844  i = (n*p->sizeY() + iy)*p->sizeX() + ix;
845  v00 = (p->complex(i)).amp();
846  i = (n*p->sizeY() + iy)*p->sizeX() + ix1;
847  v01 = (p->complex(i)).amp();
848  i = (n*p->sizeY() + iy1)*p->sizeX() + ix;
849  v10 = (p->complex(i)).amp();
850  i = (n*p->sizeY() + iy1)*p->sizeX() + ix1;
851  v11 = (p->complex(i)).amp();
852  pps->set(j, (v00 - fx*(v00 - v01) - fy*(v00 - v10) + fx*fy*(v00 - v01 - v10 + v11)) * r);
853  }
854  }
855  }
856 
857  return pps;
858 }
