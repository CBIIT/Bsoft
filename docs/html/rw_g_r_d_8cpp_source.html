42 unsigned char* grid_compress(Bimage* p, long& compress)
43 {
44  compress = 0;
45  if ( p->data_type() != UCharacter ) return NULL;
46  if ( !p->data_pointer() ) return NULL;

48  long i, j, cnt(0);
49  unsigned char val;
50  unsigned char* data = p->data_pointer();
51  unsigned char* buf = new unsigned char[p->alloc_size()];
52 
53  for ( i=j=0, val = data[0]; i<p->data_size(); ++i ) {
54  if ( val != data[i] || cnt > 254 ) {
55 // cout << cnt << tab << int(val) << endl;
56  buf[j++] = cnt;
57  buf[j++] = val;
58  val = data[i];
59  cnt = 1;

61  }

63  buf[j++] = cnt;
64  buf[j++] = val;
