101 template <typename T>
102 inline void GetWeightsSpline3(T w, T weights[])
103 {
104  weights[3] = (T)(1.0 / 6.0) * w * w * w;
105  weights[0] = (T)(1.0 / 6.0) + (T)(1.0 / 2.0) * w * (w - (T)1.0) - weights[3];
106  weights[2] = w + weights[0] - (T)2.0 * weights[3];
107  weights[1] = (T)1.0 - weights[0] - weights[2] - weights[3];
108 }

132 template<int upBits>
133 struct electronAdderBSpline
134 {
135  static const unsigned widthIn = 4096;
136  static const int widthOut = widthIn << upBits;
137 
138  inline static void apply(float* p, int posX, int posY, int subX, int subY)
139  {
140  int posXS = ((posX - ((subX>>1)&1)) << upBits) + (subX >> (nSubPixBits-upBits));
141  int posYS = ((posY - ((subY>>1)&1)) << upBits) + (subY >> (nSubPixBits-upBits));
142 
143  subX = (subX << upBits) & 3; //repl MASK
144  subY = (subY << upBits) & 3; //repl MASK
145  int yStart = (posYS>0)? 0 : 1; // repl other splines sizes now not taken care of...
146  int yStop = (posYS>widthOut-3)? (widthOut+1-posYS): splineSize;
147  for (int y=yStart; y<yStop; ++y)
148  {
149  int xStart = (posXS>0)? 0 : 1; // repl other splines sizes now not taken care of...
150  int xStop = (posXS>widthOut-3)? (widthOut+1-posXS): splineSize;
151  int posCur = (posYS+y-1)*widthOut + posXS-1;
152  for (int x=xStart; x<xStop; ++x)
153  {
154  p[posCur+x] += BSplineLUT[subY][y] * BSplineLUT[subX][x];
155  }
156  }
157 #ifdef SUBPIXHIST
158  TSThist[(subX^2)+(subY^2)*4] ++;
159 #endif
160  }
161 };

167 template<class operationFunc, typename ImType>
168 unsigned doDecompressImage(BitStreamer& myBitStreamer, ImType* p, unsigned w, unsigned h)
169 {
170  static const int maxVal = ((1<<nBitsRLE)-1);
171  //std::cout<<"TST"<<(superPosFunc::upBitsVal)<<", "<<w<<", "<<h<<std::endl;
172  //std::cerr<<"superfactorr"<<operationFunc::widthOut<<std::endl;
173 
174  int N = (int)(w*h);
175 
176  int symbol = (int)myBitStreamer.getBits(nBitsPerCode);
177  int value = symbol & maxVal;
178  int outCount = value;
179  unsigned nElect = 0;
180  while (outCount<N)
181  {
182  if (value < maxVal)
183  {
184  int subPix = symbol>>nBitsRLE;
185  operationFunc::apply(p, (outCount & 4095), (outCount >> 12), (subPix & 3), (subPix >> 2));
186  ++outCount;
187  ++nElect;
188  }
189  else if (no_bit_waste_on_overflow_code) //this is constant expression so I assume compiler eliminates the check
190  {
191  myBitStreamer.rewind(2*nSubPixBits); // only for no-waste EER RLE implementation.
192  }
193  symbol = (int)myBitStreamer.getBits(nBitsPerCode);
194  value = symbol & maxVal;
195  outCount += value;
196  }
197  if (outCount != N)
198  {
199  std::ostringstream oss;
200  oss << "ElectronCountedFramesDecompressor: invalid RLE decoding. resulting outCount = " << outCount << " while it should equal total # pixels = " << N;
201  //throw std::range_error(oss.str());
202  std::cerr << "Warning "<<oss.str()<<std::endl;
203  }
204  //std::cout << " nDecodedTST "<<nDecodedTST<<", nOvflTST "<<nOvflTST<<std::endl;
205  return nElect;
206 }

266 {

268  unsigned nxo = nx, nyo = ny;

276  case -16: nElect = doDecompressImage<electronSetterDelta<-4> >(myBitStreamer, p, nx, ny); break;
277  case -8: nElect = doDecompressImage<electronSetterDelta<-3> >(myBitStreamer, p, nx, ny); break;
278  case -4: nElect = doDecompressImage<electronSetterDelta<-2> >(myBitStreamer, p, nx, ny); break;
279  case -2: nElect = doDecompressImage<electronSetterDelta<-1> >(myBitStreamer, p, nx, ny); break;
280  case 1: nElect = doDecompressImage<electronSetterDelta<0> >(myBitStreamer, p, nx, ny); break;
