385 /*
386 @brief Determines if a vector is inside or outside a polyhedral structure.
387 @param vec vector to test for.
388 @param ndim number of dimensions.
389 @param nvert number of vertices.
390 @param *poly set of vertices.
391 @return double distance from closest edge (inside positive).
392 
393  Whether a point is inside or outside is based on the polygon edges.
394  First the normal to an edge is calculated and tested whether it points
395  inside or outside the polygon by finding the distance to the center:
396  dist_cent = normal . edge_vertex
397  which must be positive. Then the distance to the given point is:
398  dist = normal . (edge_vertex - point)
399  The minimum distance gives the distance of the point to the closest edge.
400  If this distance is positive, the point is inside the polygon.
401  Requirement: The polygon vertices must be given in order of connectivity.
402 **/
403 double vector3_inside_outside(Vector3<double> vec, int ndim, int nvert, Vector3<double>* poly)
404 {
405  int i, j;
406  double d, dmin;
407  Vector3<double> normal;
408 
409  // Find the closest edge
410  dmin = 1e30;
411  for ( i=0, j=1; i<nvert; i++, j++ ) {
412  if ( j >= nvert ) j = 0;
413  normal = poly[i] - poly[j];
414  normal = Vector3<double>(-normal[1], normal[0], 0);
415  normal.normalize();
416  if ( normal.scalar(poly[i]) < 0 ) normal = -normal;
417  d = normal.scalar(poly[i] - vec);
418  if ( dmin > d ) dmin = d;
419  }
420 
421  return dmin;
422 }
