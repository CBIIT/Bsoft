28  color[2] = (pColorData[0] & 0x1F);
29 
30  // Convert 5-bit channels to 8-bit channels by left shifting by three
31  // and repeating the first three bits to cover the range [0,255] with

33  // channel input bits: 4 3 2 1 0
34  // channel output bits: 4 3 2 1 0 4 3 2
35  color[0] = (color[0] << 3) | (color[0] >> 2);

43  color[0] = pColorData[2];

45  color[2] = pColorData[0];
46  break;
