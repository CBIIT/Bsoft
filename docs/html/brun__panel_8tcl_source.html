46 # constructCommand
47 # Constructs the command line from a list containing the program name,
48 # the options, and the file names
49 #
50 # Arguments:
51 # c the entry for the command
52 # f the frame for files
53 # t the text with options
54 
55 proc constructCommand { c f t } {
56  set cmd [lindex [$c get] 0]
57 
58  foreach w [winfo children $t] {
59  if [winfo exists $w.label] {
60  upvar [$w.label cget -variable] on
61  set opt [$w.label cget -text]
62  if { $on == 1 } {
63  append cmd " " $opt " " [$w.e1 get]
64  }
65  }
66  }
67 
68  foreach w [winfo children $f] {
69  upvar [$w.label cget -variable] on
70  if { $on == 1 } {
71  append cmd " " [$w.name get]
72  }
73  }
74 
75  $c delete 0 end
76  $c insert 0 $cmd
77 }

113 # parseOptionLine
114 # Parses one line starting with an option
115 #
116 # Arguments:
117 # c the entry for the command
118 # f the frame for files
119 # t the text with options
120 # line the line
121 
122 proc parseOptionLine { c f t line } {
123  global font smallfont
124  if [regexp "^--" $line] { return }
125 
126  set words [split $line]
127  set b [lindex $words 0]
128  set v [lindex $words 1]
129 
130  set d ""
131  foreach word [lrange $words 2 end] {
132  if { $word != "" } {
133  lappend d $word
134  }
135  }
136 
137  set w $t.$b
138  frame $w
139  entry $w.e1 -width [string length $v]
140  $w.e1 insert 0 $v
141 
142  checkbutton $w.label -text "$b" -variable $b -relief flat \
143  -background cyan -command "constructCommand $c $f $t"
144  $w.label deselect
145 
146  if { $v > " " } {
147  pack $w.label $w.e1 -side left -pady 1 -padx 4
148  } else {
149  pack $w.label -side left -pady 1 -padx 4
150  }
151 
152  $t window create end -window $w -padx 4 -pady 1
153  $t insert end [join $d " "]
154  $t insert end "\n"
155 
156  bind $w.e1 <KeyRelease> "constructCommand $c $f $t"
157 }

159 # parseCommand
160 # Parses a command line to fill in options
161 #
162 # Argument:
163 # t text window name
164 # cmd command line
165 
166 proc parseCommand { t cmd } {
167  set vf 0
168  foreach word $cmd {
169  if ![string first "-" $word] {
170  set vf 0
171 # set word [string trimleft $word "-"]
172  foreach w [winfo children $t] {
173  if [regexp -- $word $w] {
174  $w.label select
175  if { [$w.e1 get] > " " } {
176  set vf 1
177  }
178  break
179  }
180  }
181  } elseif { $vf } {
182  $w.e1 delete 0 end
183  $w.e1 insert 0 $word
184  set vf 0
185  }
186  }
187 }

196 proc selectProgram { w cmd } {
197  global font
198  global smallfont
199 
200  set wn $w
201  if [string equal "." $w] { set wn "" }
202 
203  set c $wn.cmd
204  set f $wn.file
205  set t $wn.txt
206  set g $wn.grid
207 
208  destroy $f
209  destroy $t
210  destroy $g
211  destroy $wn.hscroll
212  destroy $wn.vscroll
213 
214  set prog [lindex $cmd 0]
215 # puts "Program: $prog"
216  if { [ string first "brun" $cmd ] > -1 } { return }
217  if ![ file executable $prog ] { return }
218 
219  catch { exec $prog } result
220 
221  $c.e delete 0 end
222  $c.e insert 0 $cmd
223 
224  frame $f
225 
226  frame $g
227  pack $g -side bottom -expand yes -fill both -padx 1 -pady 1
228 
229  grid rowconfig $g 0 -weight 1 -minsize 0
230  grid columnconfig $g 0 -weight 1 -minsize 0
231 
232  text $t -wrap none -relief sunken -bd 2 \
233  -xscrollcommand "$wn.hscroll set" \
234  -yscrollcommand "$wn.vscroll set" \
235  -setgrid 1 -highlightthickness 0 -pady 2 -padx 3
236 
237  scrollbar $wn.hscroll -orient horizontal -command "$t xview"
238  scrollbar $wn.vscroll -orient vertical -command "$t yview"
239 
240  grid $t -padx 1 -in $g -pady 1 \
241  -row 0 -column 0 -rowspan 1 -columnspan 1 -sticky news
242  grid $wn.vscroll -in $g -padx 1 -pady 1 \
243  -row 0 -column 1 -rowspan 1 -columnspan 1 -sticky news
244  grid $wn.hscroll -in $g -padx 1 -pady 1 \
245  -row 1 -column 0 -rowspan 1 -columnspan 1 -sticky news
246 
247  $t insert end "Description:\n"
248  foreach line [split $result "\n"] {
249  if ![string first "-" $line] {
250  parseOptionLine $c.e $f $t $line
251  } elseif ![string first "Usage" $line] {
252  parseUsageLine $c.e $f $t $line
253  } else {
254  $t insert end "$line\n"
255  }
256  }
257 
258  parseCommand $t $cmd
259 }
