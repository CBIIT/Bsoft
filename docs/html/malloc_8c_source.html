44  * debugging malloc/free. Initialize every malloced and freed area to
45  * random values, just to make sure we are not using uninitialized

66 void *fftw_malloc(size_t n)
67 {
68  char *p;
69  int i;
70 
71  fftw_malloc_total += n;
72 
73  if (fftw_malloc_total > fftw_malloc_max)
74  fftw_malloc_max = fftw_malloc_total;
75 
76  p = (char *) malloc(PAD_FACTOR * n + TWOINTS);
77  if (!p)
78  fftw_die("fftw_malloc: out of memory\n");
79 
80  /* store the size in a known position */
81  ((int *) p)[0] = n;
82  ((int *) p)[1] = MAGIC;
83  for (i = 0; i < PAD_FACTOR * n; ++i)
84  p[i + TWOINTS] = (char) (i ^ 0xDEADBEEF);
85 
86  ++fftw_malloc_cnt;
87 
88  if (fftw_malloc_cnt > fftw_malloc_cnt_max)
89  fftw_malloc_cnt_max = fftw_malloc_cnt;
90 
91  /* skip the size we stored previously */
92  return (void *) (p + TWOINTS);
93 }

95 void fftw_free(void *p)
96 {
97  char *q;
98 
99  if (!p)
100  return;
101 
102  q = ((char *) p) - TWOINTS;
103  if (!q)
104  fftw_die("fftw_free: tried to free NULL+TWOINTS pointer!\n");
105 
106  {
107  int n = ((int *) q)[0];
108  int magic = ((int *) q)[1];
109  int i;
110 
111  WHEN_VERBOSE( {
112  printf("FFTW_FREE %d\n", n);
113  fflush(stdout);
114  })
115 
116  *((int *) q) = 0; /* set to zero to detect duplicate free's */
117 
118  if (magic != MAGIC)
119  fftw_die("Wrong magic in fftw_free()!\n");
120  ((int *) q)[1] = ~MAGIC;
121 
122  if (n < 0)
123  fftw_die("Tried to free block with corrupt size descriptor!\n");
124 
125  fftw_malloc_total -= n;
126 
127  if (fftw_malloc_total < 0)
128  fftw_die("fftw_malloc_total went negative!\n");
129 
130  /* check for writing past end of array: */
131  for (i = n; i < PAD_FACTOR * n; ++i)
132  if (q[i + TWOINTS] != (char) (i ^ 0xDEADBEEF)) {
133  fflush(stdout);
134  fprintf(stderr, "Byte %d past end of array has changed!\n",
135  i - n + 1);
136  fftw_die("Array bounds overwritten!\n");
137  }
138  for (i = 0; i < PAD_FACTOR * n; ++i)
139  q[i + TWOINTS] = (char) (i ^ 0xBEEFDEAD);
140 
141  --fftw_malloc_cnt;
142 
143  if (fftw_malloc_cnt < 0)
144  fftw_die("fftw_malloc_cnt went negative!\n");
145 
146  if (fftw_malloc_cnt == 0 && fftw_malloc_total > 0 ||
147  fftw_malloc_cnt > 0 && fftw_malloc_total == 0)
148  fftw_die("fftw_malloc_cnt/total not zero at the same time!\n");
149 
150  free(q);
151  }
152 }
