118 /************************************************************************
119 @Function: project_from_IMOD
120 @Description:
121  Creates a project structure using IMOD parameters.
122 @Algorithm:
123  Requirements:
124  Tilt series micrograph image (if 3D then converted to multi-2D)
125  2D transform file (.xf)
126  Tilt angle file (.tlt)
127  Calculations:
128  Tilt angle: from tilt angle file
129  Tilt axis: ta = (atan2(-A11, A12) + atan2(-A22, -A21))/2
130  Mg origin: o = on - Ad
131  where
132  A: 2x2 transformation matrix
133  ta: tilt axis angle
134  on: nominal micrograph origin (center of mg)
135  o: aligned micrograph origin
136  d: shift for micrograph
137 @Arguments:
138  Bstring* file_list list of IMOD parameter file names.
139  int flag flag to indicate conversion of the image file.
140 @Returns:
141  Bproject* new project parameter structure.
142 *************************************************************************/
143 Bproject* project_from_IMOD(Bstring* file_list, int flag)
144 {
145  if ( file_list->empty() ) {
146  cerr << "No filenames given to create a project!" << endl;
147  exit(-1);
148  }
149 
150  Bimage* p = NULL;
151  Bstring base, imgfile;
152  Bstring* thisfile = NULL;
153  Bstring ext;
154 
155  for ( thisfile = file_list; thisfile; thisfile = thisfile->next ) {
156  ext = thisfile->extension();
157  if ( ext.contains("pif") || ext.contains("mrc") || ext.contains("st") ) {
158  imgfile = *thisfile;
159  base = imgfile.base();
160  if ( ext.contains("st") ) imgfile += ":mrc";
161  p = read_img(imgfile, 0, -1);
162  if ( !p ) {
163  cerr << "Error: Image file " << imgfile << " not read!" << endl;
164  error_show("project_from_IMOD", __FILE__, __LINE__);
165  return(NULL);
166  }
167  if ( p->z > 1 ) {
168  p->slices_to_images();
169  if ( flag ) {
170  delete p;
171  p = read_img(imgfile, 1, 0);
172  p->slices_to_images();
173  imgfile = base + ".pif";
174  write_img(imgfile, p);
175  }
176  }
177  }
178  }
179 
180  if ( !p ) {
181  cerr << "Error: No image file provided!" << endl;
182  error_show("project_from_IMOD", __FILE__, __LINE__);
183  return(NULL);
184  }
185 
186  int j, n;
187  Bstring mg_id, rec_id, rec_name;
188  Bproject* project = project_init();
189  Bfield* field = field_add(&project->field, base);
190  Bmicrograph* mg = NULL;
191  Breconstruction* rec = NULL;
192  Bmarker* mark = NULL;
193 
194  for ( n=0; n<p->images(); n++ ) {
195  mg_id = base + Bstring(n+1, "_%03d");
196  mg = micrograph_add(&mg, mg_id);
197  if ( !field->mg ) field->mg = mg;
198  mg->block = n;
199  mg->img_num = n;
200  mg->fmg = imgfile;
201  mg->pixel_size = p->sampling().x();
202  mg->origin] = p->size()/2;
203  if ( verbose & VERB_FULL )
204  cout << "Creating field \"" << field->id << "\", micrograph \"" << mg->id << "\"" << endl;
205  }
206 
207  delete p;
208 
209  FILE* fd = NULL;
210  char aline[100], *aptr;
211  int mark_id, dum;
212  float x, y, z;
213  double a[4], dx, dy, ta;
214 
215  for ( thisfile = file_list; thisfile; thisfile = thisfile->next ) {
216  ext = thisfile->extension();
217  if ( ext.contains("xf") ) {
218  if ( ( fd = fopen(thisfile->c_str(), "r") ) == NULL ) {
219  cerr << "Error: A text file specifying 2D transformations must be given!" << endl;
220  return(NULL);
221  }
222  if ( verbose & VERB_PROCESS )
223  cout << "Reading 2D transformations from \"" << *thisfile << "\"" << endl;
224  for ( mg = field->mg; mg; mg = mg->next ) {
225  fscanf(fd, "%lf %lf %lf %lf %lf %lf", &a[0], &a[1], &a[2], &a[3], &dx, &dy);
226  mg->tilt_axis = (atan2(-a[0], a[1]) + atan2(-a[3], -a[2]))/2.0;
227  mg->origin[0] -= a[0]*dx + a[2]*dy;
228  mg->origin[1] -= a[1]*dx + a[3]*dy;
229  }
230  fclose(fd);
231  } else if ( ext.contains("tlt") ) {
232  if ( ( fd = fopen(thisfile->c_str(), "r") ) == NULL ) {
233  cerr << "Error: A text file specifying tilt angles must be given!" << endl;
234  return(NULL);
235  }
236  if ( verbose & VERB_PROCESS )
237  cout << "Reading tilt angles from \"" << *thisfile << "\"" << endl;
238  for ( mg = field->mg; mg; mg = mg->next ) {
239  fscanf(fd, "%lf", &ta);
240  mg->tilt_angle = ta*M_PI/180.0;
241  }
242  fclose(fd);
243  } else if ( ext.contains("xyz") ) {
244  if ( ( fd = fopen(thisfile->c_str(), "r") ) == NULL ) {
245  cerr << "Error: A text file specifying the 3D marker model must be given!" << endl;
246  return(NULL);
247  }
248  if ( verbose & VERB_PROCESS )
249  cout << "Reading the 3D marker model from \"" << *thisfile << "\"" << endl;
250  rec_id = base;
251  rec = reconstruction_add(&project->rec, rec_id);
252  rec_name = base + ".rec";
253  if ( access(rec_name.c_str(), R_OK) ) {
254  fgets(aline, 100, fd);
255  aptr = strstr(aline, "Pix");
256  sscanf(aptr, "Pix: %f Dim: %f %f\n", &rec->voxel_size, &rec->origin[0], &rec->origin[1]);
257  rec->origin /= 2;
258  rewind(fd);
259  } else {
260  rec->frec = rec_name;
261  rec_name += ":mrc";
262  p = read_img(rec_name, 0, -1);
263  rec->voxel_size = p->sampling().x();
264  rec->origin = p->size()/2;
265  delete p;
266  }
267  while ( fgets(aline, 100, fd) ) {
268  if ( sscanf(aline, "%d %f %f %f %d %d", &mark_id, &x, &y, &z, &dum, &dum) > 5 ) {
269  mark = (Bmarker *) add_item((char **) &mark, sizeof(Bmarker));
270  if ( !rec->mark ) rec->mark = mark;
271  mark->id = mark_id;
272  mark->loc = Vector3<float>(x - rec->origin.x(), y - rec->origin.y(), z);
273  mark->fom = 1;
274  mark->sel = 1;
275  }
276  }
277  fclose(fd);
278  }
279  }
280 
281  project_mg_tilt_to_matrix(project);
282 
283  return(project);

286 /************************************************************************
287 @Function: write_IMOD_parameters
288 @Description:
289  Creates IMOD parameter files from a project structure.
290 @Algorithm:
291  Writes both .xf and .tlt files.
292  Calculations:
293  Tilt angle: from micrograph tilt angle
294  Matrix: A = {cos(ta), -sin(ta), sin(ta), cos(ta)}
295  Shift: d = o - on
296  where
297  A: 2x2 transformation matrix
298  ta: tilt axis angle
299  on: nominal micrograph origin (center of mg)
300  o: aligned micrograph origin
301  d: shift for micrograph
302 @Arguments:
303  Bstring& imodfile IMOD parameter file name.
304  Bproject* project project structure.
305 @Returns:
306  int 0.
307 *************************************************************************/
308 int write_IMOD_parameters(Bstring& imodfile, Bproject* project)
309 {
310  Bstring paramfile = imodfile.base() + ".xf";
311  Bmicrograph* mg = project->field->mg;
312  Breconstruction* rec = project->rec;
313  Bmarker* mark = NULL;
314 
315  Vector3<float> d, origin = micrograph_get_nominal_origin(mg);
316 
317  FILE* fd = NULL;
318 
319  if ( ( fd = fopen(paramfile.c_str(), "w") ) == NULL ) {
320  error_show("write_IMOD_parameters", __FILE__, __LINE__);
321  return(NULL);
322  }
323 
324  if ( verbose & VERB_PROCESS )
325  cout << "Writing 2D transformations to \"" << paramfile << "\"" << endl;
326 
327  double a[4];
328 
329  for ( mg = project->field->mg; mg; mg = mg->next ) {
330  a[0] = a[3] = -sin(mg->tilt_axis);
331  a[1] = cos(mg->tilt_axis);
332  a[2] = -cos(mg->tilt_axis);
333  d = origin - mg->origin;
334  fprintf(fd, "%12.7lf%12.7lf%12.7lf%12.7lf%12.3lf%12.3lf\n", a[0], a[1],
335  a[2], a[3], a[0]*d.x() + a[1]*d.y(), a[2]*d.x() + a[3]*d.y());
336  }
337 
338  fclose(fd);
339 
340  paramfile = imodfile.base() + ".tlt";
341 
342  if ( ( fd = fopen(paramfile.c_str(), "w") ) == NULL ) {
343  error_show("write_IMOD_parameters", __FILE__, __LINE__);
344  return(NULL);
345  }
346 
347  if ( verbose & VERB_PROCESS )
348  cout << "Writing tilt angles to \"" << paramfile << "\"" << endl;
349 
350  for ( mg = project->field->mg; mg; mg = mg->next )
351  fprintf(fd, "%g\n", mg->tilt_angle*180.0/M_PI);
352 
353  fclose(fd);
354 
355  if ( rec && rec->mark ) {
356  paramfile = imodfile.base() + "fid.xyz";
357 
358  if ( ( fd = fopen(paramfile.c_str(), "w") ) == NULL ) {
359  error_show("write_IMOD_parameters", __FILE__, __LINE__);
360  return(NULL);
361  }
362 
363  if ( verbose & VERB_PROCESS )
364  cout << "Writing the 3D marker model to \"" << paramfile << "\"" << endl;
365 
366  for ( mark = rec->mark; mark; mark = mark->next ) {
367  fprintf(fd, "%4d%10.2f%10.2f%10.2f %6d %4d", mark->id,
368  mark->loc.x() + rec->origin.x(), mark->loc.y() + rec->origin.y(), mark->loc.z(), 1, mark->id);
369  if ( mark == rec->mark )
370  fprintf(fd, " Pix: %11.5f Dim: %5.0f %5.0f\n",
371  rec->voxel_size, 2*rec->origin.x(), 2*rec->origin.y());
372  else
373  fprintf(fd, "\n");
374  }
375 
376  fclose(fd);
377  }
378 
379  return(0);
