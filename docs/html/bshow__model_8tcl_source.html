280 proc convertImageToModel { x y z } {
281  global theimg
282  set ps [.pixel_size.x get]
283  set ori [Bimage get $theimg origin [.image.scale get]]
284  set x [expr ($x - [lindex $ori 0])*$ps]
285  set y [expr ($y - [lindex $ori 1])*$ps]
286  set z [expr ($z - [lindex $ori 2])*$ps]
287 # puts "1 $x $y $z"
288  return "$x $y $z"
289 }

555 proc drawSphere { c x y z rad col width lbl fill tag } {
556 # puts "drawSphere $x $y $z"
557  set scale [.scale.scale get]
558  if { $rad < 1 } { set rad 1 }
559  set zr [expr [.slice.scale get] - $z]
560  if { $zr >= $rad } { return }
561  set r [expr $rad*$rad - $zr*$zr]
562  if { $r < 0 } { return }
563  set xy_radius [expr $scale*sqrt($r)]
564  set rad [expr $rad * $scale]

566  set x [expr $x * $scale ]
567  set y [expr ($h - $y - 1) * $scale ]

570  set xmax [expr $x + $xy_radius + $scale - 1 ]

572  set ymax [expr $y + $xy_radius + $scale - 1 ]
573  set symbol "-outline"

581 proc drawLine { c x1 y1 z1 x2 y2 z2 width col tag } {
582 # puts "drawLine $x1 $y1 $z1 $x2 $y2 $z2"
583  set scale [.scale.scale get]
584  set width [expr int($width * $scale)]
585  if { $width < 1 } { set width 1 }
586  set z [.slice.scale get]
587  set h [Bimage get theimg height]
588  set xs [expr $x1 * $scale ]
589  set ys [expr ($h - $y1 - 1) * $scale ]
590  set zs $z1
591  set xe [expr $x2 * $scale ]
592  set ye [expr ($h - $y2 - 1) * $scale ]
593  set ze $z2
594  set zf -1.0
595  if { abs($ze - $zs) > 0.001 } {
596  set zf [expr ($z - $zs) / (1.0*($ze - $zs))]
597  } else {
598  if { abs($z - $zs) < 0.9 } { set zf 0.5 }
599  }
600  if { $xs == $xe && $ys == $ye } { set xe [expr $xe + 1] }
601  if { $zf >= 0.0 && $zf <= 1.0 } {
602  $c create line $xs $ys $xe $ye -fill $col -smooth yes -width $width -tags link
603  }
604 }
