230  double (funk)(Bsimplex2&))
231 {
232  long i, j, npnt(nparam + 1);
233  double psum, pnew, Rtry;
234  double fac1((1 - fac)/nparam);
235  double fac2(fac1 - fac);
236 
237  for ( j=0; j<nparam; j++ ) {
238  psum = 0;
239  for ( i=0; i<npnt; i++ ) psum += mp[i*nparam+j];
240  param[j] = mp[ihi*nparam+j];
241  pnew = psum*fac1 - mp[ihi*nparam+j]*fac2;
242  if ( lo.size() ) if ( pnew < lo[j] ) pnew = param[j];
243  if ( hi.size() ) if ( pnew > hi[j] ) pnew = param[j];
244  param[j] = pnew;
245  }
246

248 
249  if ( Rtry <= R[ihi] ) {
250  R[ihi] = Rtry;
251  for ( j=0; j<nparam; j++ ) mp[ihi*nparam+j] = param[j];
