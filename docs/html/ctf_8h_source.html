190 double calc_baseline(double s) {
191 double s2(s*s), b(0), ds;
192 switch ( bt ) {
193 case 1:
194 case 4:
195 b = base[0] + base[1]*s + base[2]*s2 + base[3]*s*s2 + base[4]*s2*s2;
196 break;
197 case 2:
198 case 5:
199 b = base[0] + base[1]*exp(base[2]*s2) + base[3]*exp(base[4]*s2);
200 break;
201 case 3:
202 case 6:
203 b = base[0] + base[1]*exp(base[2]*sqrt(s) + base[3]*s2);
204 break;
205 default: ;
206 }
207 if ( bt > 3 ) {
208// for ( int i=5; i<8; ++i ) cout << "\t" << base[i];
209// cout << endl;
210 ds = base[6] - s;
211 b += base[5] * exp(base[7]*ds*ds);
212 }
213 if ( !isfinite(b) )
214 cerr << "Error in CTFparam::calc_baseline: baseline value not finite!" << endl;
215 else if ( b <= 0 ) b = 0;
216 return b;
217 }

249 double l = lambda();
