46  * The timer keeps doubling the number of iterations
47  * until the program runs for more than FFTW_TIME_MIN
48  */
49 static double fftw_measure_runtime(fftw_plan plan,
50  fftw_complex *in, int istride,
51  fftw_complex *out, int ostride)
52 {
53  fftw_time begin, end, start;
54  double t, tmax, tmin;
55  int i, iter;
56  int n;
57  int repeat;
58  int howmany = plan->vector_size;
59 
60  n = plan->n;
61 
62  iter = 1;
63 
64  for (;;) {
65  tmin = 1.0E10;
66  tmax = -1.0E10;
67  init_test_array(in, istride, n * howmany);
68 
69  start = fftw_get_time();
70  /* repeat the measurement FFTW_TIME_REPEAT times */
71  for (repeat = 0; repeat < FFTW_TIME_REPEAT; ++repeat) {
72  begin = fftw_get_time();
73  for (i = 0; i < iter; ++i) {
74  fftw(plan, howmany, in, istride, istride,
75  out, ostride, ostride);
76  }
77  end = fftw_get_time();
78 
79  t = fftw_time_to_sec(fftw_time_diff(end, begin));
80  if (t < tmin)
81  tmin = t;
82  if (t > tmax)
83  tmax = t;
84 
85  /* do not run for too long */
86  t = fftw_time_to_sec(fftw_time_diff(end, start));
87  if (t > FFTW_TIME_LIMIT)
88  break;
89  }
90 
91  if (tmin >= FFTW_TIME_MIN)
92  break;
93 
94  iter *= 2;
95  }
96 
97  tmin /= (double) iter;
98  tmax /= (double) iter;
99 
100  return tmin;
101 }

145 static fftw_plan planner_wisdom(fftw_plan *table, int n,
146  fftw_direction dir, int flags,
147  int vector_size,
148  fftw_complex *in, int istride,
149  fftw_complex *out, int ostride)
150 {
151  fftw_plan best = (fftw_plan) 0;
152  fftw_plan_node *node;
153  int have_wisdom;
154  enum fftw_node_type wisdom_type;
155  int wisdom_signature;
156  fftw_recurse_kind wisdom_recurse_kind;
157 
158  /* see if we remember any wisdom for this case */
159  have_wisdom = fftw_wisdom_lookup(n, flags, dir, FFTW_WISDOM,
160  istride, ostride,
161  &wisdom_type, &wisdom_signature,
162  &wisdom_recurse_kind, 0);
163 
164  if (!have_wisdom)
165  return best;
166 
167  if (wisdom_type == FFTW_NOTW) {
168  FOR_ALL_CODELETS(p) {
169  if (p->dir == dir && p->type == wisdom_type) {
170  /* see if wisdom applies */
171  if (wisdom_signature == p->signature &&
172  p->size == n) {
173  node = fftw_make_node_notw(n, p);
174  best = fftw_make_plan(n, dir, node, flags,
175  p->type, p->signature,
176  FFTW_NORMAL_RECURSE,
177  vector_size);
178  fftw_use_plan(best);
179  run_plan_hooks(best);
180  return best;
181  }
182  }
183  }
184  }
185  if (wisdom_type == FFTW_TWIDDLE) {
186  FOR_ALL_CODELETS(p) {
187  if (p->dir == dir && p->type == wisdom_type) {
188 
189  /* see if wisdom applies */
190  if (wisdom_signature == p->signature &&
191  p->size > 1 &&
192  (n % p->size) == 0) {
193  fftw_plan r = planner(table, n / p->size, dir,
194  flags | FFTW_NO_VECTOR_RECURSE,
195  wisdom_recurse_kind ==
196  FFTW_VECTOR_RECURSE ?
197  p->size : vector_size,
198  in, istride, out, ostride);
199  node = fftw_make_node_twiddle(n, p,
200  r->root, flags);
201  best = fftw_make_plan(n, dir, node, flags,
202  p->type, p->signature,
203  wisdom_recurse_kind,
204  vector_size);
205  fftw_use_plan(best);
206  run_plan_hooks(best);
207  fftw_destroy_plan_internal(r);
208  return best;
209  }
210  }
211  }
212  }
213  /*
214  * BUG (or: TODO) Can we have generic wisdom? This is probably
215  * an academic question
216  */
217 
218  return best;
219 }

225 static fftw_plan planner_normal(fftw_plan *table, int n, fftw_direction dir,
226  int flags, int vector_size,
227  fftw_complex *in, int istride,
228  fftw_complex *out, int ostride)

232  fftw_plan_node *node;
233 
234  /* see if we have any codelet that solves the problem */
235  {
236  FOR_ALL_CODELETS(p) {
237  if (p->dir == dir && p->type == FFTW_NOTW) {
238  if (p->size == n) {
239  node = fftw_make_node_notw(n, p);
240  newplan = fftw_make_plan(n, dir, node, flags,
241  p->type, p->signature,
242  FFTW_NORMAL_RECURSE,
243  vector_size);

245  compute_cost(newplan, in, istride, out, ostride);
246  run_plan_hooks(newplan);
247  best = fftw_pick_better(newplan, best);
248  }
249  }
250  }
251  }
252 
253  /* Then, try all available twiddle codelets */
254  {
255  FOR_ALL_CODELETS(p) {
256  if (p->dir == dir && p->type == FFTW_TWIDDLE) {
257  if ((n % p->size) == 0 &&
258  p->size > 1 &&
259  (!best || n != p->size)) {
260  fftw_plan r = planner(table, n / p->size, dir,
261  flags | FFTW_NO_VECTOR_RECURSE,
262  vector_size,
263  in, istride, out, ostride);
264  node = fftw_make_node_twiddle(n, p,
265  r->root, flags);
266  newplan = fftw_make_plan(n, dir, node, flags,
267  p->type, p->signature,
268  FFTW_NORMAL_RECURSE,
269  vector_size);
270  fftw_use_plan(newplan);
271  fftw_destroy_plan_internal(r);
272  compute_cost(newplan, in, istride, out, ostride);
273  run_plan_hooks(newplan);
274  best = fftw_pick_better(newplan, best);
275  }
276  }
277  }
278  }
279 
280  /* try vector recursion unless prohibited by the flags: */
281  if (! (flags & FFTW_NO_VECTOR_RECURSE)) {
282  FOR_ALL_CODELETS(p) {
283  if (p->dir == dir && p->type == FFTW_TWIDDLE) {
284  if ((n % p->size) == 0 &&
285  p->size > 1 &&
286  (!best || n != p->size)) {
287  fftw_plan r = planner(table, n / p->size, dir,
288  flags | FFTW_NO_VECTOR_RECURSE,
289  p->size,
290  in, istride, out, ostride);
291  node = fftw_make_node_twiddle(n, p,
292  r->root, flags);
293  newplan = fftw_make_plan(n, dir, node, flags,
294  p->type, p->signature,
295  FFTW_VECTOR_RECURSE,
296  vector_size);
297  fftw_use_plan(newplan);
298  fftw_destroy_plan_internal(r);
299  compute_cost(newplan, in, istride, out, ostride);
300  run_plan_hooks(newplan);
301  best = fftw_pick_better(newplan, best);
302  }
303  }
304  }
305  }
306 
307  /*
308  * resort to generic or rader codelets for unknown factors
309  */
310  {
311  fftw_generic_codelet *codelet = (dir == FFTW_FORWARD ?
312  fftw_twiddle_generic :
313  fftwi_twiddle_generic);
314  int size, prev_size = 0, remaining_factors = n;
315  fftw_plan r;
316 
317  while (remaining_factors > 1) {
318  size = fftw_factor(remaining_factors);
319  remaining_factors /= size;
320 
321  /* don't try the same factor more than once */
322  if (size == prev_size)
323  continue;
324  prev_size = size;
325 
326  /* Look for codelets corresponding to this factor. */
327  {
328  FOR_ALL_CODELETS(p) {
329  if (p->dir == dir && p->type == FFTW_TWIDDLE
330  && p->size == size) {
331  size = 0;
332  break;
333  }
334  }
335  }
336 
337  /*
338  * only try a generic/rader codelet if there were no
339  * twiddle codelets for this factor
340  */
341  if (!size)
342  continue;
343 
344  r = planner(table, n / size, dir,
345  flags | FFTW_NO_VECTOR_RECURSE,
346  vector_size,
347  in, istride, out, ostride);
348 
349  /* Try Rader codelet: */
350  node = fftw_make_node_rader(n, size, dir, r->root, flags);
351  newplan = fftw_make_plan(n, dir, node, flags, FFTW_RADER, 0,
352  FFTW_NORMAL_RECURSE, vector_size);
353  fftw_use_plan(newplan);
354  compute_cost(newplan, in, istride, out, ostride);
355  run_plan_hooks(newplan);
356  best = fftw_pick_better(newplan, best);
357 
358  if (size < 100) { /*
359  * only try generic for small
360  * sizes
361  */
362  /* Try generic codelet: */
363  node = fftw_make_node_generic(n, size, codelet,
364  r->root, flags);
365  newplan = fftw_make_plan(n, dir, node, flags,
366  FFTW_GENERIC, 0,
367  FFTW_NORMAL_RECURSE, vector_size);
368  fftw_use_plan(newplan);
369  compute_cost(newplan, in, istride, out, ostride);
370  run_plan_hooks(newplan);
371  best = fftw_pick_better(newplan, best);
372  }
373  fftw_destroy_plan_internal(r);
374  }
375  }
376 
377  if (!best)
378  fftw_die("bug in planner\n");
379 
380  return best;

383 static fftw_plan planner(fftw_plan *table, int n, fftw_direction dir,
384  int flags, int vector_size,
385  fftw_complex *in, int istride,
386  fftw_complex *out, int ostride)
387 {
388  fftw_plan best = (fftw_plan) 0;
389 
390  if (vector_size > 1)
391  flags |= FFTW_NO_VECTOR_RECURSE;
392 
393  /* see if plan has already been computed */
394  best = fftw_lookup(table, n, flags, vector_size);
395  if (best) {
396  fftw_use_plan(best);
397  return best;
398  }
399  /* try a wise plan */
400  best = planner_wisdom(table, n, dir, flags, vector_size,
401  in, istride, out, ostride);
402 
403  if (!best) {
404  /* No wisdom. Plan normally. */
405  best = planner_normal(table, n, dir, flags,
406  vector_size,
407  in, istride, out, ostride);
408  }
409  if (best) {
410  fftw_insert(table, best);
411 
412  /* remember the wisdom */
413  fftw_wisdom_add(n, flags, dir, FFTW_WISDOM, istride, ostride,
414  best->wisdom_type,
415  best->wisdom_signature,
416  best->recurse_kind);
417  }
418  return best;
419 }
