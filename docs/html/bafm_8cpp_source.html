265 int afm_simulate_at_xy(Bimage* ptip, Bimage* p, long ii,
266  float* force, double spring, double modulus, double thick)
267 {
268  long i, ti, di, xx, yy, zz, tx, ty, tz, hiz;
269  long dx, dy, dz, zshift;
270  long ss(p->sizeX()*p->sizeY()), tss(ptip->sizeX()*ptip->sizeY());
271  double fraction, overlap_volume(1);
272  double scale(p->sampling(0).volume()/(p->maximum()*ptip->maximum())); // Adjustment for correct units
273 
274  // Tip displacement = proportion * volume overlap
275  double proportion(scale*modulus/(spring*thick));
276 
277  double* shift = new double[p->sizeZ()];
278 
279  yy = ii/p->sizeX();
280  xx = ii - yy*p->sizeX();
281  for ( zz=0; zz<p->sizeZ(); zz++ ) {
282  shift[zz] = 0;
283  if ( overlap_volume > 1e-30 ) {
284  overlap_volume = 0;
285  i = zz*ss + ii;
286  hiz = ptip->sizeZ();
287  if ( hiz > p->sizeZ() - zz ) hiz = p->sizeZ() - zz;
288  for ( ti=ty=0; ty<ptip->sizeY(); ty++ ) { // Calculate overlap volume
289  dy = yy + ty - (long)ptip->image->origin()[1];
290  while ( dy < 0 ) dy += p->sizeY(); // Wrap y
291  while ( dy >= p->sizeY() ) dy -= p->sizeY();
292  for ( tx=0; tx<ptip->sizeX(); tx++, ti++ ) {
293  dx = xx + tx - (long)ptip->image->origin()[0];
294  while ( dx < 0 ) dx += p->sizeX(); // Wrap x
295  while ( dx >= p->sizeX() ) dx -= p->sizeX();
296  di = dy*p->sizeX()+dx;
297  for ( tz=0; tz<hiz; tz++ ) {
298  dz = zz + tz;
299  overlap_volume += (*p)[dz*ss+di]*(*ptip)[tz*tss+ti];
300  }
301  }
302  }
303  }
304  shift[zz] = overlap_volume*proportion;
305 // cout << "%ld\t%g\n", z, shift[z]);
306  }
307 
308  for ( zz=0; zz<p->sizeZ(); zz++ ) { // Calculate zshift where the forces are equal
309  zshift = 0;
310  while ( zshift*p->sampling(0)[2] < shift[zz+zshift] &&
311  zshift+zz < p->sizeZ() ) zshift++;
312  if ( zshift > 0 ) zshift--;
313  fraction = (zshift*p->sampling(0)[2] - shift[zz+zshift])/
314  (zshift*p->sampling(0)[2] - shift[zz+zshift] -
315  (zshift+1)*p->sampling(0)[2] + shift[zz+zshift+1]);
316  if ( !isfinite(fraction) )
317  cout << "Problem:\t" << xx << tab << yy << tab << zz << tab <<
318  fraction << tab << zshift*p->sampling(0)[2] << tab <<
319  shift[zz+zshift] << tab << shift[zz+zshift+1] << endl;
320  i = zz*ss + ii;
321  force[i] = spring*(zshift + fraction)*p->sampling(0)[2];
322  }
323 // cout << "%ld\t%ld\t%ld\n", x, y, zshift);
324 
325  delete[] shift;
326 
327  return 0;
328 }
