38  resolution = sam[0];
39 
40  long i, h, k, x, y;
41  double wavelength = electron_wavelength(volt);
42  double arg, sx2, sy2, s2;
43  double afactor = M_PI*wavelength*thickness;
44  double nfactor = 1.0/(p->sizeX()*p->sizeY()); // Normalization for passing through FFT's
45  double resmax2 = 1.0/(resolution*resolution);
46  double xscale2 = 1.0/(p->sizeX()*p->sizeX()*sam[0]*sam[0]);
47  double yscale2 = 1.0/(p->sizeY()*p->sizeY()*sam[1]*sam[1]);
48 
49  for ( y=0; y<p->sizeY(); y++ ) {
50  k = y;
51  if ( k > (p->sizeY()-1)/2 ) k -= p->sizeY();
52  sy2 = k*k*yscale2;
53  if ( sy2 <= resmax2 ) for ( x=0; x<p->sizeX(); x++ ) {
54  h = x;
55  if ( h > (p->sizeX()-1)/2 ) h -= p->sizeX();
56  sx2 = h*h*xscale2;
57  s2 = sx2 + sy2;
58  if ( s2 <= resmax2 ) {
59  i = y*p->sizeX() + x;
60  arg = afactor*s2;
61  p->set(i, Complex<double>(cos(arg), sin(arg)) * nfactor);
62  }
63  }
64  }
65

67

252 int img_apply_complex_CTF(Bimage* p, double def_avg, double def_dev, double ast_angle,
253  double volts, double Cs, double Cc, double amp_shift, double alpha, double energy_spread)

258 // double phi_fac = sqrt(1-amp_fac*amp_fac);
259  double lambda = electron_wavelength(volts);
260  double hpil3Cs = M_PI_2*lambda*lambda*lambda*Cs;
261  double pil = M_PI*lambda;
262  double pialf = M_PI*alpha;
263  double Csl2 = Cs*lambda*lambda;
264  double focus_spread = Cc*energy_spread;
265  double pi2l2df2 = M_PI*M_PI*lambda*lambda*focus_spread*focus_spread/(16.0*log(2.0));

274  cout << "Astigmatism angle: " << ast_angle*180/M_PI << " degrees" << endl;
275  cout << "Voltage: " << volts << " V" << endl;
276  cout << "Wavelength: " << lambda << endl;
277  cout << "Cs: " << Cs << " A" << endl;
278  cout << "Cc: " << Cc << " A" << endl;
279  cout << "Amplitude contrast phase shift: " << amp_shift << endl;
280  cout << "Beam source size: " << alpha << " radians" << endl;
281  cout << "Energy spread (focus spread): " << energy_spread << " (" << focus_spread << " A)" << endl << endl;
282  }
283 
284  for ( i=n=0; n<p->images(); n++ ) {
285  for ( z=0; z<p->sizeZ(); z++ ) {
286  zz = z;
287  if ( z > (p->sizeZ() - 1)/2 ) zz -= p->sizeZ();
288  sz2 = freq_scale[2]*zz;
289  sz2 *= sz2;
290  for ( y=0; y<p->sizeY(); y++ ) {
291  yy = y;
292  if ( y > (p->sizeY() - 1)/2 ) yy -= p->sizeY();
293  sy2 = freq_scale[1]*yy;
294  sy2 *= sy2;
295  for ( x=0; x<p->sizeX(); x++, i++ ) {
296  xx = x;
297  if ( x > (p->sizeX() - 1)/2 ) xx -= p->sizeX();
298  sx2 = freq_scale[0]*xx;
299  sx2 *= sx2;
300  s2 = sx2 + sy2 + sz2;
301  s = sqrt(s2);
302  defocus = def_avg + def_dev*cos(2*(atan2(yy,xx)-ast_angle));
303  dphi = (hpil3Cs*s2 - pil*defocus)*s2 - amp_shift;
304  arg = pialf*(Csl2*s2 - defocus)*s;
305 // arg = pialf*defocus*s;
306  arg = arg*arg + pi2l2df2*s2*s2;
307  env = exp(-arg);
308  //env=1;//dphi -= arg*arg;
309  ctf = Complex<double>(cos(dphi), sin(dphi));
310  p->set(i, ((*p)[i] * ctf) * env);
311  }
312  }
313  }
314  }

316  return 0;
