93 {
94  if ( sampling_ratio <= 0 ) sampling_ratio = 1;
95 
96  // The frequency scaling is linked to the different dimensions of the
97  // data set (important when the x, y and z dimensions are different)
98  // The radius scaling is set to a value consistent with one pixel width
99  // in reciprocal space for the longest dimension.
100  Vector3<double> freq_scale(1/real_size());
101  double rad_scale = real_size()[0]/sampling_ratio;
102  if ( hi_res < 2*sampling(0).max() ) hi_res = 2*sampling(0).max();
103  if ( hi_res > real_size().max()/2.0 ) hi_res = real_size().max()/2.0;
104

106

111

113  double radius, fraction, fraction2, I1, I2, Isum, F1F2re, dphi, Id2;
114  double rx, ry, rz;
115

118  cout << "Sampling ratio: " << sampling_ratio << endl << endl;

125  double* DPR = new double[maxrad];
126 
127  for ( i=0; i<maxrad; i++ )
128  F1[i] = F2[i] = Fsum[i] = FSC[i] = DPR[i] = 0;
129 
130  for ( zz=0; zz<z; zz++ ) {
131  rz = zz;
132  if ( rz > (z - 1)/2 ) rz -= z;
133  rz *= freq_scale[2];
134  iz = -zz;
135  if ( iz < 0 ) iz += z;
136  for ( yy=0; yy<y; yy++ ) {
137  ry = yy;
138  if ( ry > (y - 1)/2 ) ry -= y;
139  ry *= freq_scale[1];
140  iy = -yy;
141  if ( iy < 0 ) iy += y;
142  for ( xx=0; xx<x/2; xx++ ) {
143  i = (zz*y + yy)*x + xx;
144  rx = xx;
145  if ( xx > (x - 1)/2 ) rx -= x;
146  rx *= freq_scale[0];
147  ix = -xx;
148  if ( ix < 0 ) ix += x;
149  radius = rad_scale*sqrt(rx*rx + ry*ry + rz*rz);
150  iradius = (long) radius;
151  iradius2 = iradius + 1;
152  if ( iradius2 < maxrad ) {
153  fraction = radius - iradius;
154  fraction2 = 1.0 - fraction;
155  j = (iz*y + iy)*x + ix;
156  sf1 = complex(i).unpack_first(complex(j));
157  sf2 = complex(i).unpack_second(complex(j));
158  I1 = sf1.power();
159  I2 = sf2.power();
160  dphi = angle_set_negPI_to_PI(sf1.phi() - sf2.phi());
161  dphi *= dphi;
162  F1[iradius] += fraction2*I1;
163  F1[iradius2] += fraction*I1;
164  F2[iradius] += fraction2*I2;
165  F2[iradius2] += fraction*I2;
166  F1F2re = sf1.real()*sf2.real() + sf1.imag()*sf2.imag();
167  FSC[iradius] += fraction2*F1F2re;
168  FSC[iradius2] += fraction*F1F2re;
169  Isum = I1 + I2;
170  Fsum[iradius] += fraction2*Isum;
171  Fsum[iradius2] += fraction*Isum;
172  Id2 = Isum*dphi;
173  DPR[iradius] += fraction2*Id2;

176  }
