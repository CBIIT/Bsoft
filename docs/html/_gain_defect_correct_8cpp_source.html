19 void reduce16k_to_4k(const float *gain16k,float *gain4k)
20 {
21  for (int y=0; y<4096; ++y)
22  for (int x=0; x<4096; ++x)
23  {
24  double r = 0;
25  for(int sy=0; sy<4; ++sy)
26  for (int sx=0; sx<4; ++sx)
27  {
28  r += gain16k[(y*4+sy)*16384 + (x*4+sx)];
29  }
30  gain4k[y*4096+x] = r;
31  }
32  std::ofstream ofile("TST4Kred.bin", std::ios::binary);
33  ofile.write((char*)(gain4k), sizeof(float)*4096*4096);
34 }

201 template <typename T>
202 void DefectCorrect(T* img, const CameraDefects& def, unsigned w , unsigned h ) // now only for 4k x 4k
203 {
204  for (auto it = def.vLineDefects.begin(); it != def.vLineDefects.end(); ++it)
205  {
206  int xStart = it->begin;
207  int xStop = it->end;
208 
209  for (int y = 0; y<h; ++y)
210  {
211  T pixLeft = img[y*h + ((xStart>0) ? xStart - 1 : xStop + 1)];
212  T pixRight = img[y*h + ((xStop<(w - 1)) ? xStop + 1 : xStart - 1)];
213  for (int x = xStart; x <= xStop; ++x)
214  {
215  double coef = static_cast<double>(x - xStart + 1) / (xStop - xStart + 2);
216  img[y*w + x] = pixLeft * (1 - coef) + pixRight * coef;
217  }
218  }
219  }
220  for (auto it = def.hLineDefects.begin(); it != def.hLineDefects.end(); ++it)
221  {
222  int yStart = it->begin;
223  int yStop = it->end;
224 
225  for (int x = 0; x<w; ++x)
226  {
227  T pixLeft = img[x + ((yStart>0) ? yStart - 1 : yStop + 1) * 4096];
228  T pixRight = img[x + ((yStop<4095) ? yStop + 1 : yStart - 1) * 4096];
229  for (int y = yStart; y <= yStop; ++y)
230  {
231  double coef = (y - yStart + 1) / (yStop - yStart + 2);
232  img[y*w + x] = pixLeft * (1 - coef) + pixRight * coef;
233  }
234  }
235  }
236  for (auto it = def.pixelDefects.begin(); it != def.pixelDefects.end(); ++it)
237  {
238  int x = it->x;
239  int y = it->y;
240  double sum = 0;
241  int n = 4;
242  if (y > 0) sum += img[(y-1)*w + x]; else --n;
243  if (x > 0) sum += img[y*w + x - 1]; else --n;
244  if (x < w-1) sum += img[y*w + x + 1]; else --n;
245  if (y < h-1) sum += img[(y+1)*w + x]; else --n;
246  img[y*w + x] = sum / n;
247  }
248  for (auto it = def.areaDefects.begin(); it != def.areaDefects.end(); ++it)
249  {
250  int xBef = ((it->beginX) > 0)? it->beginX - 1 : it->endX + 1;
251  int xAft = ((it->beginX) < w-1)? it->endX + 1 : it->beginX - 1;
252  int yBef = ((it->beginY) > 0)? it->beginY - 1 : it->endY + 1;
253  int yAft = ((it->beginY) < h-1)? it->endY + 1 : it->beginY - 1;
254  int nx = 2 * (it->endX - it->beginX + 2);
255  int ny = 2 * (it->endY - it->beginY + 2);
256  for (int y = it->beginY; y <= it->endY; ++y)
257  for (int x = it->beginX; x <= it->endX; ++x)
258  {
259  double coefX = static_cast<double>(x - it->beginX + 1) / nx;
260  double coefY = static_cast<double>(y - it->beginY + 1) / ny;
261  img[y*w + x] = (img[yBef*w + x] * (0.5 - coefY) + img[yAft*w + x] * coefY + img[y*w + xBef] * (0.5 - coefX) + img[y*w + xAft] * coefX);
262  }
263  }
