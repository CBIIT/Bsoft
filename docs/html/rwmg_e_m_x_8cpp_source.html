500 /*
501 @brief Converts micrograph data from an project structure to a XML document.
502 @param Bproject* project parameter structure.
503 @param xmlDocPtr doc XML document.
504 @param mg_select flag to only write selected micrographs.
505 @param rec_select flag to only convert selected reconstructions.
506 @return int error code (<0 means failure).
507

509  document from which the old project parameters were obtained.
510  All angles used internally are in radians, and are converted here
511  to degrees for output. The angles within the micrograph structure
512  are left in radians.

530

537

539  xmlNodePtr part_node = NULL;
540  xmlNodePtr node = NULL;
541 
542  double angle;
543  Vector3<double> center;

545 
546  for ( field=project->field; field; field=field->next ) {

549  else filename = "dummy.mrc";

551  xmlNewProp(mg_node, BAD_CAST "fileName", BAD_CAST filename.c_str());
552  xml_set_integer_attribute(mg_node, "index", mg->img_num+1, "%d");
553 
554  center = micrograph_get_nominal_origin(mg);
555 
556  emx_set_real_vector2(mg_node, "pixelSpacing", mg->pixel_size[0], mg->pixel_size[1], "A/px");
557 
558  emx_set_transformation(mg_node, mg->matrix, mg->origin - center);
559 
560  if ( mg->ctf ) {
561  emx_set_real_unit(mg_node, "acceleratingVoltage", mg->ctf->volt()/1e3, "kV");
562  emx_set_real_unit(mg_node, "amplitudeContrast", sin(mg->ctf->amp_shift()), NULL);
563  emx_set_real_unit(mg_node, "cs", mg->ctf->Cs()/1e7, "mm");
564  emx_set_real_unit(mg_node, "defocusU", (mg->ctf->defocus_average() + mg->ctf->defocus_deviation())/10, "nm");
565  emx_set_real_unit(mg_node, "defocusV", (mg->ctf->defocus_average() - mg->ctf->defocus_deviation())/10, "nm");
566  angle = mg->ctf->astigmatism_angle()*180.0/M_PI;
567  if ( angle < 0 ) angle += 180;
568  emx_set_real_unit(mg_node, "defocusUAngle", angle, "deg");
569  }
570 
571  xml_set_integer(mg_node, "activeFlag", mg->select, "%d");
572  xml_set_real(mg_node, "fom", mg->fom, "%g");
573  }
574  }

577  for ( mg=field->mg; mg; mg=mg->next ) {
578  if ( mg->fmg.length() ) filename = mg->fmg;
579  else filename = "dummy.mrc";
580  center = mg->box_size/2;
581  for ( part=mg->part; part; part=part->next ) {
582  part_node = xmlNewChild(emx_node, NULL, BAD_CAST "particle", NULL);
583  if ( part->fpart.length() )
584  xmlNewProp(part_node, BAD_CAST "fileName", BAD_CAST part->fpart.c_str());
585  else
586  xmlNewProp(part_node, BAD_CAST "fileName", BAD_CAST mg->fpart.c_str());
587  xml_set_integer_attribute(part_node, "index", part->id, "%d");
588 
589  node = xmlNewChild(part_node, NULL, BAD_CAST "micrograph", NULL);
590  xmlNewProp(node, BAD_CAST "fileName", BAD_CAST filename.c_str());
591  xml_set_integer_attribute(node, "index", mg->img_num+1, "%d");
592 
593  emx_set_real_vector2(part_node, "pixelSpacing", part->pixel_size[0], part->pixel_size[1], "A/px");
594 
595  emx_set_real_vector2(part_node, "boxSize", mg->box_size[0], mg->box_size[1], "px");
596  emx_set_real_vector2(part_node, "centerCoord", part->loc[0], part->loc[1], "px");
597 
598  emx_set_transformation(part_node, part->view, part->ori - center);
599
