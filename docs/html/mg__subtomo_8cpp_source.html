838

845  double p_to_c2; // squared distance from p to the circumference of the disc c
846

856  p_to_pprime2 = (p-pprime).length2();
857 
858  if (pprime_to_q2 > radius2) {
859

861  pprime_to_c2 = (pprime_to_q - radius) * (pprime_to_q - radius);
862 
863  p_to_c2 = pprime_to_c2 + p_to_pprime2;
864 
865  return p_to_c2;
866  }

907 Sphere locations_fit_sphere(Bparticle* part, int N, double Nstop)

922

924

948  PY2Zsum += y * y * z;
949  PZ2Xsum += z * z * x;

954  double Xn2 = PXsumsq/Pnpoints; //sum( X[n]^2 )
955  double Xn3 = PXsumcube/Pnpoints; //sum( X[n]^3 )

957  double Yn2 = PYsumsq/Pnpoints; //sum( Y[n]^2 )
958  double Yn3 = PYsumcube/Pnpoints; //sum( Y[n]^3 )
959  double Zn = PZsum/Pnpoints; //sum( Z[n] )
960  double Zn2 = PZsumsq/Pnpoints; //sum( Z[n]^2 )

962

964  double XZ = PXZsum/Pnpoints; //sum( X[n] * Z[n] )
965  double YZ = PYZsum/Pnpoints; //sum( Y[n] * Z[n] )
966  double X2Y = PX2Ysum/Pnpoints; //sum( X[n]^2 * Y[n] )
967  double X2Z = PX2Zsum/Pnpoints; //sum( X[n]^2 * Z[n] )
968  double Y2X = PY2Xsum/Pnpoints; //sum( Y[n]^2 * X[n] )
969  double Y2Z = PY2Zsum/Pnpoints; //sum( Y[n]^2 * Z[n] )
970  double Z2X = PZ2Xsum/Pnpoints; //sum( Z[n]^2 * X[n] )
971  double Z2Y = PZ2Ysum/Pnpoints; //sum( Z[n]^2 * Y[n] )
972 
973  //Reduction of multiplications
974  double F0 = Xn2 + Yn2 + Zn2;
975  double F1 = 0.5*F0;
976  double F2 = -8.0*(Xn3 + Y2X + Z2X);
977  double F3 = -8.0*(X2Y + Yn3 + Z2Y);
978  double F4 = -8.0*(X2Z + Y2Z + Zn3);
979 
980  //Set initial conditions:
981  double A = Xn;
982  double B = Yn;
983  double C = Zn;
984 
985  //First iteration computation:
986  double A2 = A*A;
987  double B2 = B*B;
988  double C2 = C*C;
989  double QS = A2 + B2 + C2;
990  double QB = - 2*(A*Xn + B*Yn + C*Zn);
991 
992  //Set initial conditions:
993  double Rsq = F0 + QB + QS;
994 
995  //First iteration computation:
996  double Q0 = 0.5*(QS - Rsq);
997  double Q1 = F1 + Q0;
998  double Q2 = 8*( QS - Rsq + QB + F0 );
999  double aA,aB,aC,nA,nB,nC,dA,dB,dC;
1000 
1001  //Iterate N times, ignore stop condition.
1002  int n(0);
1003  while( n != N ){
1004  n++;
1005 
1006  //Compute denominator:
1007  aA = Q2 + 16*(A2 - 2*A*Xn + Xn2);
1008  aB = Q2 + 16*(B2 - 2*B*Yn + Yn2);
1009  aC = Q2 + 16*(C2 - 2*C*Zn + Zn2);
1010  aA = (aA == 0) ? 1.0 : aA;
1011  aB = (aB == 0) ? 1.0 : aB;
1012  aC = (aC == 0) ? 1.0 : aC;
1013 
1014 
1015  //Compute next iteration
1016  nA = A - ((F2 + 16*( B*XY + C*XZ + Xn*(-A2 - Q0) + A*(Xn2 + Q1 - C*Zn - B*Yn) ) )/aA);
1017  nB = B - ((F3 + 16*( A*XY + C*YZ + Yn*(-B2 - Q0) + B*(Yn2 + Q1 - A*Xn - C*Zn) ) )/aB);
1018  nC = C - ((F4 + 16*( A*XZ + B*YZ + Zn*(-C2 - Q0) + C*(Zn2 + Q1 - A*Xn - B*Yn) ) )/aC);
1019 
1020  //Check for stop condition
1021  dA = (nA - A);
1022  dB = (nB - B);
1023  dC = (nC - C);
1024  if( (dA*dA + dB*dB + dC*dC) <= Nstop ){ break; }
1025 
1026  //Compute next iteration's values
1027  A = nA;
1028  B = nB;
1029  C = nC;
1030  A2 = A*A;
1031  B2 = B*B;
1032  C2 = C*C;
1033  QS = A2 + B2 + C2;
1034  QB = - 2*(A*Xn + B*Yn + C*Zn);
1035  Rsq = F0 + QB + QS;
1036  Q0 = 0.5*(QS - Rsq);
1037  Q1 = F1 + Q0;
1038  Q2 = 8*( QS - Rsq + QB + F0 );
1039  }
1040 
1041  double R = sqrt(Rsq);
1042  sphere.A = A;
1043  sphere.B = B;
1044  sphere.C = C;
1045  sphere.R = R;
1046 
1047  return (sphere);
