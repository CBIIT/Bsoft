420 ## @brief Returns all the zeroes at the average defocus
421 
422 proc getZeroes { } {
423  global project_item theimg
424  set wc [getControlWindow $theimg]
425  set Cs [expr 1e7 * [.wctf.vca.cs_entry get]]
426  set lambda [getWavelength]
427 # puts "Wavelength $lambda angstrom"
428  set invl2Cs [expr 1.0 / ($lambda * $lambda * $Cs)]
429  set inv2l3Cs [expr 2.0 / ($lambda * $lambda * $lambda * $Cs)]
430  set def_avg [expr 1e4 * [.wctf.def_avg_scale get]]
431  set ctf_fz [expr $def_avg * $invl2Cs]
432 # set max_s [expr 0.5 / [.pixel_size.entry get]]
433  set max_s [expr 0.5 / [$wc.pixel_size.x get]]
434  set nz [expr lambda*def_avg*max_s*max_s - $lambda*$lambda*$lambda*$Cs*max_s*max_s*max_s*max_s/2]
435  if { $nz < 1 } { set nz 1 }
436  set zeroes ""
437  for {set i 0} {$i < $nz} {incr i 1} {
438  set nt [expr ($i+1)*$inv2l3Cs]
439  set ctf_fz2 [expr $ctf_fz * $ctf_fz]
440  if { $ctf_fz2 > $nt } {
441  append zeroes " " [expr sqrt(ctf_fz - sqrt(ctf_fz2 - $nt))]
442  }
443  }
444  return $zeroes
445 }

472 ## @brief Calculates the contrast transfer function
473 #
474 
475 proc calcCTF { } {
476  global PI step
477  global base envl
478  global rps ctf
479  set Cs [expr 1e7 * [.wctf.vca.cs_entry get]]
480  set ampfac [.wctf.vca.amp_entry get]
481  set phifac [expr sqrt(1 - $ampfac * $ampfac)]
482  set base_eq [.wctf.base_eq.entry get]
483  set env_eq [.wctf.env_eq.entry get]
484  set def_avg [expr 1e4 * [.wctf.def_avg_scale get]]
485  set lambda [getWavelength]
486  set hpil3Cs [expr 0.5 * $PI * $lambda * $lambda * $lambda * $Cs];
487  set pil [expr $PI * $lambda];
488  set n [llength $rps]
489  set base ""
490  set envl ""
491  set ctf ""
492 # puts $base_eq
493 # puts $env_eq
494  for {set i 0} {$i < $n} {incr i 1} {
495  set s [expr $i * $step]
496  set s2 [expr $s * $s]
497  set s3 [expr $s * $s2]
498  set s4 [expr $s2 * $s2]
499  set dphi [expr $hpil3Cs*$s4 - $pil*$def_avg*$s2]
500  set amp [expr $ampfac*cos($dphi) - $phifac*sin($dphi)]
501  set e [expr $env_eq]
502  set b [expr $base_eq]
503  set c [expr $amp * $amp]
504  append base " " $b
505  append envl " " $e
506  append ctf " " $c
507  }
508 }

510 ## @brief Returns the RMSD between the experimental and calculated RPS curves.
511 
512 proc calcR { } {
513  global theimg
514  global step base envl
515  global ctf rps
516  global ctf_hires ctf_lores
517  set wc [getControlWindow $theimg]
518  set pixel_size [$wc.pixel_size.x get]
519  set lores [.wctf.pp.lores_entry get]
520  set hires [.wctf.pp.hires_entry get]
521  if { $hires < [expr 2.1 * $pixel_size] } {
522  set hires [expr 2.1 * $pixel_size]
523  .wctf.pp.hires_entry delete 0 end
524  .wctf.pp.hires_entry insert 0 $hires
525  set ctf_hires $hires
526  }
527  if { $lores < $hires } {
528  set lores [expr 2 * $hires]
529  .wctf.pp.lores_entry delete 0 end
530  .wctf.pp.lores_entry insert 0 $lores
531  set ctf_lores $lores
532  }
533  set nr [llength $rps]
534  set ilo [expr int(1.0 / ($lores*$step))]
535  set ihi [expr int(1.0 / ($hires*$step))]
536  if { $ihi <= $ilo } { set ihi [expr $ilo + 1] }
537  if { $ilo < 1 } { set ilo 1 }
538  if { $ihi >= $nr } { set ihi [expr $nr - 1] }
539  set n 0.0
540  set R 0.0
541  for {set i $ilo} {$i <= $ihi} {incr i 1} {
542  set c [expr [lindex $base $i] + [lindex $envl $i] * [lindex $ctf $i]]
543  set d [expr [lindex $rps $i] - $c]
544  set w 1.0
545 # set w [expr 1.0 - [lindex $ctf $i]]
546  if { $w > 0 } {
547  set R [expr $R + $d * $d * $w]
548  set n [expr $n + $w]
549  }
550  }
551  if { $n > 0 } {
552  set R [expr sqrt($R/$n)]
553  } else {
554  set R 1000.0
555  }
556  return $R
557 }

568  setCTFparam

570  set hi_res [$w.pp.hires_entry get]
571  set def_start [expr [$w.def.start get] * 1e4]
572  set def_end [expr [$w.def.end get] * 1e4]
573  set def_inc [expr [$w.def.inc get] * 1e4]

575 # puts "base_type=$base_type"
576  set ps [Bmg get $mg_item pixel_size]
577  if { [lindex $ps 0] > 1.5 } { # Make sure we are not trying to fit the water ring
578  if { $base_type > 3 } {
579  set base_type [expr $base_type - 3]
580  }
581  }
582  set old_env_type [Bmg get $mg_item envelope_type]
583  Bmg set $mg_item baseline_type $base_type
584  Bmg set $mg_item envelope_type $env_type
585  set result [Bmg ctf_fit $mg_item $level $lo_res $hi_res $def_start $def_end $def_inc]
586  if { $level == 1 && $env_type != $old_env_type } {
587  set result [Bmg ctf_fit $mg_item 2 $lo_res $hi_res]
588  }
589 # puts $result
590  if { [string length $result] < 2 } { return }
591  $w.base_eq.entry delete 0 end
592  $w.base_eq.entry insert 0 [lindex $result 0]
593  $w.env_eq.entry delete 0 end
594  $w.env_eq.entry insert 0 [lindex $result 1]
595  $w.def_avg_scale set [expr 1e-4 * [lindex $result 2]]
596  $w.def_dev_scale set [expr 1e-4 * [lindex $result 3]]
597  $w.ast_ang_scale set [lindex $result 4]
598  if { $level == 4 } { set cont_update 1 }
599  updateCTF

602 ## @brief Draws a number of ellipses at the CTF zeroes.

606 proc drawEllipse { theimg } {
607  global PI show_rings step
608  set c [getImageCanvas $theimg]
609  $c delete ellipse
610  if { $show_rings < 1 } { return }
611  set wc [getControlWindow $theimg]
612  set pixel_size [$wc.pixel_size.x get]
613  set Cs [expr 1e7 * [.wctf.vca.cs_entry get]]
614  set phi [expr asin([.wctf.vca.amp_entry get]) / $PI]
615  set lambda [getWavelength]
616 # puts "Wavelength $lambda angstrom"
617  set invl2Cs [expr 1.0 / ($lambda * $lambda * $Cs)]
618  set inv2l3Cs [expr 2.0 / ($lambda * $lambda * $lambda * $Cs)]
619  set width [Bimage get $theimg width]
620  set height [Bimage get $theimg height]
621  set scale [$wc.scale.scale get]
622 # set originx [expr $scale * ($width + 1) / 2]
623 # set originy [expr $scale * ($height + 2) / 2]
624  set originx [expr $scale * $width / 2]
625  set originy [expr $scale * $height / 2]
626  set def_avg [expr 1e4 * [.wctf.def_avg_scale get]]
627  set def_dev [expr 1e4 * [.wctf.def_dev_scale get]]
628  set ast_ang [expr [.wctf.ast_ang_scale get] * $PI / 180]
629  if { $def_dev > [expr 0.99 * $def_avg] } {
630  set def_dev [expr 0.99 * $def_avg]
631  .wctf.def_dev_scale set [expr $def_dev * 1e-4]
632  }
633  set ellip2 [expr ( $def_avg + $def_dev ) / ( $def_avg - $def_dev ) ]
634  set fz [expr $def_avg * $invl2Cs]
635  set fz2 [expr $fz * $fz]
636  set nmax [expr int($lambda * $def_avg / (4 * $pixel_size * $pixel_size))]
637 # puts $nmax
638  if { $nmax > 10 } { set nmax 10 }
639  set astart [expr $PI/2]
640  set aend [expr $astart + $show_rings * $PI + 0.05]
641  set astep [expr $PI/36]
642  for { set n 1 } { $n < $nmax } { incr n 1 } {
643  set r $width
644  set yscale [expr $height*1.0/$width]
645  set t [expr ($n - $phi) * $inv2l3Cs]
646  if { $fz2 > $t } {
647  set r [expr ($scale / $step) * sqrt($fz - sqrt($fz2 - $t))]
648  }
649 # puts "Pixel distance $r"
650  set a $astart
651  set da [expr $a - $ast_ang]
652  set re [expr $r * sqrt( ($ellip2 + 1) / \
653  (2 * ($ellip2 * cos($da) * cos($da) + sin($da) * sin($da))) ) ]
654  set xl [expr $re*cos($a) + $originx]
655  set yl [expr $re*$yscale*sin($a) + $originy]
656  set yl [expr $scale * $height - $yl]
657  for {set a $astart} {$a <= $aend} {set a [expr $a + $astep]} {
658  set da [expr $a - $ast_ang]
659  set re [expr $r * sqrt( ($ellip2 + 1) / \
660  (2 * ($ellip2 * cos($da) * cos($da) + sin($da) * sin($da))) ) ]
661  set x [expr $re*cos($a) + $originx]
662  set y [expr $re*$yscale*sin($a) + $originy]
663  # y is flipped because the canvas origin is upper left
664  set y [expr $scale * $height - $y]
665  $c create line $xl $yl $x $y -dash 2 -fill yellow -smooth yes -width 1 -tags ellipse
666  set xl $x
667  set yl $y
668  }
669  }
670 }
