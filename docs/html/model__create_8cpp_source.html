863 double ellipse_next_x(Vector3<double> axes, double x1, double distance)
864 {
865  double distance2 = distance*distance;
866  double temp = axes[1] * x1 / axes[0];

885  return left; //if left doesn't bracket the root, it is the root
886

894  left = mid[0];

896  right = mid[0];
