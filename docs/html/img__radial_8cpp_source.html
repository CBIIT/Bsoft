677 /*
678 @brief Least squares function for the img_radial_fit function.
679 @param Bsimplex* simp a simplex structure.
680 @return double the least squares R value.
681 
682  A radial profile is fitted to a reference profile using an iterative
683  simplex down-hill method. The equations solved are:
684  xref = x*m
685  yref(xref) = y(x)*a + b
686  where m is the dimension scaling (or magnification).
687  a is the amplitude scaling.
688  b is the amplitude shift.
689  The least squares function is:
690  sum(weight*(yref(xref) - y(x)*a - b)^2)
691  R = sqrt(---------------------------------------)
692  sum(weight)
693 **/
694 double img_radial_R(Bsimplex* simp)
695 {
696  float* y = simp->fx;
697  float* yref = simp->x;
698 
699  int i, iref, n(0);
700  double fraction, xref, df, R(0), sum(0), weight;
701 
702  for ( i=0; i<simp->npoint; i++ ) {
703  xref = simp->param[0]*i;
704  iref = (int) xref;
705  if ( iref >= 0 && iref < simp->npoint - 1 ) {
706  fraction = xref - iref;
707  df = (1.0 - fraction)*yref[iref] + fraction*yref[iref + 1]
708  - simp->param[1]*y[i] - simp->param[2];
709 // weight = i*(fabs(yref[iref]) + fabs(y[i]));
710  weight = i;
711 // weight = fabs(y[i]);
712  sum += weight;
713  R += df*df*weight;
714  n++;
715  }
716  }
717  if ( sum && n > 0.2*simp->npoint ) R = sqrt(R/sum);
718  else R = 1e30; // Out of range
719 
720  if ( verbose & VERB_DEBUG )
721  cout << "DEBUG img_radial_R: " << simp->param[0] << " " <<
722  simp->param[1] << " " << simp->param[2] << " R=" << R << endl;
723 
724  return R;
725 }
