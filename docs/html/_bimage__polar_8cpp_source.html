261 /*
262 @brief Least squares function for the radial_fit function.
263 @param &simp a simplex structure.
264 @return double the least squares R value.
265 
266  A radial profile is fitted to a reference profile using an iterative
267  simplex down-hill method. The equations solved are:
268  xref = x*m
269  yref(xref) = y(x)*a + b
270  where m is the dimension scaling (or magnification).
271  a is the amplitude scaling.
272  b is the amplitude shift.
273  The least squares function is:
274  sum(weight*(yref(xref) - y(x)*a - b)^2)
275  R = sqrt(---------------------------------------)
276  sum(weight)
277 **/
278 double radial_R(Bsimplex& simp)
279 {
280  vector<double>& y = simp.dependent_values();
281  vector<double>& yref = simp.independent_values();
282 
283  int i, iref, n(0);
284  double fraction, xref, df, R(0), sum(0), weight;
285 
286  for ( i=0; i<simp.points(); i++ ) {
287  xref = simp.parameter(0)*i;
288  iref = (int) xref;
289  if ( iref >= 0 && iref < simp.points() - 1 ) {
290  fraction = xref - iref;
291  df = (1.0 - fraction)*yref[iref] + fraction*yref[iref + 1]
292  - simp.parameter(1)*y[i] - simp.parameter(2);
293 // weight = i*(fabs(yref[iref]) + fabs(y[i]));
294  weight = i;
295 // weight = fabs(y[i]);
296  sum += weight;
297  R += df*df*weight;
298  n++;
299  }
300  }
301  if ( sum && n > 0.2*simp.points() ) R = sqrt(R/sum);
302  else R = 1e30; // Out of range
303 
304  if ( verbose & VERB_DEBUG )
305  cout << "DEBUG radial_R: " << simp.parameter(0) << " " <<
306  simp.parameter(1) << " " << simp.parameter(2) << " R=" << R << endl;
307 
308  return R;
309 }
